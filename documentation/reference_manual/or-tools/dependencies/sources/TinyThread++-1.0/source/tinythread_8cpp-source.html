00074 #if defined(_TTHREAD_WIN32_)
00075 void condition_variable::_wait()
00076 {
00077   // Wait for either event to become signaled due to notify_one() or
00078   // notify_all() being called
00079   int result = WaitForMultipleObjects(2, mEvents, FALSE, INFINITE);
00080 
00081   // Check if we are the last waiter
00082   EnterCriticalSection(&mWaitersCountLock);
00083   -- mWaitersCount;
00084   bool lastWaiter = (result == (WAIT_OBJECT_0 + _CONDITION_EVENT_ALL)) &&
00085                     (mWaitersCount == 0);
00086   LeaveCriticalSection(&mWaitersCountLock);
00087 
00088   // If we are the last waiter to be notified to stop waiting, reset the event
00089   if(lastWaiter)
00090     ResetEvent(mEvents[_CONDITION_EVENT_ALL]);
00091 }

00124 // POSIX pthread_t to unique thread::id mapping logic.
00125 // Note: Here we use a global thread safe std::map to convert instances of
00126 // pthread_t to small thread identifier numbers (unique within one process).
00127 // This method should be portable across different POSIX implementations.
00128 //------------------------------------------------------------------------------

00130 #if defined(_TTHREAD_POSIX_)
00131 static thread::id _pthread_t_to_ID(const pthread_t &aHandle)
00132 {
00133   static mutex idMapLock;
00134   static std::map<pthread_t, unsigned long int> idMap;
00135   static unsigned long int idCount(1);
00136 
00137   lock_guard<mutex> guard(idMapLock);
00138   if(idMap.find(aHandle) == idMap.end())
00139     idMap[aHandle] = idCount ++;
00140   return thread::id(idMap[aHandle]);
00141 }

00156 // Thread wrapper function.
00157 #if defined(_TTHREAD_WIN32_)
00158 unsigned WINAPI thread::wrapper_function(void * aArg)
00159 #elif defined(_TTHREAD_POSIX_)
00160 void * thread::wrapper_function(void * aArg)
00161 #endif
00162 {
00163   // Get thread startup information
00164   _thread_start_info * ti = (_thread_start_info *) aArg;
00165 
00166   try
00167   {
00168     // Call the actual client thread function
00169     ti->mFunction(ti->mArg);
00170   }
00171   catch(...)
00172   {
00173     // Uncaught exceptions will terminate the application (default behavior
00174     // according to the C++0x draft)
00175     std::terminate();
00176   }
00177 
00178   // The thread is no longer executing
00179   lock_guard<mutex> guard(ti->mThread->mDataMutex);
00180   ti->mThread->mNotAThread = true;
00181 
00182   // The thread is responsible for freeing the startup information
00183   delete ti;
00184 
00185   return 0;
00186 }

00188 thread::thread(void (*aFunction)(void *), void * aArg)
00189 {
00190   // Serialize access to this thread structure
00191   lock_guard<mutex> guard(mDataMutex);
00192 
00193   // Fill out the thread startup information (passed to the thread wrapper,
00194   // which will eventually free it)
00195   _thread_start_info * ti = new _thread_start_info;
00196   ti->mFunction = aFunction;
00197   ti->mArg = aArg;
00198   ti->mThread = this;
00199 
00200   // The thread is now alive
00201   mNotAThread = false;
00202 
00203   // Create the thread
00204 #if defined(_TTHREAD_WIN32_)
00205   mHandle = (HANDLE) _beginthreadex(0, 0, wrapper_function, (void *) ti, 0, &mWin32ThreadID);
00206 #elif defined(_TTHREAD_POSIX_)
00207   if(pthread_create(&mHandle, NULL, wrapper_function, (void *) ti) != 0)
00208     mHandle = 0;
00209 #endif
00210 
00211   // Did we fail to create the thread?
00212   if(!mHandle)
00213   {
00214     mNotAThread = true;
00215     delete ti;
00216   }
00217 }
