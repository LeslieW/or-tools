00142 // Core iteration function
00143 Pixel Iterate(const double &cre, const double &cim, int aIterMax)
00144 {
00145   double zre = 0.0;
00146   double zim = 0.0;
00147   int n = 0;
00148   double absZSqr = 0.0;
00149   while((absZSqr < 4.0) && (n < aIterMax))
00150   {
00151     double tmp = zre * zre - zim * zim + cre;
00152     zim = 2.0 * zre * zim + cim;
00153     zre = tmp;
00154     absZSqr = zre * zre + zim * zim;
00155     ++ n;
00156   }
00157   if(n >= aIterMax)
00158     return Pixel(0,0,0);
00159   else
00160   {
00161     double nSmooth = n + 1 - log(log(sqrt(absZSqr))) / log(2.0);
00162     return Pixel(
00163              (unsigned char)(128.0 - 127.0 * cos(0.02 * nSmooth + 0.3)),
00164              (unsigned char)(128.0 - 127.0 * cos(0.016 * nSmooth + 1.2)),
00165              (unsigned char)(128.0 - 127.0 * cos(0.013 * nSmooth + 2.6))
00166            );
00167   }
00168 }

00170 // Calculation thread
00171 void CalcThread(void * arg)
00172 {
00173   RowDispatcher * dispatcher = (RowDispatcher *) arg;
00174   Image * img = dispatcher->GetImage();
00175 
00176   // Set min/max interval for the fractal set
00177   double xMin = MAND_MID_RE - MAND_SIZE * 0.5;
00178   double yMin = MAND_MID_IM - MAND_SIZE * 0.5;
00179   double xMax = MAND_MID_RE + MAND_SIZE * 0.5;
00180   double yMax = MAND_MID_IM + MAND_SIZE * 0.5;
00181 
00182   // Set step size (distance between two adjacent pixels)
00183   double xStep = (xMax - xMin) / img->Width();
00184   double yStep = (yMax - yMin) / img->Height();
00185 
00186   // Until no more rows to process...
00187   while(true)
00188   {
00189     // Get the next row to calculate
00190     int y = dispatcher->NextRow();
00191 
00192     // Done?
00193     if(y < 0)
00194       break;
00195 
00196     // Generate one row of the image
00197     Pixel * line = &(*img)[y * img->Width()];
00198     double cim = y * yStep + yMin;
00199     double cre = xMin;
00200     for(int x = 0; x < img->Width(); ++ x)
00201     {
00202       *line ++ = Iterate(cre, cim, MAND_MAX_ITER);
00203       cre += xStep;
00204     }
00205   }
00206 }
