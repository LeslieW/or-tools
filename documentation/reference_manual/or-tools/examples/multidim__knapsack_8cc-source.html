00249 int64 EvaluateItem(MultiDimKnapsackData* const data, int64 var, int64 val) {
00250   if (val == 0) {
00251     return 0LL;
00252   }
00253   const int profit = data->profit(var);
00254   int max_weight = 0;
00255   for (int i = 0; i < data->dims(); ++i) {
00256     const int weight = data->weight(i, var);
00257     if (weight > max_weight) {
00258       max_weight = weight;
00259     }
00260   }
00261   return -(profit * 100 / max_weight);
00262 }

00264 void SolveKnapsack(MultiDimKnapsackData* const data) {
00265   Solver solver("MultiDim Knapsack");
00266   std::vector<IntVar*> assign;
00267   solver.MakeBoolVarArray(data->items(), "assign", &assign);
00268   for (int i = 0; i < data->dims(); ++i) {
00269     const int capacity = data->capacity(i);
00270     std::vector<int64> coefs;
00271     for (int j = 0; j < data->items(); ++j) {
00272       coefs.push_back(data->weight(i, j));
00273     }
00274     solver.AddConstraint(
00275         solver.MakeScalProdLessOrEqual(assign, coefs, capacity));
00276   }
00277 
00278   // Build objective.
00279   std::vector<int64> profits;
00280   for (int i = 0; i < data->items(); ++i) {
00281     profits.push_back(data->profit(i));
00282   }
00283 
00284   IntVar* const objective = solver.MakeScalProd(assign, profits)->Var();
00285 
00286   std::vector<SearchMonitor*> monitors;
00287   OptimizeVar* const objective_monitor = solver.MakeMaximize(objective, 1);
00288   monitors.push_back(objective_monitor);
00289   SearchMonitor* const search_log = solver.MakeSearchLog(1000000, objective);
00290   monitors.push_back(search_log);
00291   DecisionBuilder* const db =
00292       solver.MakePhase(assign,
00293                        NewPermanentCallback(&EvaluateItem, data),
00294                        Solver::CHOOSE_STATIC_GLOBAL_BEST);
00295   if (FLAGS_time_limit_in_ms != 0) {
00296     LOG(INFO) << "adding time limit of " << FLAGS_time_limit_in_ms << " ms";
00297     SearchLimit* const limit = solver.MakeLimit(FLAGS_time_limit_in_ms,
00298                                                 kint64max,
00299                                                 kint64max,
00300                                                 kint64max);
00301     monitors.push_back(limit);
00302   }
00303 
00304   if (FLAGS_simplex_frequency > 0) {
00305     SearchMonitor* const simplex =
00306         solver.MakeSimplexConstraint(FLAGS_simplex_frequency);
00307     monitors.push_back(simplex);
00308   }
00309 
00310   if (solver.Solve(db, monitors)) {
00311     LOG(INFO) << "Best solution found = " << objective_monitor->best();
00312   }
00313 }
