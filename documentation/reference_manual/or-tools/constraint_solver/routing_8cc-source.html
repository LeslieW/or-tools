00298   const int64 prev = prevs_[BaseNode(0)];
00299   if (prev < 0) {
00300     return false;
00301   }

00396                             const Assignment* deltadelta) {
00397   const Assignment::IntContainer& container = delta->IntVarContainer();
00398   const int delta_size = container.Size();
00399   // Determining touched paths. Number of touched paths should be very small
00400   // given the set of available operators (1 or 2 paths), so performing
00401   // a linear search to find an element is faster than using a set.
00402   std::vector<int64> touched_paths;
00403   for (int i = 0; i < delta_size; ++i) {
00404     const IntVarElement& new_element = container.Element(i);

00410           && find(touched_paths.begin(), touched_paths.end(), start)
00411           == touched_paths.end()) {

00413       }
00414     }
00415   }

00417   const int touched_paths_size = touched_paths.size();
00418   for (int i = 0; i < touched_paths_size; ++i) {

00992 void RoutingModel::SetStartEnd(
00993     const std::vector<std::pair<NodeIndex, NodeIndex> >& start_end) {
00994   if (is_depot_set_) {
00995     LOG(WARNING) << "A depot has already been specified, ignoring new ones";
00996     return;
00997   }
00998   CHECK_EQ(start_end.size(), vehicles_);
00999   const int size = Size();
01000   hash_set<NodeIndex> starts;
01001   hash_set<NodeIndex> ends;
01002   for (int i = 0; i < vehicles_; ++i) {
01003     const NodeIndex start = start_end[i].first;
01004     const NodeIndex end = start_end[i].second;
01005     CHECK_GE(start, 0);
01006     CHECK_GE(end, 0);
01007     CHECK_LE(start, nodes_);
01008     CHECK_LE(end, nodes_);
01009     starts.insert(start);
01010     ends.insert(end);
01011   }
01012   index_to_node_.resize(size + vehicles_);
01013   node_to_index_.resize(nodes_, kUnassigned);
01014   int index = 0;
01015   for (NodeIndex i = kFirstNode; i < nodes_; ++i) {
01016     if (starts.count(i) != 0 || ends.count(i) == 0) {
01017       index_to_node_[index] = i;
01018       node_to_index_[i] = index;
01019       ++index;
01020     }
01021   }
01022   hash_set<NodeIndex> node_set;
01023   index_to_vehicle_.resize(size + vehicles_, kUnassigned);
01024   for (int i = 0; i < vehicles_; ++i) {
01025     const NodeIndex start = start_end[i].first;
01026     if (node_set.count(start) == 0) {
01027       node_set.insert(start);
01028       const int start_index = node_to_index_[start];
01029       starts_[i] = start_index;
01030       CHECK_NE(kUnassigned, start_index);
01031       index_to_vehicle_[start_index] = i;
01032     } else {
01033       starts_[i] = index;
01034       index_to_node_[index] = start;
01035       index_to_vehicle_[index] = i;
01036       ++index;
01037     }
01038   }
01039   for (int i = 0; i < vehicles_; ++i) {
01040     NodeIndex end = start_end[i].second;
01041     index_to_node_[index] = end;
01042     ends_[i] = index;
01043     CHECK_LE(size, index);
01044     index_to_vehicle_[index] = i;
01045     ++index;
01046   }
01047   for (int i = 0; i < size; ++i) {
01048     for (int j = 0; j < vehicles_; ++j) {
01049       // "start" node: nexts_[i] != start
01050       solver_->AddConstraint(solver_->MakeNonEquality(nexts_[i], starts_[j]));
01051     }
01052     // Extra constraint to state a node can't point to itself
01053     solver_->AddConstraint(solver_->MakeIsDifferentCstCt(nexts_[i],
01054                                                          i,
01055                                                          active_[i]));
01056   }
01057   is_depot_set_ = true;
01058 
01059   // Logging model information.
01060   VLOG(1) << "Number of nodes: " << nodes_;
01061   VLOG(1) << "Number of vehicles: " << vehicles_;
01062   for (int index = 0; index < index_to_node_.size(); ++index) {
01063     VLOG(1) << "Variable index " << index
01064             << " -> Node index " << index_to_node_[index];
01065   }
01066   for (NodeIndex node = kFirstNode; node < node_to_index_.size(); ++node) {
01067     VLOG(2) << "Node index " << node
01068             << " -> Variable index " << node_to_index_[node];
01069   }
01070 }

01072 void RoutingModel::CloseModel() {
01073   if (closed_) {
01074     LOG(WARNING) << "Model already closed";
01075     return;
01076   }
01077   closed_ = true;
01078 
01079   CheckDepot();
01080 
01081   AddNoCycleConstraintInternal();
01082 
01083   const int size = Size();
01084 
01085   // Vehicle variable constraints
01086   for (int i = 0; i < vehicles_; ++i) {
01087     solver_->AddConstraint(solver_->MakeEquality(
01088         vehicle_vars_[starts_[i]], solver_->MakeIntConst(i)));
01089     solver_->AddConstraint(solver_->MakeEquality(
01090         vehicle_vars_[ends_[i]], solver_->MakeIntConst(i)));
01091   }
01092   std::vector<IntVar*> zero_transit(size, solver_->MakeIntConst(Zero()));
01093   solver_->AddConstraint(solver_->MakePathCumul(nexts_.get(),
01094                                                 active_.get(),
01095                                                 vehicle_vars_.get(),
01096                                                 zero_transit.data(),
01097                                                 size,
01098                                                 size + vehicles_));
01099 
01100   // Add constraints to bind vehicle_vars_[i] to -1 in case that node i is not
01101   // active.
01102   for (int i = 0; i < size; ++i) {
01103     solver_->AddConstraint(solver_->MakeIsDifferentCstCt(vehicle_vars_[i], -1,
01104                                                          active_[i]));
01105   }
01106 
01107   // Set all active unless there are disjunctions
01108   if (disjunctions_.size() == 0) {
01109     AddAllActive();
01110   }
01111 
01112   // Associate first and "logical" last nodes
01113   for (int i = 0; i < vehicles_; ++i) {
01114     for (int j = 0; j < vehicles_; ++j) {
01115       if (i != j) {
01116         nexts_[starts_[i]]->RemoveValue(ends_[j]);
01117       }
01118     }
01119   }
01120 
01121   std::vector<IntVar*> cost_elements;
01122   // Arc costs: the cost of an arc (i, nexts_[i], vehicle_vars_[i]) is
01123   // costs_(nexts_[i], vehicle_vars_[i]); the total cost is the sum of arc
01124   // costs.
01125   for (int i = 0; i < size; ++i) {
01126     IntExpr* expr = NULL;
01127     if (homogeneous_costs_) {
01128       expr = solver_->MakeElement(
01129           NewPermanentCallback(this,
01130                                &RoutingModel::GetHomogeneousCost,
01131                                static_cast<int64>(i)),
01132           nexts_[i]);
01133     } else {
01134       expr = solver_->MakeElement(
01135           NewPermanentCallback(this,
01136                                &RoutingModel::GetCost,
01137                                static_cast<int64>(i)),
01138           nexts_[i],
01139           vehicle_vars_[i]);
01140     }
01141     IntVar* var = solver_->MakeProd(expr, active_[i])->Var();
01142     cost_elements.push_back(var);
01143   }
01144   // Penalty costs
01145   for (int i = 0; i < disjunctions_.size(); ++i) {
01146     IntVar* penalty_var = CreateDisjunction(i);
01147     if (penalty_var != NULL) {
01148       cost_elements.push_back(penalty_var);
01149     }
01150   }
01151   cost_ = solver_->MakeSum(cost_elements)->Var();
01152   cost_->set_name("Cost");
01153 
01154   SetUpSearch();
01155 }

01232 // Computing a lower bound to the cost of a vehicle routing problem solving a
01233 // a linear assignment problem (minimum-cost perfect bipartite matching).
01234 // A bipartite graph is created with left nodes representing the nodes of the
01235 // routing problem and right nodes representing possible node successors; an
01236 // arc between a left node l and a right node r is created if r can be the
01237 // node folowing l in a route (Next(l) = r); the cost of the arc is the transit
01238 // cost between l and r in the routing problem.
01239 // This is a lower bound given the solution to assignment problem does not
01240 // necessarily produce a (set of) closed route(s) from a starting node to an
01241 // ending node.
01242 int64 RoutingModel::ComputeLowerBound() {
01243   if (!closed_) {
01244     LOG(WARNING) << "Non-closed model not supported.";
01245     return 0;
01246   }
01247   if (!homogeneous_costs_) {
01248     LOG(WARNING) << "Non-homogeneous vehicle costs not supported";
01249     return 0;
01250   }
01251   if (disjunctions_.size() > 0) {
01252     LOG(WARNING)
01253         << "Node disjunction constraints or optional nodes not supported.";
01254     return 0;
01255   }
01256   const int num_nodes = Size() + vehicles_;
01257   ForwardStarGraph graph(2 * num_nodes, num_nodes * num_nodes);
01258   LinearSumAssignment<ForwardStarGraph> linear_sum_assignment(graph, num_nodes);
01259   // Adding arcs for non-end nodes, based on possible values of next variables.
01260   // Left nodes in the bipartite are indexed from 0 to num_nodes - 1; right
01261   // nodes are indexed from num_nodes to 2 * num_nodes - 1.
01262   for (int tail = 0; tail < Size(); ++tail) {
01263     scoped_ptr<IntVarIterator> iterator(
01264         nexts_[tail]->MakeDomainIterator(false));
01265     for (iterator->Init(); iterator->Ok(); iterator->Next()) {
01266       const int head = iterator->Value();
01267       // Given there are no disjunction constraints, a node cannot point to
01268       // itself. Doing this explicitely given that outside the search,
01269       // propagation hasn't removed this value from next variables yet.
01270       if (head == tail) {
01271         continue;
01272       }
01273       // The index of a right node in the bipartite graph is the index
01274       // of the successor offset by the number of nodes.
01275       const ArcIndex arc = graph.AddArc(tail, num_nodes + head);
01276       const CostValue cost = GetHomogeneousCost(tail, head);
01277       linear_sum_assignment.SetArcCost(arc, cost);
01278     }
01279   }
01280   // The linear assignment library requires having as many left and right nodes.
01281   // Therefore we are creating fake assignments for end nodes, forced to point
01282   // to the equivalent start node with a cost of 0.
01283   for (int tail = Size(); tail < num_nodes; ++tail) {
01284     const ArcIndex arc = graph.AddArc(tail, num_nodes + starts_[tail - Size()]);
01285     linear_sum_assignment.SetArcCost(arc, 0);
01286   }
01287   if (linear_sum_assignment.ComputeAssignment()) {
01288     return linear_sum_assignment.GetCost();
01289   }
01290   return 0;
01291 }

01293 bool RoutingModel::RouteCanBeUsedByVehicle(const Assignment& assignment,
01294                                            int start_index,
01295                                            int vehicle) const {
01296   int current_index = IsStart(start_index) ?
01297                       Next(assignment, start_index) : start_index;

01301       return false;

01304     CHECK_NE(next_index, current_index) << "Inactive node inside a route";
01305     current_index = next_index;
01306   }
01307   return true;

01310 bool RoutingModel::ReplaceUnusedVehicle(
01311     int unused_vehicle,
01312     int active_vehicle,
01313     Assignment* const compact_assignment) const {
01314   CHECK_NOTNULL(compact_assignment);
01315   CHECK(!IsVehicleUsed(*compact_assignment, unused_vehicle));
01316   CHECK(IsVehicleUsed(*compact_assignment, active_vehicle));
01317   // Swap NextVars at start nodes.
01318   const int unused_vehicle_start = Start(unused_vehicle);
01319   IntVar* const unused_vehicle_start_var = NextVar(unused_vehicle_start);
01320   const int unused_vehicle_end = End(unused_vehicle);
01321   const int active_vehicle_start = Start(active_vehicle);
01322   const int active_vehicle_end = End(active_vehicle);
01323   IntVar* const active_vehicle_start_var = NextVar(active_vehicle_start);
01324   const int active_vehicle_next =
01325       compact_assignment->Value(active_vehicle_start_var);
01326   compact_assignment->SetValue(unused_vehicle_start_var, active_vehicle_next);
01327   compact_assignment->SetValue(active_vehicle_start_var, End(active_vehicle));
01328 
01329   // Update VehicleVars along the route, update the last NextVar.
01330   int current_index = active_vehicle_next;
01331   while (!IsEnd(current_index)) {
01332     IntVar* const vehicle_var = VehicleVar(current_index);
01333     compact_assignment->SetValue(vehicle_var, unused_vehicle);
01334     const int next_index = Next(*compact_assignment, current_index);
01335     if (IsEnd(next_index)) {
01336       IntVar* const last_next_var = NextVar(current_index);
01337       compact_assignment->SetValue(last_next_var, End(unused_vehicle));
01338     }
01339     current_index = next_index;
01340   }
01341 
01342   // Update dimensions: update transits at the start.
01343   for (VarMap::const_iterator dimension = transits_.begin();
01344        dimension != transits_.end(); ++dimension) {
01345     IntVar** const transit_variables = dimension->second;
01346     IntVar* const unused_vehicle_transit_var =
01347         transit_variables[unused_vehicle_start];
01348     IntVar* const active_vehicle_transit_var =
01349         transit_variables[active_vehicle_start];
01350     const bool contains_unused_vehicle_transit_var =
01351         compact_assignment->Contains(unused_vehicle_transit_var);
01352     const bool contains_active_vehicle_transit_var =
01353         compact_assignment->Contains(active_vehicle_transit_var);
01354     if (contains_unused_vehicle_transit_var !=
01355         contains_active_vehicle_transit_var) {
01356       LG << "The assignment contains transit variable for dimension '"
01357          << dimension->first << "' for some vehicles, but not for all";
01358       return false;
01359     }
01360     if (contains_unused_vehicle_transit_var) {
01361       const int64 old_unused_vehicle_transit =
01362           compact_assignment->Value(unused_vehicle_transit_var);
01363       const int64 old_active_vehicle_transit =
01364           compact_assignment->Value(active_vehicle_transit_var);
01365       compact_assignment->SetValue(unused_vehicle_transit_var,
01366                                    old_active_vehicle_transit);
01367       compact_assignment->SetValue(active_vehicle_transit_var,
01368                                    old_unused_vehicle_transit);
01369     }
01370 
01371     // Update dimensions: update cumuls at the end.
01372     IntVar** const cumul_variables =
01373         FindWithDefault(cumuls_, dimension->first, NULL);
01374     CHECK_NOTNULL(cumul_variables);
01375     IntVar* const unused_vehicle_cumul_var =
01376         cumul_variables[unused_vehicle_end];
01377     IntVar* const active_vehicle_cumul_var =
01378         cumul_variables[active_vehicle_end];
01379     const int64 old_unused_vehicle_cumul =
01380         compact_assignment->Value(unused_vehicle_cumul_var);
01381     const int64 old_active_vehicle_cumul =
01382         compact_assignment->Value(active_vehicle_cumul_var);
01383     compact_assignment->SetValue(unused_vehicle_cumul_var,
01384                                  old_active_vehicle_cumul);
01385     compact_assignment->SetValue(active_vehicle_cumul_var,
01386                                  old_unused_vehicle_cumul);
01387   }
01388   return true;
01389 }

01391 Assignment* RoutingModel::CompactAssignment(
01392     const Assignment& assignment) const {
01393   CHECK_EQ(assignment.solver(), solver_.get());
01394   if (!homogeneous_costs_) {
01395     LG << "The costs are not homogeneous, routes cannot be rearranged";
01396     return NULL;
01397   }
01398 
01399   Assignment* compact_assignment = new Assignment(&assignment);
01400   for (int vehicle = 0; vehicle < vehicles_ - 1; ++vehicle) {
01401     if (IsVehicleUsed(*compact_assignment, vehicle)) {
01402       continue;
01403     }
01404     const int vehicle_start = Start(vehicle);
01405     const int vehicle_end = End(vehicle);
01406     // Find the last vehicle, that can swap routes with this one.
01407     int swap_vehicle = vehicles_ - 1;
01408     bool has_more_vehicles_with_route = false;
01409     for (; swap_vehicle > vehicle; --swap_vehicle) {
01410       // If a vehicle was already swapped, it will appear in compact_assignment
01411       // as unused.
01412       if (!IsVehicleUsed(*compact_assignment, swap_vehicle)
01413           || !IsVehicleUsed(*compact_assignment, swap_vehicle)) {
01414         continue;
01415       }
01416       has_more_vehicles_with_route = true;
01417       const int swap_vehicle_start = Start(swap_vehicle);
01418       const int swap_vehicle_end = End(swap_vehicle);
01419       if (IndexToNode(vehicle_start) != IndexToNode(swap_vehicle_start)
01420           || IndexToNode(vehicle_end) != IndexToNode(swap_vehicle_end)) {
01421         continue;
01422       }
01423 
01424       // Check that updating VehicleVars is OK.
01425       if (RouteCanBeUsedByVehicle(*compact_assignment, swap_vehicle_start,
01426                                   vehicle)) {
01427         break;
01428       }
01429     }
01430 
01431     if (swap_vehicle == vehicle) {
01432       if (has_more_vehicles_with_route) {
01433         // No route can be assigned to this vehicle, but there are more vehicles
01434         // with a route left. This would leave a gap in the indices.
01435         LG << "No vehicle that can be swapped with " << vehicle << " was found";
01436         delete compact_assignment;
01437         return NULL;
01438       } else {
01439         break;
01440       }
01441     } else {
01442       if (!ReplaceUnusedVehicle(vehicle, swap_vehicle, compact_assignment)) {
01443         delete compact_assignment;
01444         return NULL;
01445       }
01446     }
01447   }
01448   if (FLAGS_routing_check_compact_assignment) {
01449     if (!solver_->CheckAssignment(compact_assignment)) {
01450       LG << "The compacted assignment is not a valid solution";
01451       delete compact_assignment;
01452       return NULL;
01453     }
01454   }
01455   return compact_assignment;
01456 }

01567 bool RoutingModel::RoutesToAssignment(const std::vector<std::vector<NodeIndex> >& routes,
01568                                       bool ignore_inactive_nodes,
01569                                       bool close_routes,
01570                                       Assignment* const assignment) const {
01571   CHECK_NOTNULL(assignment);
01572   if (!closed_) {
01573     LOG(ERROR) << "The model is not closed yet";
01574     return false;
01575   }
01576   const int num_routes = routes.size();
01577   if (num_routes > vehicles_) {
01578     LOG(ERROR) << "The number of vehicles in the assignment (" << routes.size()
01579                << ") is greater than the number of vehicles in the model ("
01580                << vehicles_ << ")";
01581     return false;
01582   }
01583 
01584   hash_set<int> visited_indices;
01585   // Set value to NextVars based on the routes.
01586   for (int vehicle = 0; vehicle < num_routes; ++vehicle) {
01587     const std::vector<NodeIndex>& route = routes[vehicle];
01588     int from_index = Start(vehicle);
01589     std::pair<hash_set<int>::iterator, bool> insert_result =
01590         visited_indices.insert(from_index);
01591     if (!insert_result.second) {
01592       LOG(ERROR) << "Index " << from_index << " (start node for vehicle "
01593                  << vehicle << ") was already used";
01594       return false;
01595     }
01596 
01597     for (int i = 0; i < route.size(); ++i) {
01598       const NodeIndex to_node = route[i];
01599       if (to_node < 0 || to_node >= nodes()) {
01600         LOG(ERROR) << "Invalid node index: " << to_node;
01601         return false;
01602       }
01603       const int to_index = NodeToIndex(to_node);
01604       if (to_index < 0 || to_index >= Size()) {
01605         LOG(ERROR) << "Invalid index: " << to_index << " from node "
01606                    << to_node;
01607         return false;
01608       }
01609 
01610       IntVar* const active_var = ActiveVar(to_index);
01611       if (active_var->Max() == 0) {
01612         if (ignore_inactive_nodes) {
01613           continue;
01614         } else {
01615           LOG(ERROR) << "Index " << to_index << " (node " << to_node
01616                      << ") is not active";
01617           return false;
01618         }
01619       }
01620 
01621       insert_result = visited_indices.insert(to_index);
01622       if (!insert_result.second) {
01623         LOG(ERROR) << "Index " << to_index << " (node " << to_node
01624                    << ") is used multiple times";
01625         return false;
01626       }
01627 
01628       const IntVar* const vehicle_var = VehicleVar(to_index);
01629       if (!vehicle_var->Contains(vehicle)) {
01630         LOG(ERROR) << "Vehicle " << vehicle << " is not allowed at index "
01631                    << to_index << " (node " << to_node << ")";
01632         return false;
01633       }
01634 
01635       IntVar* const from_var = NextVar(from_index);
01636       if (!assignment->Contains(from_var)) {
01637         assignment->Add(from_var);
01638       }
01639       assignment->SetValue(from_var, to_index);
01640 
01641       from_index = to_index;
01642     }
01643 
01644     if (close_routes) {
01645       IntVar* const last_var = NextVar(from_index);
01646       if (!assignment->Contains(last_var)) {
01647         assignment->Add(last_var);
01648       }
01649       assignment->SetValue(last_var, End(vehicle));
01650     }
01651   }
01652 
01653   // Do not use the remaining vehicles.
01654   for (int vehicle = num_routes; vehicle < vehicles_; ++vehicle) {
01655     const int start_index = Start(vehicle);
01656     // Even if close_routes is false, we still need to add the start index to
01657     // visited_indices so that deactivating other nodes works correctly.
01658     std::pair<hash_set<int>::iterator, bool> insert_result =
01659         visited_indices.insert(start_index);
01660     if (!insert_result.second) {
01661       LOG(ERROR) << "Index " << start_index << " is used multiple times";
01662       return false;
01663     }
01664     if (close_routes) {
01665       IntVar* const start_var = NextVar(start_index);
01666       if (!assignment->Contains(start_var)) {
01667         assignment->Add(start_var);
01668       }
01669       assignment->SetValue(start_var, End(vehicle));
01670     }
01671   }
01672 
01673   // Deactivate other nodes (by pointing them to themselves).
01674   if (close_routes) {
01675     for (int index = 0; index < Size(); ++index) {
01676       if (!ContainsKey(visited_indices, index)) {
01677         IntVar* const next_var = NextVar(index);
01678         if (!assignment->Contains(next_var)) {
01679           assignment->Add(next_var);
01680         }
01681         assignment->SetValue(next_var, index);
01682       }
01683     }
01684   }
01685 
01686   return true;
01687 }

01746 int64 RoutingModel::GetArcCost(int64 i, int64 j, int64 vehicle) {
01747   if (vehicle < 0) {
01748     // If a node is inactive, vehicle will be set to -1; handle this situation
01749     // correctly.
01750     // Even though the constraint model does not allow vehice == -1 when i != j,
01751     // this method gets called with such parameters when the model is created,
01752     // because the constraing solver needs to find the minimal and maximal arc
01753     // costs and calls the method with all possible combinations of parameters.
01754     // Zero should be a safe value here.
01755     return 0;
01756   }
01757   CostCacheElement& cache = cost_cache_[i];
01758   if (cache.node == j && cache.vehicle == vehicle) {
01759     return cache.cost;
01760   }
01761   const NodeIndex node_i = IndexToNode(i);
01762   const NodeIndex node_j = IndexToNode(j);
01763   int64 cost = 0;
01764   if (!IsStart(i)) {
01765     cost = costs_[vehicle]->Run(node_i, node_j);
01766   } else if (!IsEnd(j)) {
01767     // Apply route fixed cost on first non-first/last node, in other words on
01768     // the arc from the first node to its next node if it's not the last node.
01769     cost = costs_[vehicle]->Run(node_i, node_j)
01770       + fixed_costs_[index_to_vehicle_[i]];
01771   } else {
01772     // If there's only the first and last nodes on the route, it is considered
01773     // as an empty route thus the cost of 0.
01774     cost = 0;
01775   }
01776   cache.node = j;
01777   cache.vehicle = vehicle;
01778   cache.cost = cost;
01779   return cost;
01780 }

01879 void RoutingModel::SetUpSearch() {

01882   assignment_->Add(nexts_.get(), size);
01883   if (!homogeneous_costs_) {
01884     assignment_->Add(vehicle_vars_.get(), size + vehicles_);
01885   }
01886   assignment_->AddObjective(cost_);
01887 
01888   Assignment* full_assignment =
01889       solver_->RevAlloc(new Assignment(solver_.get()));
01890   for (ConstIter<VarMap> it(cumuls_); !it.at_end(); ++it) {
01891     full_assignment->Add(it->second, size + vehicles_);
01892   }
01893   for (int i = 0; i < extra_vars_.size(); ++i) {
01894     full_assignment->Add(extra_vars_[i]);
01895   }
01896   full_assignment->Add(nexts_.get(), size);
01897   full_assignment->Add(active_.get(), size);
01898   full_assignment->Add(vehicle_vars_.get(), size + vehicles_);
01899   full_assignment->AddObjective(cost_);
01900 
01901   collect_assignments_ =
01902       solver_->MakeBestValueSolutionCollector(full_assignment, false);
01903   monitors_.push_back(collect_assignments_);
01904 
01905   SearchMonitor* optimize;
01906   switch (GetSelectedMetaheuristic()) {
01907     case ROUTING_GUIDED_LOCAL_SEARCH:
01908       LG << "Using Guided Local Search";
01909       if (homogeneous_costs_) {
01910         optimize = solver_->MakeGuidedLocalSearch(
01911             false,
01912             cost_,
01913             NewPermanentCallback(this, &RoutingModel::GetHomogeneousCost),
01914             FLAGS_routing_optimization_step,
01915             nexts_.get(), size,
01916             FLAGS_routing_guided_local_search_lamda_coefficient);
01917       } else {
01918         optimize = solver_->MakeGuidedLocalSearch(
01919             false,
01920             cost_,
01921             BuildCostCallback(),
01922             FLAGS_routing_optimization_step,
01923             nexts_.get(), vehicle_vars_.get(), size,
01924             FLAGS_routing_guided_local_search_lamda_coefficient);
01925       }
01926       break;
01927     case ROUTING_SIMULATED_ANNEALING:
01928       LG << "Using Simulated Annealing";
01929       optimize =
01930           solver_->MakeSimulatedAnnealing(false, cost_,
01931                                           FLAGS_routing_optimization_step,
01932                                           100);
01933       break;
01934     case ROUTING_TABU_SEARCH:
01935       LG << "Using Tabu Search";
01936       optimize = solver_->MakeTabuSearch(false, cost_,
01937                                          FLAGS_routing_optimization_step,
01938                                          nexts_.get(), size,
01939                                          10, 10, .8);
01940       break;
01941     default:
01942       LG << "Using greedy descent";
01943       optimize = solver_->MakeMinimize(cost_, FLAGS_routing_optimization_step);
01944   }
01945   monitors_.push_back(optimize);
01946 
01947   limit_ = solver_->MakeLimit(time_limit_ms_,
01948                               kint64max,
01949                               kint64max,
01950                               FLAGS_routing_solution_limit,
01951                               true);
01952   monitors_.push_back(limit_);
01953 
01954   ls_limit_ = solver_->MakeLimit(time_limit_ms_,
01955                                  kint64max,
01956                                  kint64max,
01957                                  1,
01958                                  true);
01959 
01960   lns_limit_ = solver_->MakeLimit(lns_time_limit_ms_,
01961                                   kint64max,
01962                                   kint64max,
01963                                   kint64max);
01964 
01965   std::vector<LocalSearchOperator*> operators = extra_operators_;
01966   if (pickup_delivery_pairs_.size() > 0) {
01967     const IntVar* const* vehicle_vars =
01968         homogeneous_costs_ ? NULL : vehicle_vars_.get();
01969     operators.push_back(MakePairActive(solver_.get(),
01970                                        nexts_.get(),
01971                                        vehicle_vars,
01972                                        pickup_delivery_pairs_,
01973                                        size));
01974     operators.push_back(MakePairRelocate(solver_.get(),
01975                                          nexts_.get(),
01976                                          vehicle_vars,
01977                                          pickup_delivery_pairs_,
01978                                          size));
01979   }
01980   if (vehicles_ > 1) {
01981     if (!FLAGS_routing_no_relocate) {
01982       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::RELOCATE);
01983     }
01984     if (!FLAGS_routing_no_exchange) {
01985       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::EXCHANGE);
01986     }
01987     if (!FLAGS_routing_no_cross) {
01988       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::CROSS);
01989     }
01990   }
01991   if (!FLAGS_routing_no_lkh
01992       && !FLAGS_routing_tabu_search
01993       && !FLAGS_routing_simulated_annealing) {
01994     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::LK);
01995   }
01996   if (!FLAGS_routing_no_2opt) {
01997     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::TWOOPT);
01998   }
01999   if (!FLAGS_routing_no_oropt) {
02000     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::OROPT);
02001   }
02002   if (!FLAGS_routing_no_make_active && disjunctions_.size() != 0) {
02003     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::MAKEINACTIVE);
02004     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::MAKEACTIVE);
02005     if (!FLAGS_routing_use_extended_swap_active) {
02006       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::SWAPACTIVE);
02007     } else {
02008       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::EXTENDEDSWAPACTIVE);
02009     }
02010   }
02011   // TODO(user): move the following operators to a second local search loop.
02012   if (!FLAGS_routing_no_tsp
02013       && !FLAGS_routing_tabu_search
02014       && !FLAGS_routing_simulated_annealing) {
02015     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::TSPOPT);
02016   }
02017   if (!FLAGS_routing_no_tsplns
02018       && !FLAGS_routing_tabu_search
02019       && !FLAGS_routing_simulated_annealing) {
02020     CP_ROUTING_PUSH_BACK_CALLBACK_OPERATOR(Solver::TSPLNS);
02021   }
02022   if (!FLAGS_routing_no_lns) {
02023     CP_ROUTING_PUSH_BACK_OPERATOR(Solver::PATHLNS);
02024     if (disjunctions_.size() != 0) {
02025       CP_ROUTING_PUSH_BACK_OPERATOR(Solver::UNACTIVELNS);
02026     }
02027   }
02028

02030       solver_->ConcatenateOperators(operators);
02031 
02032   DecisionBuilder* finalize_solution =
02033       solver_->MakePhase(nexts_.get(), size,
02034                          Solver::CHOOSE_FIRST_UNBOUND,
02035                          Solver::ASSIGN_MIN_VALUE);
02036

02039     case ROUTING_GLOBAL_CHEAPEST_ARC:

02043                              NewPermanentCallback(
02044                                  this,
02045                                  &RoutingModel::GetFirstSolutionCost),
02046                              Solver::CHOOSE_STATIC_GLOBAL_BEST);
02047       break;
02048     case ROUTING_LOCAL_CHEAPEST_ARC:
02049       LG << "Using ROUTING_LOCAL_CHEAPEST_ARC";
02050       first_solution =
02051           solver_->MakePhase(nexts_.get(), size,
02052                              Solver::CHOOSE_FIRST_UNBOUND,
02053                              NewPermanentCallback(
02054                                  this,
02055                                  &RoutingModel::GetFirstSolutionCost));
02056       break;
02057     case ROUTING_PATH_CHEAPEST_ARC:
02058       LG << "Using ROUTING_PATH_CHEAPEST_ARC";
02059       first_solution =
02060           solver_->MakePhase(nexts_.get(), size,
02061                              Solver::CHOOSE_PATH,
02062                              NewPermanentCallback(
02063                                  this,
02064                                  &RoutingModel::GetFirstSolutionCost));
02065       break;
02066     case ROUTING_EVALUATOR_STRATEGY:
02067       LG << "Using ROUTING_EVALUATOR_STRATEGY";
02068       CHECK(first_solution_evaluator_ != NULL);
02069       first_solution =
02070           solver_->MakePhase(nexts_.get(), size,
02071                              Solver::CHOOSE_PATH,
02072                              NewPermanentCallback(
02073                                  first_solution_evaluator_.get(),
02074                                  &Solver::IndexEvaluator2::Run));
02075       break;
02076     case ROUTING_DEFAULT_STRATEGY:
02077       LG << "Using DEFAULT";
02078       break;
02079     case ROUTING_ALL_UNPERFORMED:
02080       first_solution =
02081           solver_->RevAlloc(new AllUnperformed(this));
02082       break;
02083     default:
02084       LOG(WARNING) << "Unknown argument for routing_first_solution, "
02085           "using default";
02086   }
02087   if (FLAGS_routing_use_first_solution_dive) {
02088     DecisionBuilder* apply =
02089         solver_->MakeApplyBranchSelector(NewPermanentCallback(&LeftDive));
02090     first_solution = solver_->Compose(apply, first_solution);
02091   }
02092 
02093   std::vector<LocalSearchFilter*> filters;
02094   if (FLAGS_routing_use_objective_filter) {
02095     if (homogeneous_costs_) {
02096       LocalSearchFilter* filter =
02097           solver_->MakeLocalSearchObjectiveFilter(
02098               nexts_.get(),
02099               size,
02100               NewPermanentCallback(this,
02101                                    &RoutingModel::GetHomogeneousFilterCost),
02102               cost_,
02103               Solver::EQ,
02104               Solver::SUM);
02105       filters.push_back(filter);
02106     } else {
02107       LocalSearchFilter* filter =
02108           solver_->MakeLocalSearchObjectiveFilter(
02109               nexts_.get(),
02110               vehicle_vars_.get(),
02111               size,
02112               NewPermanentCallback(this, &RoutingModel::GetFilterCost),
02113               cost_,
02114               Solver::EQ,
02115               Solver::SUM);
02116       filters.push_back(filter);
02117     }
02118   }
02119   if (FLAGS_routing_use_pickup_and_delivery_filter
02120       && pickup_delivery_pairs_.size() > 0) {
02121     filters.push_back(solver_->RevAlloc(new NodePrecedenceFilter(
02122         nexts_.get(),
02123         Size(),
02124         Size() + vehicles_,
02125         pickup_delivery_pairs_,
02126         "")));
02127   }
02128   if (FLAGS_routing_use_path_cumul_filter) {
02129     for (ConstIter<VarMap> iter(cumuls_); !iter.at_end(); ++iter) {
02130       const string& name = iter->first;
02131       filters.push_back(solver_->RevAlloc(
02132           new PathCumulFilter(nexts_.get(),
02133                               Size(),
02134                               iter->second,
02135                               Size() + vehicles_,
02136                               transit_evaluators_[name],
02137                               name)));
02138     }
02139   }
02140 
02141   LocalSearchPhaseParameters* parameters =
02142       solver_->MakeLocalSearchPhaseParameters(
02143           local_search_operator,
02144           solver_->MakeSolveOnce(finalize_solution, lns_limit_),
02145           ls_limit_,
02146           filters);
02147 
02148   if (FLAGS_routing_dfs) {
02149     solve_db_ = finalize_solution;
02150   } else {
02151     if (homogeneous_costs_) {
02152       solve_db_ = solver_->MakeLocalSearchPhase(nexts_.get(),
02153                                                 size,
02154                                                 first_solution,
02155                                                 parameters);
02156     } else {

02163         all_vars[i] = vehicle_vars_[i - size];

02165       solve_db_ = solver_->MakeLocalSearchPhase(all_vars.get(),
02166                                                 all_size,
02167                                                 first_solution,
02168                                                 parameters);
02169     }
02170   }
02171   DecisionBuilder* restore_preassignment =
02172       solver_->MakeRestoreAssignment(preassignment_);
02173   solve_db_ = solver_->Compose(restore_preassignment, solve_db_);
02174   improve_db_ =
02175       solver_->Compose(restore_preassignment,
02176                        solver_->MakeLocalSearchPhase(assignment_, parameters));
02177   restore_assignment_ =
02178       solver_->Compose(solver_->MakeRestoreAssignment(assignment_),
02179                        finalize_solution);
02180 
02181   if (FLAGS_routing_trace) {
02182     const int kLogPeriod = 10000;
02183     SearchMonitor* trace = solver_->MakeSearchLog(kLogPeriod, cost_);
02184     monitors_.push_back(trace);
02185   }
02186   if (FLAGS_routing_search_trace) {
02187     SearchMonitor* trace = solver_->MakeSearchTrace("Routing ");
02188     monitors_.push_back(trace);
02189   }
02190 }
