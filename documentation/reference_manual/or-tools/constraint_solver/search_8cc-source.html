00791 bool PathSelector::UpdateIndex(int64* index) const {
00792   if (*index >= size_) {

00795     }

00800 // Select variables on a path:
00801 //  1. Try to extend an existing route: look for an unbound variable, to which
00802 //     some other variable points.
00803 //  2. If no such road is found, try to find a start node of a route: look for
00804 //     an unbound variable, to which no other variable can point.
00805 //  3. If everything else fails, pick the first unbound variable.
00806 bool PathSelector::FindPathStart(int64* index) const {
00807   // Try to extend an existing path
00808   for (int i = size_ - 1; i >= 0; --i) {
00809     if (vars_[i]->Bound()) {
00810       const int next = vars_[i]->Value();
00811       if (next < size_ && !vars_[next]->Bound()) {
00812         *index = next;
00813         return true;
00814       }
00815     }
00816   }
00817   // Pick path start
00818   for (int i = size_ - 1; i >= 0; --i) {
00819     if (!vars_[i]->Bound()) {
00820       bool has_possible_prev = false;
00821       for (int j = 0; j < size_; ++j) {
00822         if (vars_[j]->Contains(i)) {
00823           has_possible_prev = true;
00824           break;
00825         }
00826       }
00827       if (!has_possible_prev) {
00828         *index = i;
00829         return true;
00830       }
00831     }
00832   }
00833   // Pick first unbound
00834   for (int i = 0; i < size_; ++i) {
00835     if (!vars_[i]->Bound()) {
00836       *index = i;
00837       return true;
00838     }
00839   }
00840   return false;
00841 }

00910           }
00911         }

02625 
02626 void TabuSearch::ApplyDecision(Decision* const d) {
02627   Solver* const s = solver();
02628   if (d == s->balancing_decision()) {
02629     return;
02630   }
02631   // Aspiration criterion
02632   // Accept a neighbor if it improves the best solution found so far
02633   IntVar* aspiration = s->MakeBoolVar();
02634   if (maximize_) {
02635     s->AddConstraint(s->MakeIsGreaterOrEqualCstCt(objective_,
02636                                                   best_ + step_,
02637                                                   aspiration));
02638   } else {
02639     s->AddConstraint(s->MakeIsLessOrEqualCstCt(objective_,
02640                                                best_ - step_,
02641                                                aspiration));
02642   }
02643 
02644   // Tabu criterion
02645   // A variable in the "keep" list must keep its value, a variable in the
02646   // "forbid" list must not take its value in the list. The tabu criterion is
02647   // softened by the tabu factor which gives the number of violations to
02648   // the tabu criterion which is tolerated; a factor of 1 means no violations
02649   // allowed, a factor of 0 means all violations allowed.
02650   std::vector<IntVar*> tabu_vars;
02651   for (ConstIter<TabuList > it(keep_tabu_list_); !it.at_end(); ++it) {
02652     IntVar* tabu_var = s->MakeBoolVar();
02653     Constraint* keep_cst =
02654         s->MakeIsEqualCstCt((*it).var_, (*it).value_, tabu_var);
02655     s->AddConstraint(keep_cst);
02656     tabu_vars.push_back(tabu_var);
02657   }
02658   for (ConstIter<TabuList > it(forbid_tabu_list_); !it.at_end(); ++it) {
02659     IntVar* tabu_var = s->MakeBoolVar();
02660     Constraint* forbid_cst =
02661         s->MakeIsDifferentCstCt((*it).var_, (*it).value_, tabu_var);
02662     s->AddConstraint(forbid_cst);
02663     tabu_vars.push_back(tabu_var);
02664   }
02665   if (tabu_vars.size() > 0) {
02666     IntVar* tabu = s->MakeBoolVar();
02667     s->AddConstraint(s->MakeIsGreaterOrEqualCstCt(s->MakeSum(tabu_vars)->Var(),
02668                                                   tabu_vars.size() *
02669                                                   tabu_factor_,
02670                                                   tabu));
02671     s->AddConstraint(s->MakeGreaterOrEqual(s->MakeSum(aspiration, tabu), 1LL));
02672   }
02673 
02674   // Go downhill to the next local optimum
02675   if (maximize_) {
02676     const int64 bound = (current_ > kint64min) ? current_ + step_ : current_;
02677     s->AddConstraint(s->MakeGreaterOrEqual(objective_, bound));
02678   } else {
02679     const int64 bound = (current_ < kint64max) ? current_ - step_ : current_;
02680     s->AddConstraint(s->MakeLessOrEqual(objective_, bound));
02681   }
02682 
02683   // Avoid cost plateau's which lead to tabu cycles
02684   if (found_initial_solution_) {
02685     s->AddConstraint(s->MakeNonEquality(objective_, last_));
02686   }
02687 }
02688 
02689 bool TabuSearch::AtSolution() {
02690   if (!Metaheuristic::AtSolution()) {
02691     return false;
02692   }
02693   found_initial_solution_ = true;
02694   last_ = current_;
02695 
02696   // New solution found: add new assignments to tabu lists; this is only
02697   // done after the first local optimum (stamp_ != 0)
02698   if (0 != stamp_) {
02699     for (int i = 0; i < size_; ++i) {
02700       IntVar* const var = vars_[i];
02701       const int64 old_value = assignment_.Value(var);
02702       const int64 new_value = var->Value();
02703       if (old_value != new_value) {
02704         VarValue keep_value(var, new_value, stamp_);
02705         keep_tabu_list_.push_front(keep_value);
02706         VarValue forbid_value(var, old_value, stamp_);
02707         forbid_tabu_list_.push_front(forbid_value);
02708       }
02709     }
02710   }
02711   assignment_.Store();
02712 
02713   return true;
02714 }

03068 // Add the following constraint (includes aspiration criterion):
03069 // if minimizing,
03070 //      objective =< Max(current penalized cost - penalized_objective - step,
03071 //                       best solution cost - step)
03072 // if maximizing,
03073 //      objective >= Min(current penalized cost - penalized_objective + step,
03074 //                       best solution cost + step)
03075 void GuidedLocalSearch::ApplyDecision(Decision* const d) {
03076   if (d == solver()->balancing_decision()) {
03077     return;
03078   }
03079   std::vector<IntVar*> elements;
03080   assignment_penalized_value_ = 0;
03081   if (penalties_->HasValues()) {
03082     for (int i = 0; i < size_; ++i) {
03083       IntExpr* expr = MakeElementPenalty(i);
03084       elements.push_back(expr->Var());
03085       const int64 penalty = AssignmentElementPenalty(assignment_, i);
03086       current_penalized_values_[i] = penalty;
03087       delta_cache_[i] = penalty;
03088       assignment_penalized_value_ += penalty;
03089     }
03090     old_penalized_value_ = assignment_penalized_value_;
03091     incremental_ = false;
03092     penalized_objective_ = solver()->MakeSum(elements)->Var();
03093     if (maximize_) {
03094       IntExpr* min_pen_exp = solver()->MakeDifference(current_ + step_,
03095                                                       penalized_objective_);
03096       IntVar* min_exp = solver()->MakeMin(min_pen_exp, best_ + step_)->Var();
03097       solver()->AddConstraint(solver()->MakeGreaterOrEqual(objective_,
03098                                                            min_exp));
03099     } else {
03100       IntExpr* max_pen_exp = solver()->MakeDifference(current_ - step_,
03101                                                       penalized_objective_);
03102       IntVar* max_exp = solver()->MakeMax(max_pen_exp, best_ - step_)->Var();
03103       solver()->AddConstraint(solver()->MakeLessOrEqual(objective_, max_exp));
03104     }
03105   } else {
03106     penalized_objective_ = NULL;
03107     if (maximize_) {
03108       const int64 bound = (current_ > kint64min) ? current_ + step_ : current_;
03109       objective_->SetMin(bound);
03110     } else {
03111       const int64 bound = (current_ < kint64max) ? current_ - step_ : current_;
03112       objective_->SetMax(bound);
03113     }
03114   }
03115 }

03122     current_ += penalized_objective_->Value();

03125   return true;

03138 // GLS filtering; compute the penalized value corresponding to the delta and
03139 // modify objective bound accordingly.
03140 bool GuidedLocalSearch::AcceptDelta(Assignment* delta, Assignment* deltadelta) {
03141   if ((delta != NULL || deltadelta != NULL) && penalties_->HasValues()) {
03142     int64 penalty = 0;
03143     if (!deltadelta->Empty()) {
03144       if (!incremental_) {
03145         penalty = Evaluate(delta,
03146                            assignment_penalized_value_,
03147                            current_penalized_values_.get(),
03148                            true);
03149       } else {
03150         penalty = Evaluate(deltadelta,
03151                            old_penalized_value_,
03152                            delta_cache_.get(),
03153                            true);
03154       }
03155       incremental_ = true;
03156     } else {
03157       if (incremental_) {
03158         for (int i = 0; i < size_; ++i) {
03159           delta_cache_[i] = current_penalized_values_[i];
03160         }
03161         old_penalized_value_ = assignment_penalized_value_;
03162       }
03163       incremental_ = false;
03164       penalty = Evaluate(delta,
03165                          assignment_penalized_value_,
03166                          current_penalized_values_.get(),
03167                          false);
03168     }
03169     old_penalized_value_ = penalty;
03170     if (!delta->HasObjective()) {
03171       delta->AddObjective(objective_);
03172     }
03173     if (delta->Objective() == objective_) {
03174       if (maximize_) {
03175         delta->SetObjectiveMin(std::max(std::min(current_ + step_ - penalty,
03176                                                  best_ + step_),
03177                                         delta->ObjectiveMin()));
03178       } else {
03179         delta->SetObjectiveMax(std::min(std::max(current_ - step_ - penalty,
03180                                                  best_ - step_),
03181                                         delta->ObjectiveMax()));
03182       }
03183     }
03184   }
03185   return true;
03186 }

03188 int64 GuidedLocalSearch::Evaluate(const Assignment* delta,
03189                                   int64 current_penalty,
03190                                   const int64* const out_values,
03191                                   bool cache_delta_values) {
03192   int64 penalty = current_penalty;
03193   const Assignment::IntContainer& container = delta->IntVarContainer();
03194   const int size = container.Size();
03195   for (int i = 0; i < size; ++i) {
03196     const IntVarElement& new_element = container.Element(i);
03197     const IntVar* var = new_element.Var();
03198     int64 index = -1;
03199     if (FindCopy(indices_, var, &index)) {
03200       penalty -= out_values[index];
03201       int64 new_penalty = 0;
03202       if (EvaluateElementValue(container, index, &i, &new_penalty)) {
03203         penalty += new_penalty;
03204         if (cache_delta_values) {
03205           delta_cache_[index] = new_penalty;
03206         }
03207       }
03208     }
03209   }
03210   return penalty;
03211 }

03213 // Penalize all the most expensive arcs (var, value) according to their utility:
03214 // utility(i, j) = cost(i, j) / (1 + penalty(i, j))
03215 bool GuidedLocalSearch::LocalOptimum() {
03216   std::vector<std::pair<Arc, double> > utility(size_);
03217   for (int i = 0; i < size_; ++i) {
03218     const int64 var_value = assignment_.Value(vars_[i]);
03219     const int64 value =
03220         (var_value != i) ? AssignmentPenalty(assignment_, i, var_value) : 0;
03221     const Arc arc(i, var_value);
03222     const int64 penalty = penalties_->Value(arc);
03223     utility[i] = std::pair<Arc, double>(arc, value / (penalty + 1.0));
03224   }
03225   Comparator comparator;
03226   std::stable_sort(utility.begin(), utility.end(), comparator);
03227   int64 utility_value = utility[0].second;
03228   penalties_->Increment(utility[0].first);
03229   for (int i = 1;
03230        i < utility.size() && utility_value == utility[i].second;
03231        ++i) {
03232     penalties_->Increment(utility[i].first);
03233   }
03234   if (maximize_) {
03235     current_ = kint64min;
03236   } else {
03237     current_ = kint64max;
03238   }
03239   return true;
03240 }

03726 bool RegularLimit::CheckTime() {
03727   const int64 kMaxSkip = 100;
03728   const int64 kCheckWarmupIterations = 100;
03729   ++check_count_;
03730   if (wall_time_ != kint64max && next_check_ <= check_count_) {
03731     Solver* const s = solver();
03732     int64 time_delta = s->wall_time() -  wall_time_offset_;
03733     if (smart_time_check_
03734         && check_count_ > kCheckWarmupIterations
03735         && time_delta > 0) {
03736       int64 approximate_calls = (wall_time_ * check_count_) / time_delta;
03737       next_check_ = check_count_ + std::min(kMaxSkip, approximate_calls);
03738     }
03739     return time_delta >= wall_time_;
03740   } else {
03741     return false;
03742   }
03743 }

04174   // Luby Strategy
04175 int64 NextLuby(int i) {
04176   DCHECK_GT(i, 0);
04177   DCHECK_LT(i, kint32max);
04178   int64 power;
04179 
04180   // let's find the least power of 2 >= (i+1).
04181   power = 2;
04182   // Cannot overflow, because bounded by kint32max + 1.
04183   while (power < (i + 1)) {
04184     power <<= 1;
04185   }
04186   if (power == i + 1) {
04187     return (power / 2);
04188   }
04189   return NextLuby(i - (power / 2) + 1);
04190 }

04261 // ---------- Symmetry Breaking ----------
04262 
04263 // The symmetry manager maintains a list of problem symmetries.  Each
04264 // symmetry is called on each decision and should return a term
04265 // representing the boolean status of the symmetrical decision.
04266 // e.g. : the decision is x == 3, the symmetrical decision is y == 5
04267 // then the symmetry breaker should use
04268 // AddIntegerVariableEqualValueClause(y, 5).  Once this is done, upon
04269 // refutation, for each symmetry breaker, the system adds a constraint
04270 // that will forbid the symmetrical variation of the current explored
04271 // search tree. This constraint can be expressed very simply just by
04272 // keeping the list of current symmetrical decisions.
04273 //
04274 // This is called Symmetry Breaking During Search (Ian Gent, Barbara
04275 // Smith, ECAI 2000).
04276 // http://citeseerx.ist.psu.edu/viewdoc/download?doi=10.1.1.42.3788&rep=rep1&type=pdf  // NOLINT
04277 class SymmetryManager : public SearchMonitor {
04278  public:
04279   SymmetryManager(Solver* const s,
04280                   SymmetryBreaker* const * visitors,
04281                   int size)
04282       : SearchMonitor(s),
04283         visitors_(new SymmetryBreaker*[size]),
04284         clauses_(new SimpleRevFIFO<IntVar*>[size]),
04285         decisions_(new SimpleRevFIFO<Decision*>[size]),
04286         directions_(new SimpleRevFIFO<bool>[size]),  // false = left.
04287         size_(size) {
04288     CHECK_GT(size, 0);
04289     memcpy(visitors_.get(), visitors, size_ * sizeof(*visitors));
04290     for (int i = 0; i < size_; ++i) {
04291       visitors_[i]->set_symmetry_manager_and_index(this, i);
04292     }
04293   }
04294 
04295   virtual ~SymmetryManager() {}
04296 
04297   virtual void EndNextDecision(DecisionBuilder* const db, Decision* const d) {
04298     if (d) {
04299       for (int i = 0; i < size_; ++i) {
04300         const void* const last = clauses_[i].Last();
04301         d->Accept(visitors_[i]);
04302         if (last != clauses_[i].Last()) {
04303           // Synchroneous push of decision as marker.
04304           decisions_[i].Push(solver(), d);
04305           directions_[i].Push(solver(), false);
04306         }
04307       }
04308     }
04309   }
04310 
04311   virtual void RefuteDecision(Decision* d) {
04312     for (int i = 0; i < size_; ++i) {
04313       if (decisions_[i].Last() != NULL && decisions_[i].LastValue() == d) {
04314         CheckSymmetries(i);
04315       }
04316     }
04317   }
04318 
04319   // TODO(user) : Improve speed, cache previous min and build them
04320   // incrementally.
04321   void CheckSymmetries(int index) {
04322     SimpleRevFIFO<IntVar*>::Iterator tmp(&clauses_[index]);
04323     SimpleRevFIFO<bool>::Iterator tmp_dir(&directions_[index]);
04324     Constraint* ct = NULL;
04325     {
04326       std::vector<IntVar*> guard;
04327       // keep the last entry for later, if loop doesn't exit.
04328       ++tmp;
04329       ++tmp_dir;
04330       while (tmp.ok()) {
04331         IntVar* const term = *tmp;
04332         if (!*tmp_dir) {
04333           if (term->Max() == 0) {
04334             // Premise is wrong. The clause will never apply.
04335             return;
04336           }
04337           if (term->Min() == 0) {
04338             DCHECK_EQ(1, term->Max());
04339             // Premise may be true. Adding to guard vector.
04340             guard.push_back(term);
04341           }
04342         }
04343         ++tmp;
04344         ++tmp_dir;
04345       }
04346       guard.push_back(clauses_[index].LastValue());
04347       directions_[index].SetLastValue(true);
04348       // Given premises: xi = ai
04349       // and a term y != b
04350       // The following is equivalent to
04351       // And(xi == a1) => y != b.
04352       ct = solver()->MakeEquality(solver()->MakeMin(guard), Zero());
04353     }
04354     DCHECK(ct != NULL);
04355     solver()->AddConstraint(ct);
04356   }
04357 
04358   void AddTermToClause(SymmetryBreaker* const visitor, IntVar* const term) {
04359     clauses_[visitor->index_in_symmetry_manager()].Push(solver(), term);
04360   }
04361 
04362  private:
04363   scoped_array<SymmetryBreaker*> visitors_;
04364   scoped_array<SimpleRevFIFO<IntVar*> > clauses_;
04365   scoped_array<SimpleRevFIFO<Decision*> > decisions_;
04366   scoped_array<SimpleRevFIFO<bool> > directions_;
04367   const int size_;
04368 };
