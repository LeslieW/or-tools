00633     if (top()->HasIntegerExpression(ModelVisitor::kLeftArgument)) {
00634       MPConstraint* const ct = mp_solver_->MakeRowConstraint(0.0, 0.0);
00635       const IntExpr* const left =
00636           top()->FindIntegerExpressionArgumentOrDie(
00637               ModelVisitor::kLeftArgument);
00638       const IntExpr* const right =
00639           top()->FindIntegerExpressionArgumentOrDie(
00640               ModelVisitor::kRightArgument);
00641       ct->SetCoefficient(Translated(left), 1.0);
00642       ct->SetCoefficient(Translated(right), -1.0);
00643       RegisterExpression(cp_expr);
00644       ct->SetCoefficient(Translated(cp_expr), -1.0);
00645     } else {
00646       const IntExpr* const expr =
00647           top()->FindIntegerExpressionArgumentOrDie(
00648               ModelVisitor::kExpressionArgument);
00649       const int64 value =
00650           top()->FindIntegerArgumentOrDie(ModelVisitor::kValueArgument);
00651       MPConstraint* const ct = mp_solver_->MakeRowConstraint(value, value);
00652       ct->SetCoefficient(Translated(expr), 1.0);
00653       RegisterExpression(cp_expr);
00654       ct->SetCoefficient(Translated(cp_expr), 1.0);
00655     }
00656   }
00657 
00658   void VisitOpposite(const IntExpr* const cp_expr) {
00659     const IntExpr* const expr =
00660         top()->FindIntegerExpressionArgumentOrDie(
00661             ModelVisitor::kExpressionArgument);
00662     MPConstraint* const ct = mp_solver_->MakeRowConstraint(0.0, 0.0);
00663     ct->SetCoefficient(Translated(expr), 1.0);
00664     RegisterExpression(cp_expr);
00665     ct->SetCoefficient(Translated(cp_expr), -1.0);
00666   }
00667 
00668   void VisitProduct(const IntExpr* const cp_expr) {
00669     if (top()->HasIntegerExpression(ModelVisitor::kExpressionArgument)) {
00670       const IntExpr* const expr =
00671           top()->FindIntegerExpressionArgumentOrDie(
00672               ModelVisitor::kExpressionArgument);
00673       const int64 value =
00674           top()->FindIntegerArgumentOrDie(ModelVisitor::kValueArgument);
00675       MPConstraint* const ct = mp_solver_->MakeRowConstraint(0.0, 0.0);
00676       ct->SetCoefficient(Translated(expr), value);
00677       RegisterExpression(cp_expr);
00678       ct->SetCoefficient(Translated(cp_expr), -1.0);
00679     } else {
00680       RegisterExpression(cp_expr);
00681     }
00682   }
00683 
00684   void VisitIntegerExpression(const IntExpr* const cp_expr) {
00685     RegisterExpression(cp_expr);
00686   }
00687 
00688   void VisitObjective() {
00689     *maximize_ =
00690         top()->FindIntegerArgumentOrDie(ModelVisitor::kMaximizeArgument);
00691     *objective_ =
00692         const_cast<IntExpr*>(top()->FindIntegerExpressionArgumentOrDie(
00693             ModelVisitor::kExpressionArgument))->Var();
00694     mp_solver_->SetObjectiveCoefficient(Translated(*objective_), 1.0);
00695     mp_solver_->SetOptimizationDirection(*maximize_);
00696   }

00698   MPSolver* const mp_solver_;
00699   ExprTranslation* const translation_;
00700   IntVar** objective_;
00701   bool* maximize_;
00702   std::vector<ArgumentHolder*> holders_;
00703   std::vector<ArgumentHolder*> extensions_;
00704   std::vector<bool> actives_;
00705 };

00709 class AutomaticLinearization : public SearchMonitor {
00710  public:
00711   AutomaticLinearization(Solver* const solver, int frequency)
00712     : SearchMonitor(solver),
00713       mp_solver_("InSearchSimplex", MPSolver::CLP_LINEAR_PROGRAMMING),
00714       counter_(0),
00715       simplex_frequency_(frequency),
00716       objective_(NULL),
00717       maximize_(false) {}
00718 
00719   virtual ~AutomaticLinearization() {}
00720 
00721   virtual void BeginInitialPropagation() {
00722     mp_solver_.Clear();
00723     translation_.clear();
00724     BuildModel();
00725   }
00726 
00727   virtual void EndInitialPropagation() {
00728     RunOptim();
00729   }
00730 
00731   virtual void BeginNextDecision(DecisionBuilder* const b) {
00732     if (++counter_ % simplex_frequency_ == 0) {
00733       RunOptim();
00734     }
00735   }
00736 
00737   void RunOptim() {
00738     AssignVariables();
00739     SolveProblem();
00740   }
00741 
00742   void BuildModel() {
00743     Linearizer linearizer(&mp_solver_, &translation_, &objective_, &maximize_);
00744     solver()->Accept(&linearizer);
00745   }
00746 
00747   void AssignVariables() {
00748     for (ConstIter<ExprTranslation> it(translation_); !it.at_end(); ++it) {
00749       it->second->SetBounds(it->first->Min(), it->first->Max());
00750     }
00751   }
00752 
00753   void SolveProblem() {
00754     if (objective_ != NULL) {
00755       switch (mp_solver_.Solve()) {
00756         case MPSolver::OPTIMAL: {
00757           const double obj_value = mp_solver_.objective_value();
00758           if (maximize_) {
00759             const int64 int_obj_value = static_cast<int64>(ceil(obj_value));
00760             objective_->SetMax(int_obj_value);
00761           } else {
00762             const int64 int_obj_value = static_cast<int64>(floor(obj_value));
00763             objective_->SetMin(int_obj_value);
00764           }
00765           break;
00766         }
00767         case MPSolver::FEASIBLE:
00768           break;
00769         case MPSolver::INFEASIBLE:
00770           solver()->Fail();
00771           break;
00772         case MPSolver::UNBOUNDED:
00773           LOG(INFO) << "Error: unbounded LP status.";
00774           break;
00775         case MPSolver::ABNORMAL:
00776           LOG(INFO) << "Error: abnormal LP status.";
00777           break;
00778         default:
00779           LOG(FATAL) << "Error: Unknown LP status.";
00780           break;
00781       }
00782     }
00783   }
00784 
00785  private:
00786   MPSolver mp_solver_;
00787   int64 counter_;
00788   const int simplex_frequency_;
00789   ExprTranslation translation_;
00790   IntVar* objective_;
00791   bool maximize_;
00792 };
