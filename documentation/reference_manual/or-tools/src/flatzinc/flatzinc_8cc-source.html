00189 void SortVariableByDegree(Solver* const solver,
00190                           std::vector<IntVar*>* const int_vars) {
00191   hash_map<IntVar*, int> degree_map;
00192   for (int i = 0; i < int_vars->size(); ++i) {
00193     degree_map[(*int_vars)[i]] = 0;
00194   }
00195   ModelVisitor* const degree_visitor =
00196       solver->MakeVariableDegreeVisitor(&degree_map);
00197   solver->Accept(degree_visitor);
00198   std::vector<VarDegreeIndex> to_sort;
00199   for (int i = 0; i < int_vars->size(); ++i) {
00200     IntVar* const var = (*int_vars)[i];
00201     to_sort.push_back(VarDegreeIndex(var, degree_map[var], i));
00202   }
00203   std::sort(to_sort.begin(), to_sort.end());
00204   for (int i = 0; i < int_vars->size(); ++i) {
00205     (*int_vars)[i] = to_sort[i].v;
00206   }

00588 string FlatZincModel::DebugString(AstNode* const ai) const {
00589   string output;
00590   int k;
00591   if (ai->isArray()) {
00592     AstArray* aia = ai->getArray();
00593     int size = aia->a.size();
00594     output += "[";
00595     for (int j = 0; j < size; j++) {
00596       output += DebugString(aia->a[j]);
00597       if (j < size - 1) {
00598         output += ", ";
00599       }
00600     }
00601     output += "]";
00602   } else if (ai->isInt(k)) {
00603     output += StringPrintf("%d", k);
00604   } else if (ai->isIntVar()) {
00605     IntVar* const var = integer_variables_[ai->getIntVar()]->Var();
00606     output += StringPrintf("%d", var->Value());
00607   } else if (ai->isBoolVar()) {
00608     IntVar* const var = boolean_variables_[ai->getBoolVar()]->Var();
00609     output += var->Value() ? "true" : "false";
00610   } else if (ai->isSetVar()) {
00611     SetVar* const var = set_variables_[ai->getSetVar()];
00612     output += var->DebugString();
00613   } else if (ai->isBool()) {
00614     output += (ai->getBool() ? "true" : "false");
00615   } else if (ai->isSet()) {
00616     AstSetLit* s = ai->getSet();
00617     if (s->interval) {
00618       output += StringPrintf("%d..%d", s->imin, s->imax);
00619     } else {
00620       output += "{";
00621       for (unsigned int i=0; i<s->s.size(); i++) {
00622         output += StringPrintf("%d%s",
00623                                s->s[i],
00624                                (i < s->s.size()-1 ? ", " : "}"));
00625       }
00626     }
00627   } else if (ai->isString()) {
00628     string s = ai->getString();
00629     for (unsigned int i = 0; i < s.size(); i++) {
00630       if (s[i] == '\\' && i < s.size() - 1) {
00631         switch (s[i+1]) {
00632           case 'n': output += "\n"; break;
00633           case '\\': output += "\\"; break;
00634           case 't': output += "\t"; break;
00635           default: {
00636             output += "\\";
00637             output += s[i+1];
00638           }
00639         }
00640         i++;
00641       } else {
00642         output += s[i];
00643       }
00644     }
00645   }
00646   return output;
00647 }
