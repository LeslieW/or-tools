00351 // variables. Its main output is to find the optimal pair (variable/value)

00354  public:

00363                  const std::vector<IntVar*>& vars,
00364                  DefaultPhaseParameters::DisplayLevel display_level)
00365       : SearchMonitor(solver),
00366         domain_watcher_(domain_watcher),
00367         vars_(vars),
00368         size_(vars.size()),
00369         current_log_space_(0.0),
00370         impacts_(size_),
00371         original_min_(size_, 0LL),
00372         domain_iterators_(new IntVarIterator*[size_]),
00373         display_level_(display_level),
00374         current_var_(kUninitializedVarIndex),
00375         current_value_(0),
00376         init_done_(false) {
00377     for (int i = 0; i < size_; ++i) {
00378       domain_iterators_[i] = vars_[i]->MakeDomainIterator(true);
00379       var_map_[vars_[i]] = i;
00380     }
00381   }

00383   virtual void ApplyDecision(Decision* const d) {
00384     if (!init_done_) {
00385       return;
00386     }
00387     d->Accept(&find_var_);
00388     if (find_var_.valid()) {
00389       current_var_ = var_map_[find_var_.var()];
00390       current_value_ = find_var_.value();
00391       current_log_space_ = domain_watcher_->LogSearchSpaceSize();
00392     } else {
00393       current_var_ = kUninitializedVarIndex;
00394       current_value_ = 0;
00395     }
00396   }
00397 
00398   virtual void AfterDecision(Decision* const d, bool apply) {
00399     if (init_done_ && current_var_ != kUninitializedVarIndex) {
00400       CHECK_GT(current_log_space_, 0.0);
00401       const double log_space = domain_watcher_->LogSearchSpaceSize();
00402       if (apply) {
00403         const double impact = kPerfectImpact - log_space / current_log_space_;
00404         UpdateImpact(current_var_, current_value_, impact);
00405         current_var_ = kUninitializedVarIndex;
00406         current_value_ = 0;
00407       }
00408       current_log_space_ = log_space;
00409     }
00410   }

00413     if (init_done_ && current_var_ != kUninitializedVarIndex) {
00414       UpdateImpact(current_var_, current_value_, kFailureImpact);
00415       current_var_ = kUninitializedVarIndex;
00416       current_value_ = 0;
00417     }
00418   }

00420   void ResetAllImpacts() {
00421     for (int i = 0; i < size_; ++i) {
00422       original_min_[i] = vars_[i]->Min();
00423       // By default, we init impacts to 2.0 -> equivalent to failure.
00424       // This will be overwritten to real impact values on valid domain
00425       // values during the FirstRun() method.
00426       impacts_[i].resize(vars_[i]->Max() - vars_[i]->Min() + 1,
00427                          kInitFailureImpact);

00431       for (int j = 0; j < impacts_[i].size(); ++j) {
00432         impacts_[i][j] = kInitFailureImpact;

00434     }
00435   }

00437   void UpdateImpact(int var_index, int64 value, double impact) {
00438     const int64 value_index = value - original_min_[var_index];
00439     const double current_impact = impacts_[var_index][value_index];
00440     const double new_impact =
00441         (current_impact * (FLAGS_cp_impact_divider - 1) + impact) /

00447     const double log_space = domain_watcher_->LogSearchSpaceSize();
00448     const double impact = kPerfectImpact - log_space / current_log_space_;

00450     DCHECK_LT(var_index, size_);

00456   void FirstRun(int64 splits) {
00457     Solver* const s = solver();
00458     current_log_space_ = domain_watcher_->LogSearchSpaceSize();
00459     if (display_level_ != DefaultPhaseParameters::NONE) {
00460       LOG(INFO) << "  - initial log2(SearchSpace) = " << current_log_space_;
00461     }
00462     const int64 init_time = s->wall_time();
00463     ResetAllImpacts();
00464     int64 removed_counter = 0;
00465     FirstRunVariableContainers* container = s->RevAlloc(
00466         new FirstRunVariableContainers(this, splits));
00467     // Loop on the variables, scan domains and initialize impacts.
00468     for (int var_index = 0; var_index < size_; ++var_index) {
00469       IntVar* const var = vars_[var_index];
00470       if (var->Bound()) {
00471         continue;
00472       }
00473       IntVarIterator* const iterator = domain_iterators_[var_index];
00474       DecisionBuilder* init_decision_builder = NULL;
00475       if (var->Max() - var->Min() < splits) {
00476         // The domain is small enough, we scan it completely.
00477         container->without_split()->set_update_impact_callback(
00478             container->update_impact_callback());
00479         container->without_split()->Init(var, iterator, var_index);
00480         init_decision_builder = container->without_split();
00481       } else {
00482         // The domain is too big, we scan it in initialization_splits
00483         // intervals.
00484         container->with_splits()->set_update_impact_callback(
00485             container->update_impact_callback());
00486         container->with_splits()->Init(var, iterator, var_index);
00487         init_decision_builder = container->with_splits();
00488       }
00489       // Reset the number of impacts initialized.
00490       init_count_ = 0;
00491       // Use Solve() to scan all values of one variable.
00492       s->Solve(init_decision_builder);
00493 
00494       // If we have not initialized all values, then they can be removed.
00495       // As the iterator is not stable w.r.t. deletion, we need to store
00496       // removed values in an intermediate vector.
00497       if (init_count_ != var->Size()) {
00498         container->ClearRemovedValues();
00499         for (iterator->Init(); iterator->Ok(); iterator->Next()) {
00500           const int64 value = iterator->Value();
00501           const int64 value_index = value - original_min_[var_index];
00502           if (impacts_[var_index][value_index] == kInitFailureImpact) {
00503             container->PushBackRemovedValue(value);
00504           }
00505         }
00506         CHECK(container->HasRemovedValues()) << var->DebugString();
00507         removed_counter += container->NumRemovedValues();
00508         const double old_log = domain_watcher_->Log2(var->Size());
00509         var->RemoveValues(container->removed_values());
00510         current_log_space_ += domain_watcher_->Log2(var->Size()) - old_log;
00511       }
00512     }
00513     if (display_level_ != DefaultPhaseParameters::NONE) {
00514       if (removed_counter) {
00515         LOG(INFO) << "  - init done, time = "
00516                   << s->wall_time() - init_time
00517                   << " ms, " << removed_counter
00518                   << " values removed, log2(SearchSpace) = "
00519                   << current_log_space_;
00520       } else {
00521         LOG(INFO) << "  - init done, time = "
00522                   << s->wall_time() - init_time << " ms";
00523       }
00524     }
00525     s->SaveAndSetValue(&init_done_, true);
00526   }

00528   // This method scans the domain of one variable and returns the sum
00529   // of the impacts of all values in its domain, along with the value
00530   // with minimal impact.
00531   void ScanVarImpacts(int var_index,
00532                       int64* const best_impact_value,
00533                       double* const var_impacts,
00534                       DefaultPhaseParameters::VariableSelection var_select,
00535                       DefaultPhaseParameters::ValueSelection value_select) {
00536     CHECK_NOTNULL(best_impact_value);
00537     CHECK_NOTNULL(var_impacts);
00538     double max_impact = -std::numeric_limits<double>::max();
00539     double min_impact = std::numeric_limits<double>::max();
00540     double sum_var_impact = 0.0;
00541     int64 min_impact_value = -1;
00542     int64 max_impact_value = -1;
00543     IntVarIterator* const it = domain_iterators_[var_index];
00544     for (it->Init(); it->Ok(); it->Next()) {
00545       const int64 value = it->Value();
00546       const int64 value_index = value - original_min_[var_index];
00547       DCHECK_LT(var_index, size_);
00548       DCHECK_LT(value_index, impacts_[var_index].size());
00549       const double current_impact = impacts_[var_index][value_index];
00550       sum_var_impact += current_impact;
00551       if (current_impact > max_impact) {
00552         max_impact = current_impact;
00553         max_impact_value = value;
00554       }
00555       if (current_impact < min_impact) {
00556         min_impact = current_impact;
00557         min_impact_value = value;
00558       }
00559     }
00560 
00561     switch (var_select) {
00562       case DefaultPhaseParameters::CHOOSE_MAX_AVERAGE_IMPACT: {
00563         *var_impacts = sum_var_impact / vars_[var_index]->Size();
00564         break;
00565       }
00566       case DefaultPhaseParameters::CHOOSE_MAX_VALUE_IMPACT: {
00567         *var_impacts = max_impact;
00568         break;
00569       }
00570       default: {
00571         *var_impacts = sum_var_impact;
00572         break;
00573       }
00574     }
00575 
00576     switch (value_select) {
00577       case DefaultPhaseParameters::SELECT_MIN_IMPACT: {
00578         *best_impact_value = min_impact_value;
00579         break;
00580       }
00581       case DefaultPhaseParameters::SELECT_MAX_IMPACT: {
00582         *best_impact_value = max_impact_value;
00583         break;
00584       }
00585     }
00586   }

00680 // Hook on the search to check restart before the refutation of a decision.
00681 class RestartMonitor : public SearchMonitor {
00682  public:
00683   RestartMonitor(Solver* const solver,
00684                  DefaultPhaseParameters parameters,
00685                  DomainWatcher* const domain_watcher_)
00686       : SearchMonitor(solver),
00687         parameters_(parameters),
00688         domain_watcher_(domain_watcher_),
00689         min_log_search_space_(std::numeric_limits<double>::infinity()),
00690         no_good_manager_(parameters_.restart_log_size >= 0 &&
00691                          parameters_.use_no_goods ?
00692                          solver->MakeNoGoodManager() :
00693                          NULL),
00694         branches_between_restarts_(0),
00695         min_restart_period_(ComputeBranchRestart(parameters_.restart_log_size)),
00696         maximum_restart_depth_(kint64max) {}
00697 
00698   virtual ~RestartMonitor() {}
00699 
00700   virtual void ApplyDecision(Decision* const d) {
00701     branches_between_restarts_++;
00702     d->Accept(&find_var_);
00703     if (find_var_.valid()) {
00704       choices_.Push(solver(),
00705                     ChoiceInfo(find_var_.var(), find_var_.value(), true));
00706     }
00707   }
00708 
00709   virtual void RefuteDecision(Decision* const d) {
00710     CHECK_NOTNULL(d);
00711     Solver* const s = solver();
00712     branches_between_restarts_++;
00713     d->Accept(&find_var_);
00714     if (find_var_.valid()) {
00715       choices_.Push(solver(),
00716                     ChoiceInfo(find_var_.var(), find_var_.value(), false));
00717     }
00718     if (CheckRestartOnRefute(s)) {
00719       DoRestartAndAddNoGood(s);
00720       RestartCurrentSearch();
00721       s->Fail();
00722     }
00723   }
00724 
00725   virtual void ExitSearch() {
00726     if (parameters_.display_level != DefaultPhaseParameters::NONE &&
00727         no_good_manager_ != NULL) {
00728       LOG(INFO) << "Default search has generated "
00729                 << no_good_manager_->NoGoodCount() << " no goods";
00730     }
00731   }
00732 
00733   virtual bool AtSolution() {
00734     if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00735       LOG(INFO) << "Found a solution after the following decisions:";
00736       for (SimpleRevFIFO<ChoiceInfo>::Iterator it(&choices_); it.ok(); ++it) {
00737         LOG(INFO) << "  " << (*it).DebugString();
00738       }
00739     }
00740     return false;
00741   }
00742 
00743   void Install() {
00744     SearchMonitor::Install();
00745     if (no_good_manager_ != NULL) {
00746       no_good_manager_->Install();
00747     }
00748   }
00749 
00750  private:
00751   // Called before applying the refutation of the decision.  This
00752   // method must decide if we need to restart or not.  The main
00753   // decision is based on the restart_log_size and the current search
00754   // space size. If, the current search space size is greater than the
00755   // min search space size encountered since the last restart +
00756   // restart_log_size, this means that we have just finished a search
00757   // tree of size at least restart_log_size.  If the search tree is
00758   // very sparse, then we may have visited close to restart_log_size
00759   // nodes since the last restart (instead of the maximum of
00760   // 2^restart_log_size). To fight this degenerate case, we also count
00761   // the number of branches explored since the last restart and decide
00762   // to postpone restart until we finish a sub-search tree and have
00763   // visited enough branches (2^restart_log_size). To enforce this,
00764   // when we postpone a restart, we store the current search depth -1
00765   // in maximum_restart_depth_ and will restart as soon as we reach a
00766   // node above the current one, with enough visited branches.
00767   bool CheckRestartOnRefute(Solver* const solver) {
00768     // We do nothing if restart_log_size is < 0.
00769     if (parameters_.restart_log_size >= 0) {
00770       const int search_depth = solver->SearchDepth();
00771       const double log_search_space_size =
00772           domain_watcher_->LogSearchSpaceSize();
00773       if (min_log_search_space_ > log_search_space_size) {
00774         min_log_search_space_ = log_search_space_size;
00775       }
00776 
00777       // Some verbose display.
00778       if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00779         LOG(INFO) << "search_depth = " << search_depth
00780                   << ", branches between restarts = "
00781                   << branches_between_restarts_
00782                   << ", log_search_space_size = " << log_search_space_size
00783                   << ", min_log_search_space = " << min_log_search_space_;
00784       }
00785       if (search_depth > maximum_restart_depth_) {
00786         // We are deeper than maximum_restart_depth_, we should not restart
00787         // because we have not finished a sub-tree of sufficient size.
00788         return false;
00789       }
00790       // We may restart either because of the search space criteria,
00791       // or the search depth is less than maximum_restart_depth_.
00792       if (min_log_search_space_ + parameters_.restart_log_size <
00793           log_search_space_size ||
00794           (search_depth <= maximum_restart_depth_ &&
00795            maximum_restart_depth_ != kint64max)) {
00796         // If we have not visited enough branches, we postpone the
00797         // restart and force to check at least at the parent of the
00798         // current search node.
00799         if (branches_between_restarts_ < min_restart_period_) {
00800           maximum_restart_depth_ = search_depth - 1;
00801           if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00802             LOG(INFO) << "Postpone restarting until depth <= "
00803                       << maximum_restart_depth_ << ", visited nodes = "
00804                       << branches_between_restarts_
00805                       << " / " << min_restart_period_;
00806           }
00807           return false;
00808         }
00809         return true;
00810       }
00811     }
00812     return false;
00813   }
00814 
00815   // Performs the restart. It resets various counters and adds a
00816   // non-reversible nogood if need be.
00817   void DoRestartAndAddNoGood(Solver* const solver) {
00818     const int search_depth = solver->SearchDepth();
00819     if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00820       LOG(INFO) << "Restarting at depth " << search_depth;
00821     }
00822     min_log_search_space_ = std::numeric_limits<double>::infinity();
00823     branches_between_restarts_ = 0;
00824     maximum_restart_depth_ = kint64max;
00825     // Creates nogood.
00826     if (parameters_.use_no_goods) {
00827       DCHECK(no_good_manager_ != NULL);
00828       NoGood* const nogood = no_good_manager_->MakeNoGood();
00829       // if the nogood contains both x == 3 and x != 4, we can simplify
00830       // to keep only x == 3.
00831       hash_set<IntVar*> positive_variable_indices;
00832       for (SimpleRevFIFO<ChoiceInfo>::Iterator it(&choices_);
00833            it.ok();
00834            ++it) {
00835         const ChoiceInfo& choice = *it;
00836         if (choice.left()) {
00837           positive_variable_indices.insert(choice.var());
00838         }
00839       }
00840       // We fill the nogood structure.
00841       for (SimpleRevFIFO<ChoiceInfo>::Iterator it(&choices_);
00842            it.ok();
00843            ++it) {
00844         const ChoiceInfo& choice = *it;
00845         IntVar* const var = choice.var();
00846         const int64 value = choice.value();
00847         if (choice.left()) {
00848           nogood->AddIntegerVariableEqualValueTerm(var, value);
00849         } else if (!ContainsKey(positive_variable_indices, choice.var())) {
00850           nogood->AddIntegerVariableNotEqualValueTerm(var, value);
00851         }
00852       }
00853       if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
00854         LOG(INFO) << "Adding no good no " << no_good_manager_->NoGoodCount()
00855                   << ": " << nogood->DebugString();
00856       }
00857       // Adds the nogood to the nogood manager.
00858       no_good_manager_->AddNoGood(nogood);
00859     }
00860   }
00861 
00862   const DefaultPhaseParameters parameters_;
00863   DomainWatcher* const domain_watcher_;
00864   double min_log_search_space_;
00865   NoGoodManager* no_good_manager_;
00866   int64 branches_between_restarts_;
00867   const int64 min_restart_period_;
00868   int64 maximum_restart_depth_;
00869   SimpleRevFIFO<ChoiceInfo> choices_;
00870   FindVar find_var_;
00871 };

01111   void CheckInit(Solver* const solver) {
01112     if (init_done_) {
01113       return;
01114     }
01115     if (parameters_.use_impacts) {
01116       // Decide if we are doing impacts, no if one variable is too big.
01117       for (int i = 0; i < vars_.size(); ++i) {
01118         if (vars_[i]->Max() - vars_[i]->Min() > 0xFFFFFF) {
01119           if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
01120             LOG(INFO) << "Domains are too large, switching to simple "
01121                       << "heuristics";
01122           }
01123           parameters_.use_impacts = false;
01124           init_done_ = true;
01125           return;
01126         }
01127       }
01128       // No if the search space is too small.
01129       if (domain_watcher_.LogSearchSpaceSize() < 10) {
01130         if (parameters_.display_level == DefaultPhaseParameters::VERBOSE) {
01131           LOG(INFO) << "Search space is too small, switching to simple "
01132                     << "heuristics";
01133         }
01134         parameters_.use_impacts = false;
01135         init_done_ = true;
01136         return;
01137       }
01138 
01139       if (parameters_.display_level != DefaultPhaseParameters::NONE) {
01140         LOG(INFO) << "Init impact based search phase on " << vars_.size()
01141                   << " variables, initialization splits = "
01142                   << parameters_.initialization_splits
01143                   << ", heuristic_period = " << parameters_.heuristic_period
01144                   << ", run_all_heuristics = "
01145                   << parameters_.run_all_heuristics
01146                   << ", restart_log_size = " << parameters_.restart_log_size;
01147       }
01148       // We need to reset the impacts because FirstRun calls RemoveValues
01149       // which can result in a Fail() therefore calling this method again.
01150       impact_recorder_.FirstRun(parameters_.initialization_splits);
01151     }
01152     if (parameters_.persistent_impact) {
01153       init_done_ = true;
01154     } else {
01155       solver->SaveAndSetValue(&init_done_, true);
01156     }
01157   }

01159   bool FindVarValue(IntVar** const var, int64* const value) {
01160     return parameters_.use_impacts ?
01161         FindVarValueWithImpact(var, value) :
01162         FindVarValueNoImpact(var, value);
01163   }
01164 
01165   // This method will do an exhaustive scan of all domains of all
01166   // variables to select the variable with the maximal sum of impacts
01167   // per value in its domain, and then select the value with the
01168   // minimal impact.
01169   bool FindVarValueWithImpact(IntVar** const var, int64* const value) {
01170     CHECK_NOTNULL(var);
01171     CHECK_NOTNULL(value);
01172     *var = NULL;
01173     *value = 0;
01174     double best_var_impact = -std::numeric_limits<double>::max();
01175     for (int i = 0; i < vars_.size(); ++i) {
01176       if (!vars_[i]->Bound()) {
01177         int64 current_value = 0;
01178         double current_var_impact = 0.0;
01179         impact_recorder_.ScanVarImpacts(i,
01180                                         &current_value,
01181                                         &current_var_impact,
01182                                         parameters_.var_selection_schema,
01183                                         parameters_.value_selection_schema);
01184         if (current_var_impact > best_var_impact) {
01185           *var = vars_[i];
01186           *value = current_value;
01187           best_var_impact = current_var_impact;
01188         }
01189       }
01190     }
01191     return (*var != NULL);
01192   }
01193 
01194   bool FindVarValueNoImpact(IntVar** const var, int64* const value) {
01195     CHECK_NOTNULL(var);
01196     CHECK_NOTNULL(value);
01197     *var = NULL;
01198     *value = 0;
01199     for (int i = 0; i < vars_.size(); ++i) {
01200       if (!vars_[i]->Bound()) {
01201         *var = vars_[i];;
01202         *value = vars_[i]->Min();
01203         return true;
01204       }
01205     }
01206     return false;
01207   }
