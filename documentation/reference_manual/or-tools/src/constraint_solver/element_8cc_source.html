00137 void BaseIntExprElement::UpdateSupports() const {

00140       || !expr_->Contains(max_support_)) {
00141     const int64 emin = ExprMin();
00142     const int64 emax = ExprMax();
00143     int64 min_value = ElementValue(emax);
00144     int64 max_value = min_value;
00145     int min_support = emax;
00146     int max_support = emax;
00147     const int64 expr_size = expr_->Size();
00148     if (expr_size > 1) {
00149       if (expr_size == emax - emin + 1) {
00150         // Value(emax) already stored in min_value, max_value.
00151         for (int64 index = emin; index < emax; ++index) {
00152           const int64 value = ElementValue(index);
00153           if (value > max_value) {
00154             max_value = value;
00155             max_support = index;
00156           } else if (value < min_value) {
00157             min_value = value;
00158             min_support = index;
00159           }
00160         }
00161       } else {
00162         for (expr_iterator_->Init();
00163              expr_iterator_->Ok();
00164              expr_iterator_->Next()) {
00165           const int64 index = expr_iterator_->Value();
00166           if (index >= emin && index <= emax) {
00167             const int64 value = ElementValue(index);
00168             if (value > max_value) {
00169               max_value = value;
00170               max_support = index;
00171             } else if (value < min_value) {
00172               min_value = value;
00173               min_support = index;
00174             }
00175           }
00176         }
00177       }
00178     }

00182     s->SaveAndSetValue(&max_, max_value);

00503 // ----- Solver::MakeElement(int array, int var) -----
00504 IntExpr* BuildElement(Solver* const solver,
00505                       ConstIntArray* values,
00506                       IntVar* const index) {
00507   // Various checks.
00508   // Is array constant?
00509   if (values->HasProperty(ConstIntArray::IS_CONSTANT)) {
00510     solver->AddConstraint(solver->MakeBetweenCt(index, 0, values->size() - 1));
00511     return solver->MakeIntConst(values->get(0));
00512   }
00513   // Is array built with booleans only?
00514   // TODO(user): We could maintain the index of the first one.
00515   if (values->HasProperty(ConstIntArray::IS_BOOLEAN)) {
00516     std::vector<int64> ones;
00517     int first_zero = -1;
00518     for (int i = 0; i < values->size(); ++i) {
00519       if (values->get(i) == 1LL) {
00520         ones.push_back(i);
00521       } else {
00522         first_zero = i;
00523       }
00524     }
00525     if (ones.size() == 1) {
00526       DCHECK_EQ(1LL, values->get(ones.back()));
00527       solver->AddConstraint(
00528           solver->MakeBetweenCt(index, 0, values->size() - 1));
00529       return solver->MakeIsEqualCstVar(index, ones.back());
00530     } else if (ones.size() == values->size() - 1) {
00531       solver->AddConstraint(
00532           solver->MakeBetweenCt(index, 0, values->size() - 1));
00533       return solver->MakeIsDifferentCstVar(index, first_zero);
00534     } else if (ones.size() == ones.back() - ones.front() + 1) {  // contiguous.
00535       solver->AddConstraint(
00536           solver->MakeBetweenCt(index, 0, values->size() - 1));
00537       IntVar* const b = solver->MakeBoolVar("ContiguousBooleanElementVar");
00538       solver->AddConstraint(
00539           solver->MakeIsBetweenCt(index, ones.front(), ones.back(), b));
00540       return b;
00541     } else {
00542       IntVar* const b = solver->MakeBoolVar("NonContiguousBooleanElementVar");
00543       solver->AddConstraint(
00544           solver->MakeBetweenCt(index, 0, values->size() - 1));
00545       solver->AddConstraint(solver->MakeIsMemberCt(index, ones, b));
00546       return b;
00547     }
00548   }
00549   IntExpr* const cache = solver->Cache()->FindVarConstantArrayExpression(
00550       index,
00551       values,
00552       ModelCache::VAR_CONSTANT_ARRAY_ELEMENT);
00553   if (cache != NULL) {
00554     return cache;
00555   } else {
00556     IntExpr* result= NULL;
00557     // Is Array increasing
00558     if (values->HasProperty(ConstIntArray::IS_INCREASING)) {
00559     result = solver->RegisterIntExpr(solver->RevAlloc(
00560         new IncreasingIntExprElement(solver, values->Copy(), index)));
00561     } else {
00562       result = solver->RegisterIntExpr(solver->RevAlloc(
00563           new IntExprElement(solver, values->Copy(), index)));
00564     }
00565     solver->Cache()->InsertVarConstantArrayExpression(
00566         result,
00567         index,
00568         values,
00569         ModelCache::VAR_CONSTANT_ARRAY_ELEMENT);
00570     return result;
00571   }

01143 void IntExprArrayElementCt::Propagate() {
01144   const int64 emin = std::max(0LL, index_->Min());
01145   const int64 emax = std::min(size_ - 1LL, index_->Max());
01146   const int64 vmin = target_var_->Min();
01147   const int64 vmax = target_var_->Max();
01148   if (emin == emax) {
01149     index_->SetValue(emin);  // in case it was reduced by the above min/max.
01150     vars_[emin]->SetRange(vmin, vmax);
01151   } else {
01152     int64 nmin = emin;
01153     while (nmin <= emax && (vars_[nmin]->Min() > vmax ||
01154                             vars_[nmin]->Max() < vmin)) {
01155       nmin++;
01156     }
01157     int64 nmax = emax;
01158     while (nmax >= nmin && (vars_[nmax]->Max() < vmin ||
01159                             vars_[nmax]->Min() > vmax)) {
01160       nmax--;
01161     }
01162     index_->SetRange(nmin, nmax);
01163     if (nmin == nmax) {
01164       vars_[nmin]->SetRange(vmin, vmax);
01165     }
01166   }
01167   if (min_support_ == -1 || max_support_ == -1) {
01168     int min_support = -1;
01169     int max_support = -1;
01170     int64 gmin = kint64max;
01171     int64 gmax = kint64min;
01172     for (int i = index_->Min(); i <= index_->Max(); ++i) {
01173       const int64 vmin = vars_[i]->Min();
01174       if (vmin < gmin) {
01175         gmin = vmin;
01176       }
01177       const int64 vmax = vars_[i]->Max();
01178       if (vmax > gmax) {
01179         gmax = vmax;
01180       }
01181     }
01182     solver()->SaveAndSetValue(&min_support_, min_support);
01183     solver()->SaveAndSetValue(&max_support_, max_support);
01184     target_var_->SetRange(gmin, gmax);
01185   }

01306 class IntExprArrayPositionCt : public Constraint {
01307  public:
01308   IntExprArrayPositionCt(Solver* const s,
01309                          const std::vector<IntVar*>& vars,
01310                          IntVar* const index,
01311                          int64 target)
01312       : Constraint(s),
01313         vars_(vars),
01314         size_(vars.size()),
01315         index_(index),
01316         target_(target),
01317         demons_(size_),
01318         index_iterator_(index->MakeHoleIterator(true)) {}
01319 
01320   virtual ~IntExprArrayPositionCt() {}
01321 
01322   virtual void Post() {
01323     for (int i = 0; i < size_; ++i) {
01324       demons_[i] = MakeConstraintDemon1(solver(),
01325                                         this,
01326                                         &IntExprArrayPositionCt::Propagate,
01327                                         "Propagate",
01328                                         i);
01329       vars_[i]->WhenDomain(demons_[i]);
01330     }
01331     Demon* const index_demon =
01332         MakeConstraintDemon0(solver(),
01333                              this,
01334                              &IntExprArrayPositionCt::PropagateIndex,
01335                              "PropagateIndex");
01336     index_->WhenDomain(index_demon);
01337   }
01338 
01339   virtual void InitialPropagate() {
01340     for (int i = 0; i < size_; ++i) {
01341       if (!index_->Contains(i)) {
01342         vars_[i]->RemoveValue(target_);
01343       } else if (!vars_[i]->Contains(target_)) {
01344         index_->RemoveValue(i);
01345         demons_[i]->inhibit(solver());
01346       } else if (vars_[i]->Bound()) {
01347         index_->SetValue(i);
01348         demons_[i]->inhibit(solver());
01349       }
01350     }
01351   }
01352 
01353   void Propagate(int index) {
01354     if (!vars_[index]->Contains(target_)) {
01355       index_->RemoveValue(index);
01356       demons_[index]->inhibit(solver());
01357     } else if (vars_[index]->Bound()) {
01358       index_->SetValue(index);
01359     }
01360   }
01361 
01362   void PropagateIndex() {
01363     const int64 oldmax = index_->OldMax();
01364     const int64 vmin = index_->Min();
01365     const int64 vmax = index_->Max();
01366     for (int64 value = index_->OldMin(); value < vmin; ++value) {
01367       vars_[value]->RemoveValue(target_);
01368       demons_[value]->inhibit(solver());
01369     }
01370     for (index_iterator_->Init();
01371          index_iterator_->Ok();
01372          index_iterator_->Next()) {
01373       const int64 value = index_iterator_->Value();
01374       vars_[value]->RemoveValue(target_);
01375       demons_[value]->inhibit(solver());
01376     }
01377     for (int64 value = vmax + 1; value <= oldmax; ++value) {
01378       vars_[value]->RemoveValue(target_);
01379       demons_[value]->inhibit(solver());
01380     }
01381     if (index_->Bound()) {
01382       vars_[index_->Min()]->SetValue(target_);
01383     }
01384   }
01385 
01386   virtual string DebugString() const {
01387     return StringPrintf("IntExprArrayPosition([%s], %s) == %" GG_LL_FORMAT "d",
01388                         DebugStringVector(vars_, ", ").c_str(),
01389                         index_->DebugString().c_str(),
01390                         target_);
01391   }

01393   virtual void Accept(ModelVisitor* const visitor) const {
01394     visitor->BeginVisitConstraint(ModelVisitor::kElementEqual, this);
01395     visitor->VisitIntegerVariableArrayArgument(ModelVisitor::kVarsArgument,
01396                                                vars_.data(),
01397                                                size_);
01398     visitor->VisitIntegerExpressionArgument(ModelVisitor::kIndexArgument,
01399                                             index_);
01400     visitor->VisitIntegerArgument(ModelVisitor::kTargetArgument,
01401                                   target_);
01402     visitor->EndVisitConstraint(ModelVisitor::kElementEqual, this);
01403   }

01407   const int size_;

01411   IntVarIterator* const index_iterator_;

01426 IntExpr* Solver::MakeElement(const std::vector<IntVar*>& vars, IntVar* const index) {
01427   if (AreAllBound(vars)) {
01428     std::vector<int64> values(vars.size());
01429     for (int i = 0; i < vars.size(); ++i) {
01430       values[i] = vars[i]->Value();
01431     }
01432     return MakeElement(values, index);
01433   }
01434   int64 emin = kint64max;
01435   int64 emax = kint64min;
01436   scoped_ptr<IntVarIterator> iterator(index->MakeDomainIterator(false));
01437   for (iterator->Init(); iterator->Ok(); iterator->Next()) {
01438     const int64 index_value = iterator->Value();
01439     if (index_value >= 0 && index_value < vars.size()) {
01440       emin = std::min(emin, vars[index_value]->Min());
01441       emax = std::max(emax, vars[index_value]->Max());
01442     }
01443   }
01444   const string vname =
01445       vars.size() > 10 ?
01446       StringPrintf("ElementVar(var array of size %" GG_LL_FORMAT "d, %s)",
01447                    vars.size(), index->DebugString().c_str()) :
01448       StringPrintf("ElementVar([%s], %s)",
01449                    NameVector(vars, ", ").c_str(), index->name().c_str());
01450   IntVar* const element_var = MakeIntVar(emin, emax, vname);
01451   AddConstraint(RevAlloc(new IntExprArrayElementCt(this,
01452                                                    vars.data(),
01453                                                    vars.size(),
01454                                                    index,
01455                                                    element_var)));
01456   return element_var;
01457 }
