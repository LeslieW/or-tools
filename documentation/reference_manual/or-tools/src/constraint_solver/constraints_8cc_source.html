00280 // This constraint ensures there are no cycles in the variable/value graph.
00281 // "Sink" values are values outside the range of the array of variables; they
00282 // are used to end paths.
00283 // The constraint does essentially two things:
00284 // - forbid partial paths from looping back to themselves
00285 // - ensure each variable/node can be connected to a "sink".
00286 // If assume_paths is true, the constraint assumes the 'next' variables
00287 // represent paths (and performs a faster propagation); otherwise the
00288 // constraint assumes the 'next' variables represent a forest.

00366 void NoCycle::InitialPropagate() {
00367   // Reduce next domains to sinks + range of nexts
00368   for (int i = 0; i < size_; ++i) {
00369     IntVar* next = nexts_[i];
00370     for (int j = next->Min(); j < 0; ++j) {
00371       if (!sink_handler_->Run(j)) {
00372         next->RemoveValue(j);
00373       }
00374     }
00375     for (int j = next->Max(); j >= size_; --j) {
00376       if (!sink_handler_->Run(j)) {
00377         next->RemoveValue(j);
00378       }
00379     }
00380   }

00382     if (nexts_[i]->Bound()) {
00383       NextBound(i);
00384     }
00385   }

00389 void NoCycle::Post() {
00390   if (size_ == 0) return;
00391   for (int i = 0; i < size_; ++i) {
00392     IntVar* next = nexts_[i];
00393     Demon* d = MakeConstraintDemon1(solver(),
00394                                     this,
00395                                     &NoCycle::NextBound,
00396                                     "NextBound",
00397                                     i);
00398     next->WhenBound(d);
00399     Demon* support_demon = MakeConstraintDemon1(solver(),
00400                                                 this,
00401                                                 &NoCycle::CheckSupport,
00402                                                 "CheckSupport",
00403                                                 i);
00404     next->WhenDomain(support_demon);
00405     Demon* active_demon = MakeConstraintDemon1(solver(),
00406                                                this,
00407                                                &NoCycle::ActiveBound,
00408                                                "ActiveBound",
00409                                                i);
00410     active_[i]->WhenBound(active_demon);
00411   }
00412   // Setting up sinks
00413   int64 min_min = nexts_[0]->Min();
00414   int64 max_max = nexts_[0]->Max();
00415   for (int i = 1; i < size_; ++i) {
00416     const IntVar* next = nexts_[i];
00417     min_min = std::min(min_min, next->Min());
00418     max_max = std::max(max_max, next->Max());
00419   }
00420   sinks_.clear();
00421   for (int i = min_min; i <= max_max; ++i) {
00422     if (sink_handler_->Run(i)) {
00423       sinks_.push_back(i);
00424     }
00425   }

00441 void NoCycle::NextBound(int index) {
00442   if (active_[index]->Min() == 0) return;
00443   const int64 next = nexts_[index]->Value();
00444   const int64 chain_start = starts_[index];
00445   const int64 chain_end = !sink_handler_->Run(next) ? ends_[next] : next;
00446   Solver* const s = solver();
00447   s->SaveAndSetValue(&ends_[chain_start], chain_end);
00448   if (!sink_handler_->Run(chain_end)) {
00449     s->SaveAndSetValue(&starts_[chain_end], chain_start);
00450     nexts_[chain_end]->RemoveValue(chain_start);
00451     if (!assume_paths_) {
00452       for (int i = 0; i < size_; ++i) {
00453         int64 current = i;
00454         bool found = (current == chain_end);
00455         // Counter to detect implicit cycles.
00456         int count = 0;
00457         while (!found
00458                && count < size_
00459                && !sink_handler_->Run(current)
00460                && nexts_[current]->Bound()) {
00461           current = nexts_[current]->Value();
00462           found = (current == chain_end);
00463           ++count;
00464         }
00465         if (found) {
00466           nexts_[chain_end]->RemoveValue(i);
00467         }
00468       }
00469     }

00473 // For each variable, find a path connecting to a sink. Starts partial paths
00474 // from the sinks down to all unconnected variables. If some variables remain
00475 // unconnected, fail. Resulting paths are used as supports.
00476 void NoCycle::ComputeSupports() {
00477   scoped_array<int64> supported(new int64[size_]);
00478   int64 support_count = 0;
00479   for (int i = 0; i < size_; ++i) {
00480     if (nexts_[i]->Bound()) {
00481       supported[support_count] = i;
00482       outbound_supports_[i] = nexts_[i]->Value();
00483       ++support_count;
00484     } else {
00485       outbound_supports_[i] = -1;
00486     }
00487   }
00488   if (size_ == support_count) {
00489     return;
00490   }
00491   const int size = sinks_.size();
00492   for (int i = 0; i < size_; ++i) {
00493     const IntVar* next = nexts_[i];
00494     if (!nexts_[i]->Bound()) {
00495       for (int j = 0; j < size; ++j) {
00496         if (next->Contains(sinks_[j])) {
00497           supported[support_count] = i;
00498           outbound_supports_[i] = sinks_[j];
00499           ++support_count;
00500           break;
00501         }
00502       }
00503     }
00504   }
00505   for (int i = 0; i < support_count && support_count < size_; ++i) {
00506     const int64 supported_i = supported[i];
00507     for (int j = 0; j < size_; ++j) {
00508       if (outbound_supports_[j] < 0 && nexts_[j]->Contains(supported_i)) {
00509         supported[support_count] = j;
00510         outbound_supports_[j] = supported_i;
00511         ++support_count;
00512       }
00513     }
00514   }
00515   if (size_ != support_count) {
00516     supported.reset(NULL);
00517     for (int i = 0; i < size_; ++i) {
00518       if (outbound_supports_[i] < 0) {
00519         active_[i]->SetMax(0);
00520       }
00521     }
00522   }
00523 }

00659 void PathCumul::Post() {
00660   for (int i = 0; i < size_; ++i) {
00661     IntVar* var = nexts_[i];
00662     Demon* d = MakeConstraintDemon1(solver(),
00663                                     this,
00664                                     &PathCumul::NextBound,
00665                                     "NextBound",
00666                                     i);
00667     var->WhenBound(d);
00668     Demon* ds = MakeConstraintDemon1(solver(),
00669                                      this,
00670                                      &PathCumul::UpdateSupport,
00671                                      "UpdateSupport",
00672                                      i);
00673     var->WhenDomain(ds);
00674     Demon* active_demon = MakeConstraintDemon1(solver(),
00675                                                this,
00676                                                &PathCumul::ActiveBound,
00677                                                "ActiveBound",
00678                                                i);
00679     active_[i]->WhenBound(active_demon);
00680     Demon* transit_demon = MakeConstraintDemon1(solver(),
00681                                                 this,
00682                                                 &PathCumul::TransitRange,
00683                                                 "TransitRange",
00684                                                 i);
00685     transits_[i]->WhenRange(transit_demon);
00686   }
00687   for (int i = 0; i < cumul_size_; ++i) {
00688     IntVar* cumul = cumuls_[i];
00689     Demon* d = MakeConstraintDemon1(solver(),
00690                                     this,
00691                                     &PathCumul::CumulRange,
00692                                     "CumulRange",
00693                                     i);
00694     cumul->WhenRange(d);
00695   }

00704 void PathCumul::NextBound(int index) {
00705   if (active_[index]->Min() == 0) return;
00706   const int64 next = nexts_[index]->Value();
00707   IntVar* cumul = cumuls_[index];
00708   IntVar* cumul_next = cumuls_[next];
00709   IntVar* transit = transits_[index];
00710   cumul_next->SetRange(cumul->Min() + transit->Min(),
00711                        CapAdd(cumul->Max(), transit->Max()));
00712   cumul->SetRange(CapSub(cumul_next->Min(), transit->Max()),
00713                   CapSub(cumul_next->Max(), transit->Min()));
00714   transit->SetRange(CapSub(cumul_next->Min(), cumul->Max()),
00715                     CapSub(cumul_next->Max(), cumul->Min()));
00716   if (prevs_[next] < 0) {
00717     prevs_.SetValue(solver(), next, index);
00718   }
00719 }

00773 bool PathCumul::AcceptLink(int i, int j) const {
00774   const IntVar* const cumul_i = cumuls_[i];
00775   const IntVar* const cumul_j = cumuls_[j];
00776   const IntVar* const transit_i = transits_[i];
00777   return transit_i->Min() <= CapSub(cumul_j->Max(), cumul_i->Min())
00778       && CapSub(cumul_j->Min(), cumul_i->Max()) <= transit_i->Max();
00779 }

00781 void PathCumul::FilterNext(int index) {
00782   IntVar* var = nexts_[index];
00783   bool found = false;
00784   scoped_ptr<IntVarIterator> it(var->MakeDomainIterator(false));
00785   std::vector<int64> to_remove;
00786   for (it->Init(); it->Ok(); it->Next()) {
00787     const int value = it->Value();
00788     if (AcceptLink(index, value)) {
00789       if (!found) {
00790         found = true;
00791         supports_[index] = value;
00792       }
00793     } else {
00794       to_remove.push_back(value);
00795     }
00796   }
00797   if (!found) {
00798     active_[index]->SetMax(0);
00799   } else {
00800     var->RemoveValues(to_remove);
00801   }
00802 }
00803 
00804 void PathCumul::UpdateSupport(int index) {
00805   int support = supports_[index];
00806   if (support < 0 || !AcceptLink(index, support)) {
00807     IntVar* var = nexts_[index];
00808     for (int i = var->Min(); i <= var->Max(); ++i) {
00809       if (i != support && AcceptLink(index, i)) {
00810         supports_[index] = i;
00811         return;
00812       }
00813     }
00814     active_[index]->SetMax(0);
00815   }
00816 }

00842 class IntModulo : public Constraint {
00843  public:
00844   IntModulo(Solver* const solver, IntVar* const x, int64 mod, IntVar* const y)
00845       : Constraint(solver),
00846         x_(x),
00847         mod_(mod),
00848         y_(y),
00849         x_iterator_(x_->MakeDomainIterator(true)),
00850         y_iterator_(y_->MakeDomainIterator(true)) {
00851     CHECK_GE(mod_, 0);
00852   }
00853 
00854   virtual ~IntModulo() {}
00855 
00856   virtual void Post() {
00857     demon_ = solver()->MakeConstraintInitialPropagateCallback(this);
00858     x_->WhenDomain(demon_);
00859     y_->WhenDomain(demon_);
00860   }
00861 
00862   virtual void InitialPropagate() {
00863     if (x_->Bound()) {
00864       y_->SetValue(x_->Min() % mod_);
00865       return;
00866     }
00867 
00868     y_->SetRange(0, mod_ - 1);
00869 
00870     if (y_->Bound()) {
00871       const int64 result = y_->Min();
00872       to_remove_.clear();
00873       for (x_iterator_->Init(); x_iterator_->Ok(); x_iterator_->Next()) {
00874         const int64 value = x_iterator_->Value();
00875         if (value % mod_ != result) {
00876           to_remove_.push_back(value);
00877         }
00878       }
00879       x_->RemoveValues(to_remove_);
00880       demon_->inhibit(solver());
00881     } else {
00882       if (y_->Size() <= mod_) {
00883         to_remove_.clear();
00884         for (x_iterator_->Init(); x_iterator_->Ok(); x_iterator_->Next()) {
00885           const int64 value = x_iterator_->Value();
00886           if (!y_->Contains(value % mod_)) {
00887             to_remove_.push_back(value);
00888           }
00889         }
00890         x_->RemoveValues(to_remove_);
00891         to_remove_.clear();
00892       }
00893 
00894       for (y_iterator_->Init(); y_iterator_->Ok(); y_iterator_->Next()) {
00895         const int64 value = y_iterator_->Value();
00896         bool support = false;
00897         for (int64 w = 0; w <= x_->Max() / mod_; ++w) {
00898           if (x_->Contains(w * mod_ + value)) {
00899             support = true;
00900             break;
00901           }
00902         }
00903         if (!support) {
00904           to_remove_.push_back(value);
00905         }
00906       }
00907       y_->RemoveValues(to_remove_);
00908     }
00909   }
00910 
00911   virtual string DebugString() const {
00912     return StringPrintf("(%s %% %" GG_LL_FORMAT "d == %s",
00913                         x_->DebugString().c_str(),
00914                         mod_,
00915                         y_->DebugString().c_str());
00916   }
00917 
00918  private:
00919   IntVar* const x_;
00920   const int64 mod_;
00921   IntVar* const y_;
00922   IntVarIterator* const x_iterator_;
00923   IntVarIterator* const y_iterator_;
00924   std::vector<int64> to_remove_;
00925   Demon* demon_;
00926 };

01064                                          IntVar* const mod,
01065                                          IntVar* const y) {
01066   if (mod->Bound()) {
