00197 Constraint* Solver::MakeCount(const std::vector<IntVar*>& vars, int64 v, int64 c) {
00198   for (ConstIter<std::vector<IntVar*> > it(vars); !it.at_end(); ++it) {
00199     CHECK_EQ(this, (*it)->solver());
00200   }
00201   int count_min = 0;
00202   int count_max = 0;
00203   int count_active = 0;
00204   int count_forced = 0;
00205   for (int i = 0; i < vars.size(); ++i) {
00206     if (vars[i]->Contains(v)) {
00207       count_active++;
00208       if (vars[i]->Bound()) {
00209         count_forced++;
00210       }
00211       if (vars[i]->Max() - vars[i]->Min() <= 1) {
00212         if (vars[i]->Min() == v) {
00213           count_min++;
00214         } else if (vars[i]->Max() == v) {
00215           count_max++;
00216         }
00217       }
00218     }
00219   }
00220   if (count_active == 0) {
00221     return (c == 0 ? MakeTrueConstraint() : MakeFalseConstraint());
00222   } else if (count_forced == count_active) {
00223     return (c == count_forced ? MakeTrueConstraint() : MakeFalseConstraint());
00224   }
00225   if (count_min == count_active) {
00226     std::vector<IntVar*> terms;
00227     for (int i = 0; i < vars.size(); ++i) {
00228       if (vars[i]->Contains(v)) {
00229         terms.push_back(MakeSum(vars[i], -v)->Var());
00230       }
00231     }
00232     return MakeSumEquality(terms, count_active - c);
00233   } else if (count_max == count_active) {
00234     std::vector<IntVar*> terms;
00235     for (int i = 0; i < vars.size(); ++i) {
00236       if (vars[i]->Contains(v)) {
00237         terms.push_back(MakeSum(vars[i], -v + 1)->Var());
00238       }
00239     }
00240     return MakeSumEquality(terms, c);
00241   }
00242   return RevAlloc(new CountValueEqCst(this, vars.data(), vars.size(), v, c));
00243 }

00777                                "OneDomain",

00780     }
00781   }

00783     if (!cards_[card_index]->Bound()) {
00784       Demon* d = MakeConstraintDemon1(solver(),
00785                                       this,

00794 void FastDistribute::InitialPropagate() {
00795   Solver* const s = solver();
00796   for (int card_index = 0; card_index < card_size_; ++card_index) {
00797     int min = 0;
00798     int max = 0;
00799     for (int var_index = 0; var_index < var_size_; ++var_index) {
00800       IntVar* const var = vars_[var_index];
00801       if (var->Bound() && var->Min() == card_index) {
00802         min++;
00803         max++;
00804       } else if (var->Contains(card_index)) {
00805         max++;
00806         undecided_.SetToOne(s, var_index, card_index);
00807       }
00808     }
00809     min_.SetValue(s, card_index, min);
00810     max_.SetValue(s, card_index, max);
00811     CountVar(card_index);
00812   }
00813 }

00816   IntVar* const var = vars_[index];
00817   for (int card_index = 0; card_index < card_size_; ++card_index) {
00818     if (undecided_.IsSet(index, card_index)) {
00819       if (var->Min() == card_index) {
00820         SetRevDoContribute(index, card_index);
00821       } else {
00822         SetRevCannotContribute(index, card_index);
00823       }
00824     }

00828 void FastDistribute::OneDomain(int index) {
00829   IntVar* const var = vars_[index];
00830   const int64 oldmin = var->OldMin();
00831   const int64 oldmax = var->OldMax();
00832   const int64 vmin = var->Min();
00833   const int64 vmax = var->Max();
00834   for (int64 card_index = std::max(oldmin, 0LL);
00835        card_index < std::min(vmin, card_size_);
00836        ++card_index) {
00837     if (undecided_.IsSet(index, card_index)) {
00838       SetRevCannotContribute(index, card_index);
00839     }
00840   }
00841   IntVarIterator* const holes = holes_[index];
00842   for (holes->Init(); holes->Ok(); holes->Next()) {
00843     const int64 card_index = holes->Value();
00844     if (card_index >= 0 &&
00845         card_index < card_size_ &&
00846         undecided_.IsSet(index, card_index)) {
00847       SetRevCannotContribute(index, card_index);
00848     }
00849   }
00850   for (int64 card_index = std::max(vmax + 1, 0LL);
00851        card_index <= std::min(oldmax, card_size_ - 1);
00852        ++card_index) {
00853     if (undecided_.IsSet(index, card_index)) {
00854       SetRevCannotContribute(index, card_index);
00855     }
00856   }
00857 }

01026 void BoundedDistribute::InitialPropagate() {
01027   Solver* const s = solver();
01028 
01029   int64 sum_card_min = 0;
01030   for (int i = 0; i < card_size_; ++i) {
01031     if (card_max_[i] < card_min_[i]) {
01032       solver()->Fail();
01033     }
01034     sum_card_min += card_min_[i];
01035   }
01036   if (sum_card_min > var_size_) {
01037     s->Fail();
01038   }
01039   if (sum_card_min == var_size_) {
01040     for (int i = 0; i < var_size_; ++i) {
01041       vars_[i]->SetRange(0, card_size_ - 1);
01042     }
01043   }
01044 
01045   for (int card_index = 0; card_index < card_size_; ++card_index) {
01046     int min = 0;
01047     int max = 0;
01048     for (int i = 0; i < var_size_; ++i) {
01049       IntVar* const var = vars_[i];
01050       if (var->Bound()) {
01051         if (var->Min() == card_index) {
01052           min++;
01053           max++;
01054         }
01055       } else if (var->Contains(card_index)) {
01056         max++;
01057         undecided_.SetToOne(s, i, card_index);
01058       }
01059     }
01060     min_.SetValue(s, card_index, min);
01061     max_.SetValue(s, card_index, max);

01063   }
