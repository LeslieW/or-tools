00510 bool NotLast::Propagate() {
00511   // ---- Init ----
00512   Sort(&by_start_max_, StartMaxLessThan<DisjunctiveTask>);
00513   Sort(&by_end_max_, EndMaxLessThan<DisjunctiveTask>);
00514   // Update start min positions
00515   Sort(&by_start_min_, StartMinLessThan<DisjunctiveTask>);
00516   for (int i = 0; i < size_; ++i) {
00517     by_start_min_[i]->set_start_min_index(i);
00518   }
00519   theta_tree_.Clear();
00520   for (int i = 0; i < size_; ++i) {
00521     new_lct_[i] = by_start_min_[i]->EndMax();
00522   }
00523 
00524   // --- Execute ----
00525   int j = 0;
00526   for (int i = 0; i < size_; ++i) {
00527     DisjunctiveIndexedTask* const twi = by_end_max_[i];
00528     while (j < size_ &&
00529            twi->EndMax() > by_start_max_[j]->StartMax()) {
00530       if (j > 0 && theta_tree_.ECT() > by_start_max_[j]->StartMax()) {
00531         const int64 new_end_max = by_start_max_[j - 1]->StartMax();
00532         new_lct_[by_start_max_[j]->start_min_index()] = new_end_max;
00533       }
00534       theta_tree_.Insert(by_start_max_[j]);
00535       j++;
00536     }
00537     const bool inserted = theta_tree_.IsInserted(twi);
00538     if (inserted) {
00539       theta_tree_.Remove(twi);
00540     }
00541     const int64 ect_theta_less_i = theta_tree_.ECT();
00542     if (inserted) {
00543       theta_tree_.Insert(twi);
00544     }
00545     if (ect_theta_less_i > twi->EndMax() && j > 0) {
00546       const int64 new_end_max = by_start_max_[j - 1]->EndMax();
00547       if (new_end_max > new_lct_[twi->start_min_index()]) {

00558       by_start_min_[i]->mutable_interval()->SetEndMax(new_lct_[i]);

00656 bool EdgeFinderAndDetectablePrecedences::DetectablePrecedences() {
00657   // Init
00658   UpdateEst();
00659   for (int i = 0; i < size_; ++i) {
00660     new_est_[i] = kint64min;
00661   }
00662 
00663   // Propagate in one direction
00664   Sort(&by_end_min_, EndMinLessThan<DisjunctiveTask>);
00665   Sort(&by_start_max_, StartMaxLessThan<DisjunctiveTask>);
00666   theta_tree_.Clear();
00667   int j = 0;
00668   for (int i = 0; i < size_; ++i) {
00669     DisjunctiveIndexedTask* const task_i = by_end_min_[i];
00670     if (j < size_) {
00671       DisjunctiveIndexedTask* task_j = by_start_max_[j];
00672       while (task_i->EndMin() > task_j->StartMax()) {
00673         theta_tree_.Insert(task_j);
00674         j++;
00675         if (j == size_)
00676           break;
00677         task_j = by_start_max_[j];
00678       }
00679     }
00680     const int64 esti = task_i->StartMin();
00681     bool inserted = theta_tree_.IsInserted(task_i);
00682     if (inserted) {
00683       theta_tree_.Remove(task_i);
00684     }
00685     const int64 oesti = theta_tree_.ECT();
00686     if (inserted) {
00687       theta_tree_.Insert(task_i);
00688     }
00689     if (oesti > esti) {
00690       new_est_[task_i->start_min_index()] = oesti;
00691     } else {
00692       new_est_[task_i->start_min_index()] = kint64min;
00693     }
00694   }
00695 
00696   // Apply modifications
00697   bool modified = false;
00698   for (int i = 0; i < size_; ++i) {
00699     if (new_est_[i] != kint64min) {
00700       modified = true;
00701       by_start_min_[i]->mutable_interval()->SetStartMin(new_est_[i]);
00702     }
00703   }
00704   return modified;
00705 }

00707 bool EdgeFinderAndDetectablePrecedences::EdgeFinder() {
00708   // Init
00709   UpdateEst();
00710   for (int i = 0; i < size_; ++i) {
00711     new_est_[i] = by_start_min_[i]->StartMin();
00712   }
00713 
00714   // Push in one direction.
00715   Sort(&by_end_max_, EndMaxLessThan<DisjunctiveTask>);
00716   lt_tree_.Clear();
00717   for (int i = 0; i < size_; ++i) {
00718     lt_tree_.Insert(by_start_min_[i]);
00719     DCHECK_EQ(i, by_start_min_[i]->start_min_index());
00720   }
00721   for (int j = size_ - 2; j >= 0; --j) {
00722     lt_tree_.Grey(by_end_max_[j+1]);
00723     DisjunctiveIndexedTask* const twj = by_end_max_[j];
00724     // We should have checked for overloading earlier.
00725     DCHECK_LE(lt_tree_.ECT(), twj->EndMax());
00726     while (lt_tree_.ECT_opt() > twj->EndMax()) {
00727       const int i = lt_tree_.Responsible_opt();
00728       DCHECK_GE(i, 0);
00729       if (lt_tree_.ECT() > new_est_[i]) {
00730         new_est_[i] = lt_tree_.ECT();
00731       }
00732       lt_tree_.Reset(i);
00733     }
00734   }
00735 
00736   // Apply modifications.
00737   bool modified = false;
00738   for (int i = 0; i < size_; ++i) {
00739     if (by_start_min_[i]->StartMin() < new_est_[i]) {
00740       modified = true;
00741       by_start_min_[i]->mutable_interval()->SetStartMin(new_est_[i]);
00742     }
00743   }
00744   return modified;
00745 }

00751 class RankedPropagator : public Constraint {
00752  public:
00753   RankedPropagator(Solver* const solver,
00754                    IntVar* const * nexts,
00755                    IntervalVar* const * intervals,
00756                    IntVar* const * transits,
00757                    int size)
00758       : Constraint(solver),
00759         nexts_(new IntVar*[size + 1]),
00760         intervals_(new IntervalVar*[size]),
00761         transits_(new IntVar*[size]),
00762         size_(size),
00763         partial_sequence_(size),
00764         previous_(new int[size +  2]) {
00765     memcpy(nexts_.get(), nexts, (size_ + 1) * sizeof(*nexts));
00766     memcpy(intervals_.get(), intervals, size_ * sizeof(*intervals));
00767     memcpy(transits_.get(), transits, size_ * sizeof(*transits));
00768     memset(previous_.get(), 0, (size_ + 2) * sizeof(*previous_.get()));
00769   }
00770 
00771   virtual ~ RankedPropagator() {}
00772 
00773   virtual void Post() {
00774     Demon* const delayed =
00775         solver()->MakeDelayedConstraintInitialPropagateCallback(this);
00776     for (int i = 0; i < size_; ++i) {
00777       nexts_[i]->WhenBound(delayed);
00778       intervals_[i]->WhenAnything(delayed);
00779       transits_[i]->WhenRange(delayed);
00780     }
00781     nexts_[size_]->WhenBound(delayed);
00782   }
00783 
00784   virtual void InitialPropagate() {
00785     PropagateNexts();
00786     PropagateSequence();
00787   }
00788 
00789   void PropagateNexts() {
00790     Solver* const s = solver();
00791     const int ranked_first = partial_sequence_.FirstRanked();
00792     const int ranked_last = partial_sequence_.LastRanked();
00793     const int sentinel = ranked_last == 0 ?
00794         size_ + 1 :
00795         partial_sequence_[size_ - ranked_last] + 1;
00796     int first = 0;
00797     int counter = 0;
00798     while (nexts_[first]->Bound()) {
00799       first = nexts_[first]->Min();
00800       if (first == sentinel) {
00801         return;
00802       }
00803       if (++counter > ranked_first) {
00804         partial_sequence_.RankFirst(s, first - 1);
00805         VLOG(1) << "RankFirst " << first - 1
00806                 << " -> " << partial_sequence_.DebugString();
00807       }
00808     }
00809     for (int i = 0; i < size_ + 2; ++i) {
00810       previous_[i] = -1;
00811     }
00812     for (int i = 0; i < size_ + 1; ++i) {
00813       if (nexts_[i]->Bound()) {
00814         previous_[nexts_[i]->Min()] = i;
00815       }
00816     }
00817     int last = size_ + 1;
00818     counter = 0;
00819     while (previous_[last] != -1) {
00820       last = previous_[last];
00821       if (++counter > ranked_last) {
00822         partial_sequence_.RankLast(s, last - 1);
00823         VLOG(1) << "RankLast " << last - 1
00824                 << " -> " << partial_sequence_.DebugString();
00825       }
00826     }
00827     VLOG(1) << "PropagateNext("
00828             << DebugStringArray(nexts_.get(), size_ + 1, ", ")
00829             << ", " << partial_sequence_.DebugString() << ")";
00830   }
00831 
00832   void PropagateSequence() {
00833     const int last_position = size_ - 1;
00834     const int first_sentinel = partial_sequence_.FirstRanked();
00835     const int last_sentinel = last_position - partial_sequence_.LastRanked();
00836     // Propagates on ranked first from left to right.
00837     for (int i = 0; i < first_sentinel - 1; ++i) {
00838       IntervalVar* const interval = RankedInterval(i);
00839       IntervalVar* const next_interval = RankedInterval(i + 1);
00840       IntVar* const transit = RankedTransit(i);
00841       next_interval->SetStartRange(interval->StartMin() + transit->Min(),
00842                                    interval->StartMax() + transit->Max());
00843     }
00844     // Propagates on ranked last from right to left.
00845     for (int i = last_position; i > last_sentinel; --i) {
00846       IntervalVar* const interval = RankedInterval(i -1);
00847       IntervalVar* const next_interval = RankedInterval(i);
00848       IntVar* const transit = RankedTransit(i - 1);
00849       interval->SetStartRange(next_interval->StartMin() - transit->Max(),
00850                               next_interval->StartMax() - transit->Min());
00851     }
00852     // Propagate across.
00853     IntervalVar* const first_interval =
00854         first_sentinel > 0 ?
00855         RankedInterval(first_sentinel - 1) :
00856         NULL;
00857     IntVar* const first_transit =
00858         first_sentinel > 0 ?
00859         RankedTransit(first_sentinel - 1) :
00860         NULL;
00861     IntervalVar* const last_interval =
00862         last_sentinel < last_position ?
00863         RankedInterval(last_sentinel + 1) :
00864         NULL;
00865     IntVar* const last_transit =
00866         last_sentinel < last_position ?
00867         RankedTransit(last_sentinel + 1) :
00868         NULL;
00869 
00870     // Nothing to do afterwards, exiting.
00871     if (first_interval == NULL && last_interval == NULL) {
00872       return;
00873     }
00874     // Propagates to the middle part.
00875     for (int i = first_sentinel; i <= last_sentinel; ++i) {
00876       IntervalVar* const interval = RankedInterval(i);
00877       IntVar* const transit = RankedTransit(i);
00878       if (interval->MayBePerformed()) {
00879         if (first_interval != NULL) {
00880           interval->SetStartRange(
00881               first_interval->StartMin() + first_transit->Min(),
00882               first_interval->StartMax() + first_transit->Max());
00883         }
00884         if (last_interval != NULL) {
00885           interval->SetStartRange(
00886               last_interval->StartMin() - transit->Max(),
00887               last_interval->StartMax() - transit->Min());
00888         }
00889         if (interval->MustBePerformed()) {
00890           if (last_interval != NULL) {
00891             last_interval->SetStartRange(
00892                 interval->StartMin() + transit->Min(),
00893                 interval->StartMax() + transit->Max());
00894           }
00895           if (first_interval != NULL) {
00896             first_interval->SetStartRange(
00897                 interval->StartMin() - first_transit->Max(),
00898                 interval->StartMax() - first_transit->Min());
00899           }
00900         }
00901       }
00902     }
00903     // Propagates on ranked first from right to left.
00904     for (int i = std::min(first_sentinel - 2, last_position - 1); i >= 0; --i) {
00905       IntervalVar* const interval = RankedInterval(i);
00906       IntervalVar* const next_interval = RankedInterval(i + 1);
00907       IntVar* const transit = RankedTransit(i);
00908       interval->SetStartRange(next_interval->StartMin() - transit->Max(),
00909                               next_interval->StartMax() - transit->Min());
00910     }
00911     // Propagates on ranked last from left to right.
00912     for (int i = last_sentinel; i < last_position - 1; ++i) {
00913       IntervalVar* const interval = RankedInterval(i);
00914       IntervalVar* const next_interval = RankedInterval(i + 1);
00915       IntVar* const transit = RankedTransit(i);
00916       next_interval->SetStartRange(interval->StartMin() + transit->Min(),
00917                                    interval->StartMax() + transit->Max());
00918     }
00919     // TODO(lperron) : Propagate on transits.
00920   }
00921 
00922   IntervalVar* RankedInterval(int i) const {
00923     const int index = partial_sequence_[i];
00924     return intervals_[index];
00925   }
00926 
00927   IntVar* RankedTransit(int i) const {
00928     const int index = partial_sequence_[i];
00929     return transits_[index];
00930   }
00931 
00932   virtual string DebugString() const {
00933     return StringPrintf("RankedPropagator([%s])",
00934                         partial_sequence_.DebugString().c_str());
00935   }

00937  private:
00938   scoped_array<IntVar*> nexts_;
00939   scoped_array<IntervalVar*> intervals_;
00940   scoped_array<IntVar*> transits_;
00941   const int size_;
00942   RevPartialSequence partial_sequence_;
00943   scoped_array<int> previous_;
00944 };

01088 class DualCapacityThetaNode {

01093   // Identity constructor
01094   DualCapacityThetaNode()

01097       residual_energetic_end_min_(kint64min) {}
01098 
01099   // Constructor for a single cumulative task in the Theta set
01100   DualCapacityThetaNode(int64 capacity, int64 residual_capacity,
01101                         const CumulativeTask& task)
01102     : energy_(task.EnergyMin()),
01103       energetic_end_min_(
01104           capacity * task.interval()->StartMin() + energy_),
01105       residual_energetic_end_min_(
01106           residual_capacity * task.interval()->StartMin() + energy_) {}

01108   // Getters

01110   int64 energetic_end_min() const { return energetic_end_min_; }
01111   int64 residual_energetic_end_min() const {
01112     return residual_energetic_end_min_;
01113   }

01115   // Copy from the given node
01116   void Set(const DualCapacityThetaNode& node) {
01117     energy_ = node.energy_;
01118     energetic_end_min_ = node.energetic_end_min_;
01119     residual_energetic_end_min_ = node.residual_energetic_end_min_;
01120   }
01121 
01122   // Sets this DualCapacityThetaNode to the result of the natural binary
01123   // operation over the two given operands, corresponding to the following set
01124   // operation: Theta = left.Theta union right.Theta
01125   //
01126   // No set operation actually occur: we only maintain the relevant quantities
01127   // associated with such sets.
01128   void Compute(const DualCapacityThetaNode& left,
01129                const DualCapacityThetaNode& right) {
01130     energy_ = left.energy_ + right.energy_;
01131     energetic_end_min_ =
01132         std::max(left.energetic_end_min_ + right.energy_,
01133                  right.energetic_end_min_);
01134     residual_energetic_end_min_ =
01135         std::max(left.residual_energetic_end_min_ + right.energy_,
01136                  right.residual_energetic_end_min_);
01137   }
01138 
01139  private:
01140   // Amount of resource consumed by the Theta set, in units of demand X time.
01141   // This is energy(Theta).
01142   int64 energy_;
01143

01205     //    DCHECK_GT(energetic_end_min_alpha_, kint64min);  TODO(user): Check me.

01302 // Returns min(a * b, kint64max). a is positive.
01303 int64 SafeProduct(int64 a, int64 b) {
01304   DCHECK_GE(a, 0);
01305 
01306   const bool is_positive = b >= 0;
01307   b = std::max(b, -b);  // abs(b) for int64.
01308   // Note max(kint64min, -kint64min) = kint64min, so when b == kint64min,
01309   // the following DCHECK fails.
01310   DCHECK_GE(b, 0);
01311   const int kint64SurelyOverflow = 63;
01312   const bool surely_overflow = MostSignificantBitPosition64(a)
01313       + MostSignificantBitPosition64(b)
01314       > kint64SurelyOverflow;
01315 
01316   // When the sum of MostSignificantBitPosition64 for a and b is smaller
01317   // than 61 no overflow can appear and the product result is positive (as a
01318   // and b are positive). However when the sum is between 61 and 63 then it
01319   // is not possible to say in advance if an overflow will occur. Nevertheless
01320   // the product sign is a valid test for overflow: if an overflow occurs,
01321   // then the product is surely negative, and if the product is negative
01322   // then an overflow occurred.
01323   const int64 product = a * b;
01324   return (!surely_overflow && product >= 0)
01325       ? ((is_positive) ? product : -product)
01326       : ((is_positive) ? kint64max : -kint64max);
01327 }

01412   // Computes all possible update values for tasks of given demand, and stores
01413   // these values in update_map_[demand].
01414   // Runs in O(n log n).
01415   // This corresponds to lines 2--13 in algorithm 1.3 in Petr Vilim's paper.
01416   void ComputeConditionalStartMins(int64 demand) {
01417     DCHECK_GT(demand, 0);
01418     UpdatesForADemand* const updates = update_map_[demand];
01419     DCHECK(!updates->up_to_date());
01420     DualCapacityThetaTree dual_capa_tree(size_, capacity_);
01421     const int64 residual_capacity = capacity_ - demand;
01422     dual_capa_tree.SetResidualCapacity(residual_capacity);
01423     // It's important to initialize the update at IntervalVar::kMinValidValue
01424     // rather than at kInt64min, because its opposite may be used if it's a
01425     // mirror variable, and
01426     // -kInt64min = -(-kInt64max - 1) = kInt64max + 1 = -kInt64min
01427     int64 update = IntervalVar::kMinValidValue;
01428     for (int i = 0; i < size_; ++i) {
01429       const int64 current_end_max = by_end_max_[i]->EndMax();
01430       dual_capa_tree.Insert(by_end_max_[i]);
01431       const int64 energy_threshold = residual_capacity * current_end_max;
01432       const DualCapacityThetaNode& root = dual_capa_tree.result();
01433       const int64 res_energetic_end_min = root.residual_energetic_end_min();
01434       if (res_energetic_end_min > energy_threshold) {
01435         EnvJCComputeDiver diver(energy_threshold);
01436         dual_capa_tree.DiveInTree(&diver);
01437         const int64 enjv = diver.GetEnvJC(dual_capa_tree.result());
01438         const int64 numerator = enjv - energy_threshold;
01439         const int64 diff = MathUtil::CeilOfRatio(numerator, demand);
01440         update = std::max(update, diff);
01441       }
01442       updates->SetUpdate(i, update);
01443     }
01444     updates->SetUpToDate();
01445   }

01450                             int end_max_index) {

01453       ComputeConditionalStartMins(demand);

01459   // Propagates by discovering all end-after-end relationships purely based on
01460   // comparisons between end mins and end maxes: there is no energetic reasoning
01461   // here, but this allow updates that the standard edge-finding detection rule
01462   // misses.
01463   // See paragraph 6.2 in http://vilim.eu/petr/cp2009.pdf.
01464   void PropagateBasedOnEndMinGreaterThanEndMax() {
01465     int end_max_index = 0;
01466     int64 max_start_min = kint64min;
01467     for (int i = 0; i < size_; ++i) {
01468       CumulativeIndexedTask* const task = by_end_min_[i];
01469       const int64 end_min = task->EndMin();
01470       while (end_max_index < size_ &&
01471              by_end_max_[end_max_index]->EndMax() <= end_min) {
01472         max_start_min = std::max(max_start_min,
01473                                  by_end_max_[end_max_index]->StartMin());
01474         ++end_max_index;
01475       }
01476       if (end_max_index > 0 &&
01477           task->StartMin() <= max_start_min &&
01478           task->EndMax() > task->EndMin()) {
01479         DCHECK_LE(by_end_max_[end_max_index - 1]->EndMax(), end_min);
01480         // The update is valid and may be interesting:
01481         // * If task->StartMin() > max_start_min, then all tasks whose end_max
01482         //     is less than or equal to end_min have a start min that is less
01483         //     than task->StartMin(). In this case, any update we could
01484         //     compute would also be computed by the standard edge-finding
01485         //     rule. It's better not to compute it, then: it may not be
01486         //     needed.
01487         // * If task->EndMax() <= task->EndMin(), that means the end max is
01488         //     bound. In that case, 'task' itself belong to the set of tasks
01489         //     that must end before end_min, which may cause the result of
01490         //     ConditionalStartMin(task, end_max_index - 1) not to be a valid
01491         //     update.
01492         const int64 update = ConditionalStartMin(*task, end_max_index - 1);
01493         StartMinUpdater startMinUpdater(task->mutable_interval(), update);
01494         new_start_min_.push_back(startMinUpdater);
01495       }
01496     }
01497   }

01608 // Cumulative time-table.
01609 //
01610 // This class implements a propagator for the CumulativeConstraint which is not
01611 // incremental, and where a call to InitialPropagate() takes time which is
01612 // O(n^2) and Omega(n log n) with n the number of cumulative tasks.
01613 //
01614 // Despite the high complexity, this propagator is needed, because of those
01615 // implemented, it is the only one that satisfy that if all instantiated, no
01616 // contradiction will be detected if and only if the constraint is satisfied.
01617 //
01618 // The implementation is quite naive, and could certainly be improved, for
01619 // example by maintaining the profile incrementally.
01620 class CumulativeTimeTable : public Constraint {
01621  public:
01622   CumulativeTimeTable(Solver* const solver,
01623                       const std::vector<CumulativeTask*>& tasks,
01624                       int64 capacity)
01625     : Constraint(solver),
01626       by_start_min_(tasks),
01627       capacity_(capacity) {
01628     // There may be up to 2 delta's per interval (one on each side),
01629     // plus two sentinels
01630     const int profile_max_size = 2 * NumTasks() + 2;
01631     profile_non_unique_time_.reserve(profile_max_size);
01632     profile_unique_time_.reserve(profile_max_size);
01633   }
01634   virtual void InitialPropagate() {
01635     BuildProfile();
01636     PushTasks();
01637   }
01638   virtual void Post() {
01639     Demon* d = MakeDelayedConstraintDemon0(
01640         solver(),
01641         this,
01642         &CumulativeTimeTable::InitialPropagate,
01643         "InitialPropagate");
01644     for (int i = 0; i < NumTasks(); ++i) {
01645       by_start_min_[i]->mutable_interval()->WhenAnything(d);
01646     }
01647   }
01648   int NumTasks() const { return by_start_min_.size(); }

01651     LOG(FATAL) << "Should not be visited";
01652   }

01654  private:
01655   // Build the usage profile. Runs in O(n log n).
01656   void BuildProfile() {
01657     // Build profile with non unique time
01658     profile_non_unique_time_.clear();
01659     for (int i = 0; i < NumTasks(); ++i) {
01660       const CumulativeTask* task = by_start_min_[i];
01661       const IntervalVar* const interval = task->interval();
01662       const int64 start_max = interval->StartMax();
01663       const int64 end_min = interval->EndMin();
01664       if (interval->MustBePerformed() && start_max < end_min) {
01665         const int64 demand = task->demand();
01666         profile_non_unique_time_.push_back(ProfileDelta(start_max, +demand));
01667         profile_non_unique_time_.push_back(ProfileDelta(end_min, -demand));
01668       }
01669     }
01670     // Sort
01671     std::sort(profile_non_unique_time_.begin(),
01672               profile_non_unique_time_.end(),
01673               TimeLessThan);
01674     // Build profile with unique times
01675     int64 usage = 0;
01676     profile_unique_time_.clear();
01677     profile_unique_time_.push_back(ProfileDelta(kint64min, 0));
01678     for (int i = 0; i < profile_non_unique_time_.size(); ++i) {
01679       const ProfileDelta& profile_delta = profile_non_unique_time_[i];
01680       if (profile_delta.time == profile_unique_time_.back().time) {
01681         profile_unique_time_.back().delta += profile_delta.delta;
01682       } else {
01683         if (usage > capacity_) {
01684           solver()->Fail();
01685         }
01686         profile_unique_time_.push_back(profile_delta);
01687       }
01688       usage += profile_delta.delta;
01689     }
01690     DCHECK_EQ(0, usage);
01691     profile_unique_time_.push_back(ProfileDelta(kint64max, 0));
01692   }

01694   // Update the start min for all tasks. Runs in O(n^2) and Omega(n).
01695   void PushTasks() {
01696     Sort(&by_start_min_, TaskStartMinLessThan);
01697     int64 usage = 0;
01698     int profile_index = 0;
01699     for (int task_index = 0 ; task_index < NumTasks(); ++task_index) {
01700       CumulativeTask* const task = by_start_min_[task_index];
01701       while (task->interval()->StartMin() >
01702              profile_unique_time_[profile_index].time) {
01703         DCHECK(profile_index < profile_unique_time_.size());
01704         ++profile_index;
01705         usage += profile_unique_time_[profile_index].delta;
01706       }
01707       PushTask(task, profile_index, usage);
01708     }
01709   }

01711   // Push the given task to new_start_min, defined as the smallest integer such
01712   // that the profile usage for all tasks, excluding the current one, does not
01713   // exceed capacity_ - task->demand() on the interval
01714   // [new_start_min, new_start_min + task->interval()->DurationMin() ).
01715   void PushTask(CumulativeTask* const task,
01716                 int profile_index,
01717                 int64 usage) {
01718     // Init
01719     const IntervalVar* const interval = task->interval();
01720     const int64 demand = task->demand();
01721     const int64 residual_capacity = capacity_ - demand;
01722     const int64 duration = task->interval()->DurationMin();
01723     const ProfileDelta& first_prof_delta = profile_unique_time_[profile_index];
01724 
01725     int64 new_start_min = interval->StartMin();
01726 
01727     DCHECK_GE(first_prof_delta.time, interval->StartMin());
01728     // The check above is with a '>='. Let's first treat the '>' case
01729     if (first_prof_delta.time > interval->StartMin()) {
01730       // There was no profile delta at a time between interval->StartMin()
01731       // (included) and the current one.
01732       // As we don't delete delta's of 0 value, this means the current task
01733       // does not contribute to the usage before:
01734       DCHECK(
01735           (interval->StartMax() >= first_prof_delta.time) ||
01736           (interval->StartMax() >= interval->EndMin()));
01737       // The 'usage' given in argument is valid at first_prof_delta.time. To
01738       // compute the usage at the start min, we need to remove the last delta.
01739       const int64 usage_at_start_min = usage - first_prof_delta.delta;
01740       if (usage_at_start_min > residual_capacity) {
01741         new_start_min = profile_unique_time_[profile_index].time;
01742       }
01743     }
01744 
01745     // Influence of current task
01746     const int64 start_max = interval->StartMax();
01747     const int64 end_min = interval->EndMin();
01748     ProfileDelta delta_start(start_max, 0);
01749     ProfileDelta delta_end(end_min, 0);
01750     if (interval->MustBePerformed() && start_max < end_min) {
01751       delta_start.delta = +demand;
01752       delta_end.delta = -demand;
01753     }
01754     while (profile_unique_time_[profile_index].time <
01755            duration + new_start_min) {
01756       const ProfileDelta& profile_delta = profile_unique_time_[profile_index];
01757       DCHECK(profile_index < profile_unique_time_.size());
01758       // Compensate for current task
01759       if (profile_delta.time == delta_start.time) {
01760         usage -= delta_start.delta;
01761       }
01762       if (profile_delta.time == delta_end.time) {
01763         usage -= delta_end.delta;
01764       }
01765       // Increment time
01766       ++profile_index;
01767       DCHECK(profile_index < profile_unique_time_.size());
01768       // Does it fit?
01769       if (usage > residual_capacity) {
01770         new_start_min = profile_unique_time_[profile_index].time;
01771       }
01772       usage += profile_unique_time_[profile_index].delta;
01773     }
01774     task->mutable_interval()->SetStartMin(new_start_min);
01775   }

01884   // Post a Sequence constraint for tasks that requires strictly more than half
01885   // of the resource
01886   void PostHighDemandSequenceConstraint() {
01887     Constraint* constraint = NULL;
01888     {  // Need a block to avoid memory leaks in case the AddConstraint fails
01889       std::vector<IntervalVar*> high_demand_intervals;
01890       high_demand_intervals.reserve(size_);
01891       for (int i = 0; i < size_; ++i) {
01892         const int64 demand = tasks_[i]->demand();
01893         // Consider two tasks with demand d1 and d2 such that
01894         // d1 * 2 > capacity_ and d2 * 2 > capacity_.
01895         // Then d1 + d2 = 1/2 (d1 * 2 + d2 * 2)
01896         //              > 1/2 (capacity_ + capacity_)
01897         //              > capacity_.
01898         // Therefore these two tasks cannot overlap.
01899         if (demand * 2 > capacity_ && tasks_[i]->interval()->MayBePerformed()) {
01900           high_demand_intervals.push_back(tasks_[i]->mutable_interval());
01901         }
01902       }
01903       if (high_demand_intervals.size() >= 2) {
01904         // If there are less than 2 such intervals, the constraint would do
01905         // nothing
01906         string seq_name = StrCat(name(), "-HighDemandSequence");
01907         constraint = solver()->MakeDisjunctiveConstraint(high_demand_intervals,
01908                                                          seq_name);
01909       }
01910     }
01911     if (constraint != NULL) {
01912       solver()->AddConstraint(constraint);
01913     }
