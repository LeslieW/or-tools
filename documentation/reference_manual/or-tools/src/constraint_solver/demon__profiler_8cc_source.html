00165   virtual void BeginFail() {
00166     if (active_demon_ != NULL) {
00167       DemonRuns* const demon_run = demon_map_[active_demon_];
00168       if (demon_run != NULL) {
00169         demon_run->add_end_time(CurrentTime());
00170         demon_run->set_failures(demon_run->failures() + 1);
00171       }
00172       active_demon_ = NULL;
00173       // active_constraint_ can be non null in case of initial propagation.
00174       active_constraint_ = NULL;
00175     } else if (active_constraint_ != NULL) {
00176       ConstraintRuns* const ct_run = constraint_map_[active_constraint_];
00177       if (ct_run != NULL) {
00178         ct_run->add_initial_propagation_end_time(CurrentTime());
00179         ct_run->set_failures(1);
00180       }
00181       active_constraint_ = NULL;
00182     }

00325   // Export Information
00326   void ExportInformation(const Constraint* const constraint,
00327                          int64* const fails,
00328                          int64* const initial_propagation_runtime,
00329                          int64* const demon_invocations,
00330                          int64* const total_demon_runtime,
00331                          int* demons) {
00332     CHECK_NOTNULL(constraint);
00333     ConstraintRuns* const ct_run = constraint_map_[constraint];
00334     CHECK_NOTNULL(ct_run);
00335     *demon_invocations = 0;
00336     *fails = ct_run->failures();
00337     *initial_propagation_runtime = 0;
00338     for (int i = 0; i < ct_run->initial_propagation_start_time_size(); ++i) {
00339       *initial_propagation_runtime +=
00340           ct_run->initial_propagation_end_time(i) -
00341           ct_run->initial_propagation_start_time(i);
00342     }
00343     *total_demon_runtime = 0;
00344 
00345     // Gather information.
00346     *demons = ct_run->demons_size();
00347     CHECK_EQ(*demons, demons_per_constraint_[constraint].size());
00348     for (int demon_index = 0; demon_index < *demons; ++demon_index) {
00349       const DemonRuns& demon_runs = ct_run->demons(demon_index);
00350       *fails += demon_runs.failures();
00351       CHECK_EQ(demon_runs.start_time_size(), demon_runs.end_time_size());
00352       const int runs = demon_runs.start_time_size();
00353       *demon_invocations += runs;
00354       for (int run_index = 0; run_index < runs; ++run_index) {
00355         const int64 demon_time = demon_runs.end_time(run_index) -
00356             demon_runs.start_time(run_index);
00357         *total_demon_runtime += demon_time;
00358       }
00359     }
00360   }

00362   void ExportInformation(DemonRuns* const demon_runs,
00363                          int64* const demon_invocations,
00364                          int64* const fails,
00365                          int64* const total_demon_runtime,
00366                          double* const mean_demon_runtime,
00367                          double* const median_demon_runtime,
00368                          double* const stddev_demon_runtime) {
00369     CHECK_NOTNULL(demon_runs);
00370     CHECK_EQ(demon_runs->start_time_size(), demon_runs->end_time_size());
00371 
00372     const int runs = demon_runs->start_time_size();
00373     *demon_invocations = runs;
00374     *fails = demon_runs->failures();
00375     *total_demon_runtime = 0;
00376     *mean_demon_runtime = 0.0;
00377     *median_demon_runtime = 0.0;
00378     *stddev_demon_runtime = 0.0;
00379     std::vector<double> runtimes;
00380     for (int run_index = 0; run_index < runs; ++run_index) {
00381       const int64 demon_time = demon_runs->end_time(run_index) -
00382           demon_runs->start_time(run_index);
00383       *total_demon_runtime += demon_time;
00384       runtimes.push_back(demon_time);
00385     }
00386     // Compute mean.
00387     if (runtimes.size()) {
00388       *mean_demon_runtime = (1.0L * *total_demon_runtime) / runtimes.size();
00389 
00390       // Compute median.
00391       sort(runtimes.begin(), runtimes.end());
00392       const int pivot = runtimes.size() / 2;
00393 
00394       if (runtimes.size() == 1) {
00395         *median_demon_runtime = runtimes[0];
00396       } else {
00397         *median_demon_runtime = runtimes.size() % 2 == 1 ?
00398             runtimes[pivot] :
00399             (runtimes[pivot - 1] + runtimes[pivot]) / 2.0;
00400       }
00401 
00402       // Compute standard deviation.
00403       double total_deviation = 0.0f;
00404 
00405       for (int i = 0; i < runtimes.size(); ++i) {
00406         total_deviation += pow(runtimes[i] - *mean_demon_runtime, 2);
00407       }
00408 
00409       *stddev_demon_runtime = sqrt(total_deviation / runtimes.size());
00410     }
00411   }
