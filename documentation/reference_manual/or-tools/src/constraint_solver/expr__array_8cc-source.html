00706   virtual void InitialPropagate() {
00707     // Copy vars to leaf nodes.
00708     for (int i = 0; i < size_; ++i) {
00709       InitLeaf(i, vars_[i]->Min(), vars_[i]->Max());
00710     }
00711 
00712     // Compute up.
00713     for (int i = MaxDepth() - 1; i >= 0; --i) {
00714       for (int j = 0; j < Width(i); ++j) {
00715         int64 max_min = kint64min;
00716         int64 max_max = kint64min;
00717         const int block_start = ChildStart(j);
00718         const int block_end = ChildEnd(i, j);
00719         for (int k = block_start; k <= block_end; ++k) {
00720           max_min = std::max(max_min, Min(i + 1, k));
00721           max_max = std::max(max_max, Max(i + 1, k));
00722         }
00723         InitNode(i, j, max_min, max_max);
00724       }

00738     // Nothing to do?
00739     if (new_min <= Min(depth, position) && new_max >= Max(depth, position)) {
00740       return;
00741     }
00742 
00743     // Leaf node -> push to leaf var.
00744     if (IsLeaf(depth)) {
00745       vars_[position]->SetRange(new_min, new_max);
00746       return;
00747     }
00748 
00749     const int64 node_min = Min(depth, position);
00750     const int64 node_max = Max(depth, position);
00751 
00752     int candidate = -1;
00753     int active = 0;
00754     const int block_start = ChildStart(position);
00755     const int block_end = ChildEnd(depth, position);
00756 
00757     if (node_min < new_min) {
00758       // Look if only one candidat to push the max down.
00759       for (int i = block_start; i <= block_end; ++i) {
00760         if (Max(depth + 1, i) >= new_min) {
00761           if (active++ >= 1) {
00762             break;
00763           }
00764           candidate = i;
00765         }
00766       }
00767       if (active == 0) {
00768         solver()->Fail();
00769       }
00770     }
00771 
00772     if (node_max > new_max) {
00773       for (int i = block_start; i <= block_end; ++i) {
00774         if (i == candidate && active == 1) {
00775           PushDown(depth + 1, i, new_min, new_max);
00776         } else {
00777           PushDown(depth + 1, i, Min(depth + 1, i), new_max);
00778         }
00779       }
00780     } else if (active == 1) {
00781       PushDown(depth + 1, candidate, new_min, Max(depth + 1, candidate));
00782     }
00783   }

01619 struct Container {
01620   IntVar* var;
01621   int64 coef;
01622   Container(IntVar* v, int64 c) : var(v), coef(c) {}
01623   bool operator<(const Container& c) const { return (coef < c.coef); }

01626 // This method will sort both vars and coefficients in increasing
01627 // coefficient order. Vars with null coefficients will be
01628 // removed. Bound vars will be collected and the sum of the
01629 // corresponding products (when the var is bound to 1) is returned by
01630 // this method.
01631 // If keep_inside is true, the constant will be added back into the
01632 // scalprod as IntConst(1) * constant.
01633 int64 SortBothChangeConstant(IntVar** const vars,
01634                              int64* const coefs,
01635                              int* const size,
01636                              bool keep_inside) {
01637   CHECK_NOTNULL(vars);
01638   CHECK_NOTNULL(coefs);
01639   CHECK_NOTNULL(size);
01640   if (*size == 0) {
01641     return 0;
01642   }
01643   int64 cst = 0;
01644   std::vector<Container> to_sort;
01645   for (int index = 0; index < *size; ++index) {
01646     if (vars[index]->Bound()) {
01647       cst += coefs[index] * vars[index]->Min();
01648     } else if (coefs[index] != 0) {
01649       to_sort.push_back(Container(vars[index], coefs[index]));
01650     }
01651   }
01652   if (keep_inside && cst != 0) {
01653     CHECK_LT(to_sort.size(), *size);
01654     Solver* const solver = vars[0]->solver();
01655     to_sort.push_back(Container(solver->MakeIntConst(1), cst));
01656     cst = 0;
01657   }
01658   std::sort(to_sort.begin(), to_sort.end());
01659   *size = to_sort.size();
01660   for (int index = 0; index < *size; ++index) {
01661     vars[index] = to_sort[index].var;
01662     coefs[index] = to_sort[index].coef;
01663   }
01664   return cst;
01665 }

02541     return cst == 0 ? solver->MakeTrueConstraint()
02542         : solver->MakeFalseConstraint();

02545       AreAllPositive<T>(coefficients, size) &&

03203 namespace {
03204 template<class T> IntExpr* MakeScalProdFct(Solver* solver,
03205                                            const std::vector<IntVar*>& pre_vars,
03206                                            const std::vector<T>& pre_coefs) {
03207   hash_map<const IntExpr*, int64> map;
03208   ExprLinearizer lin(&map);
03209   for (int i = 0; i < pre_vars.size(); ++i) {
03210     lin.Visit(pre_vars[i], pre_coefs[i]);
03211   }
03212   const int64 constant = lin.Constant();
03213   std::vector<IntVar*> vars;
03214   std::vector<T> coefs;
03215   for (ConstIter<hash_map<const IntExpr*, int64> > iter(map);
03216        !iter.at_end();
03217        ++iter) {
03218      vars.push_back(const_cast<IntExpr*>(iter->first)->Var());
03219      coefs.push_back(iter->second);
03220   }
03221   if (constant != 0) {
03222     vars.push_back(solver->MakeIntConst(1));
03223     coefs.push_back(constant);
03224   }
03225 
03226   const int size = vars.size();
03227   if (vars.empty() || AreAllNull<T>(coefs.data(), size)) {
03228     return solver->MakeIntConst(0LL);
03229   }
03230   if (AreAllBoundOrNull(vars.data(), coefs.data(), size)) {
03231     int64 cst = 0;
03232     for (int i = 0; i < size; ++i) {
03233       cst += vars[i]->Min() * coefs[i];
03234     }
03235     return solver->MakeIntConst(cst);
03236   }
03237   if (AreAllOnes(coefs.data(), size)) {
03238     return solver->MakeSum(vars);
03239   }
03240   if (size == 1) {
03241     return solver->MakeProd(vars[0], coefs[0]);
03242   }
03243   if (AreAllBooleans(vars.data(), size) && size > 2) {
03244     if (AreAllPositive<T>(coefs.data(), size)) {
03245       return solver->RegisterIntExpr(solver->RevAlloc(
03246           new PositiveBooleanScalProd(
03247               solver, vars.data(), size, coefs.data())));
03248     } else {
03249       // If some coefficients are non-positive, partition coefficients in two
03250       // sets, one for the positive coefficients P and one for the negative
03251       // ones N.
03252       // Create two PositiveBooleanScalProd expressions, one on P (s1), the
03253       // other on Opposite(N) (s2).
03254       // The final expression is then s1 - s2.
03255       // If P is empty, the expression is Opposite(s2).
03256       std::vector<T> positive_coefs;
03257       std::vector<T> negative_coefs;
03258       std::vector<IntVar*> positive_coef_vars;
03259       std::vector<IntVar*> negative_coef_vars;
03260       for (int i = 0; i < size; ++i) {
03261         const T coef = coefs[i];
03262         if (coef > 0) {
03263           positive_coefs.push_back(coef);
03264           positive_coef_vars.push_back(vars[i]);
03265         } else if (coef < 0) {
03266           negative_coefs.push_back(-coef);
03267           negative_coef_vars.push_back(vars[i]);
03268         }
03269       }
03270       CHECK_GT(negative_coef_vars.size(), 0);
03271       IntExpr* const negatives =
03272           solver->RegisterIntExpr(solver->RevAlloc(
03273               new PositiveBooleanScalProd(solver,
03274                                           negative_coef_vars.data(),
03275                                           negative_coef_vars.size(),
03276                                           negative_coefs.data())));
03277       if (!positive_coefs.empty()) {
03278         IntExpr* const positives =
03279             solver->RegisterIntExpr(solver->RevAlloc(
03280                 new PositiveBooleanScalProd(solver,
03281                                             positive_coef_vars.data(),
03282                                             positive_coef_vars.size(),
03283                                             positive_coefs.data())));
03284         // Cast to var to avoid slow propagation; all operations on the expr are
03285         // O(n)!
03286         return solver->MakeDifference(positives->Var(), negatives->Var());
03287       } else {
03288         return solver->MakeOpposite(negatives);
03289       }
03290     }
03291   }
03292   std::vector<IntVar*> terms;
03293   for (int i = 0; i < size; ++i) {
03294     terms.push_back(solver->MakeProd(vars[i], coefs[i])->Var());
03295   }
03296   return solver->MakeSum(terms);
03297 }
