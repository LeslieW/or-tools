00198   void ComputeDeltaDomain(std::vector<int>* delta) {
00199     delta->clear();
00200     // we iterate the delta of the variable
00201     //
00202     // ATTENTION code for or-tools: the delta iterator does not
00203     // include the values between oldmin and min and the values
00204     // between max and oldmax.
00205     //
00206     // therefore we decompose the iteration into 3 parts
00207     // - from oldmin to min
00208     // - for the deleted values between min and max
00209     // - from max to oldmax
00210 
00211     // First iteration: from old_min to min.
00212     const int64 old_min_domain = var_->OldMin();
00213     const int64 min_domain = var_->Min();
00214     const int64 max_domain = var_->Max();
00215     for (int64 val = old_min_domain; val < min_domain; ++val) {
00216       const int index = column_.IndexFromValue(val);
00217       if (index != -1) {  // -1 means not in column.
00218         delta->push_back(index);
00219       }
00220     }
00221     // Second iteration: "delta" domain iteration.
00222     IntVarIterator* const it = delta_domain_iterator_;
00223     for (it->Init(); it->Ok(); it->Next()) {
00224       const int64 value = it->Value();
00225       if (value > min_domain && value < max_domain) {
00226         const int index = column_.IndexFromValue(it->Value());
00227         if (index != -1) {  // -1 means not in column.
00228           delta->push_back(index);
00229         }
00230       }
00231     }
00232     // Third iteration: from max to old_max.
00233     const int64 old_max_domain = var_->OldMax();
00234     for (int64 val = max_domain + 1; val <= old_max_domain; ++val) {
00235       const int index = column_.IndexFromValue(val);
00236       if (index != -1) {  // -1 means not in column.
00237         delta->push_back(index);
00238       }
00239     }
00240   }

00282   void OverwriteTuples(const std::vector<int>& tuples) {
00283     for (int k = 0; k < active_values_.Size(); k++) {
00284       tuples_per_value_[active_values_.Element(k)]->Clear(solver_);
00285     }
00286     for (int i = 0; i < tuples.size(); ++i) {
00287       const int tuple_index = tuples[i];
00288       const int value_index = column_.ValueIndex(tuple_index);
00289       tuples_per_value_[value_index]->Restore(solver_, tuple_index);
00290     }
00291 
00292     // We check for unsupported values and remove them.
00293     int count = 0;
00294     for (int k = active_values_.Size() - 1; k >= 0; k--) {
00295       const int64 value_index = active_values_.Element(k);
00296       if (tuples_per_value_[value_index]->Size() == 0) {
00297         active_values_.Remove(solver_, value_index);
00298         count++;
00299       }
00300     }
00301     // Removed values have been inserted after the last active value.
00302     for (int k = 0; k < count; ++k) {
00303       const int64 value_index = active_values_.RemovedElement(k);
00304       var_->RemoveValue(column_.ValueFromIndex(value_index));
00305     }
00306   }

00367   void FilterOneVariable(int var_index) {
00368     TableVar* const var = vars_[var_index];
00369     var->ComputeDeltaDomain(&delta_of_value_indices_);
00370     // We decide if we prefer to restart with the remaining set of
00371     // tuples, or if we incrementaly remove unsupported tuples.
00372     if (var->ShouldReset(delta_of_value_indices_)) {
00373       var->CollectTuplesToKeep(&tmp_tuples_);
00374       for (int i = 0; i < num_variables_; i++) {
00375         vars_[i]->OverwriteTuples(tmp_tuples_);
00376       }
00377     } else {
00378       var->CollectTuplesToRemove(delta_of_value_indices_, &tmp_tuples_);
00379       for (int i = 0; i < num_variables_; ++i) {
00380         vars_[i]->RemoveTuples(tmp_tuples_);
00381       }
00382     }
