01912                       const CPConstraintProto& proto) {
01913   std::vector<IntVar*> vars;
01914   VERIFY(builder->ScanArguments(ModelVisitor::kVarsArgument, proto, &vars));
01915   int64 bins = 0;
01916   VERIFY(builder->ScanArguments(ModelVisitor::kSizeArgument, proto, &bins));
01917   Pack* const pack = builder->solver()->MakePack(vars, bins);
01918   // Add dimensions. They are stored as extensions in the proto.
01919   for (int i = 0; i < proto.extensions_size(); ++i) {
01920     const CPExtensionProto& dimension_proto = proto.extensions(i);
01921     const int type_index = dimension_proto.type_index();
01922     if (IS_TYPE(type_index, builder, kUsageLessConstantExtension)) {
01923       VERIFY(AddUsageLessConstantDimension(pack, builder, dimension_proto));
01924     } else if (IS_TYPE(type_index, builder, kCountAssignedItemsExtension)) {
01925       VERIFY(AddCountAssignedItemsDimension(pack, builder, dimension_proto));
01926     } else if (IS_TYPE(type_index, builder, kCountUsedBinsExtension)) {
01927       VERIFY(AddCountUsedBinDimension(pack, builder, dimension_proto));
01928     } else if (IS_TYPE(type_index, builder, kUsageEqualVariableExtension)) {
01929       VERIFY(AddUsageEqualVariableDimension(pack, builder, dimension_proto));
01930     } else if (IS_TYPE(type_index,
01931                        builder,
01932                        kVariableUsageLessConstantExtension)) {
01933       VERIFY(AddVariableUsageLessConstantDimension(pack,
01934                                                    builder,
01935                                                    dimension_proto));
01936     } else if (IS_TYPE(type_index,
01937                        builder,
01938                        kWeightedSumOfAssignedEqualVariableExtension)) {

01940     } else {

02451   CPModelLoader builder(this);
02452   for (int i = 0; i < model_proto.tags_size(); ++i) {
02453     builder.AddTag(model_proto.tags(i));
02454   }
02455   for (int i = 0; i < model_proto.intervals_size(); ++i) {
02456     if (!builder.BuildFromProto(model_proto.intervals(i))) {
02457       LOG(ERROR) << "Interval variable proto "
02458                  << model_proto.intervals(i).DebugString()
02459                  << " was not parsed correctly";
02460       return false;
02461     }
02462   }
02463   for (int i = 0; i < model_proto.sequences_size(); ++i) {
02464     if (!builder.BuildFromProto(model_proto.sequences(i))) {
02465       LOG(ERROR) << "Sequence variable proto "
02466                  << model_proto.sequences(i).DebugString()
02467                  << " was not parsed correctly";
02468       return false;
02469     }
02470   }
02471   for (int i = 0; i < model_proto.expressions_size(); ++i) {
02472     if (!builder.BuildFromProto(model_proto.expressions(i))) {
02473       LOG(ERROR) << "Integer expression proto "
02474                  << model_proto.expressions(i).DebugString()
02475                  << " was not parsed correctly";
02476       return false;
02477     }
02478   }
02479   for (int i = 0; i < model_proto.constraints_size(); ++i) {
02480     Constraint* const constraint =
02481         builder.BuildFromProto(model_proto.constraints(i));
02482     if (constraint == NULL) {
02483       LOG(ERROR) << "Constraint proto "
02484                  << model_proto.constraints(i).DebugString()
02485                  << " was not parsed correctly";
02486       return false;
02487     }
02488     AddConstraint(constraint);
02489   }
02490   if (monitors != NULL) {
02491     if (model_proto.has_search_limit()) {
02492       monitors->push_back(MakeLimit(model_proto.search_limit()));
02493     }
02494     if (model_proto.has_objective()) {
02495       const CPObjectiveProto& objective_proto = model_proto.objective();
02496       IntVar* const objective_var =
02497           builder.IntegerExpression(objective_proto.objective_index())->Var();
02498       const bool maximize = objective_proto.maximize();
02499       const int64 step = objective_proto.step();
02500       OptimizeVar* const objective =
02501           MakeOptimize(maximize, objective_var, step);
02502       monitors->push_back(objective);
02503     }
02504   }
02505   return true;
02506 }
