00093 template <class C> void Double(C*** array_ptr, int* size_ptr) {
00094   DCHECK(array_ptr != NULL);
00095   DCHECK(size_ptr != NULL);
00096   C** const old_cell_array = *array_ptr;
00097   const int old_size = *size_ptr;
00098   (*size_ptr) *= 2;
00099   (*array_ptr) = new C*[(*size_ptr)];
00100   memset(*array_ptr, 0, (*size_ptr) * sizeof(**array_ptr));
00101   for (int i = 0; i < old_size; ++i) {
00102     C* tmp = old_cell_array[i];
00103     while (tmp != NULL) {
00104       C* const to_reinsert = tmp;
00105       tmp = tmp->next();
00106       const uint64 position = to_reinsert->Hash() % (*size_ptr);
00107       to_reinsert->set_next((*array_ptr)[position]);
00108       (*array_ptr)[position] = to_reinsert;
00109     }
00110   }
00111   delete [] (old_cell_array);
00112 }
