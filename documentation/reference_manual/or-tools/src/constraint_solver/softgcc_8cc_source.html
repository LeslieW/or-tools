00614   void FindSccVar(int64 k, FlowType ft) {
00615     // First variable matched to the value.
00616     int64* var_match = (ft == UF) ?
00617         under_variable_match_.get() :
00618         over_variable_match_.get();
00619 
00620     variable_dfs_[k] = dfs_--;
00621     variable_high_[k] = variable_dfs_[k];
00622     stack_[top_] = k;
00623     type_[top_] = 0;
00624     top_++;
00625     assert(top_ <= vars_.size() + num_values_ + 1);
00626 
00627     int64 var_min = vars_[k]->Min();
00628     int64 var_max = vars_[k]->Max();
00629     if (vars_[k]->Size() == var_max - var_min + 1) {  // Dense domain.
00630       for (int64 w = var_min; w <= var_max; w++) {
00631         // Go to every values of the domain of x that is not matched by x.
00632         if (var_match[k] != w) {
00633           if (value_dfs_[w - min_value_] == 0) {
00634             FindSccValue(w, ft);
00635             if (value_high_[w - min_value_] > variable_high_[k]) {
00636               variable_high_[k] = value_high_[w - min_value_];
00637             }
00638           } else if ((value_dfs_[w - min_value_] > variable_dfs_[k]) &&
00639                      (value_component_[w - min_value_] == 0)) {
00640             if (value_dfs_[w - min_value_] > variable_high_[k]) {
00641               variable_high_[k] = value_dfs_[w - min_value_];
00642             }
00643           }
00644         }
00645       }
00646     } else {
00647       for (int64 w = var_min; w <= var_max; w++) {
00648         // Go to every values of the domain of x that is not matched by x.
00649         if (var_match[k] != w && vars_[k]->Contains(w)) {
00650           if (value_dfs_[w - min_value_] == 0) {
00651             FindSccValue(w, ft);
00652             if (value_high_[w - min_value_] > variable_high_[k]) {
00653               variable_high_[k] = value_high_[w - min_value_];
00654             }
00655           } else if ((value_dfs_[w - min_value_] > variable_dfs_[k]) &&
00656                      (value_component_[w - min_value_] == 0)) {
00657             if (value_dfs_[w - min_value_] > variable_high_[k]) {
00658               variable_high_[k] = value_dfs_[w - min_value_];
00659             }
00660           }
00661         }
00662       }
00663     }
00664 
00665     // If x is matched go also to every other variables that are not
00666     // matched (path from x->source->x').
00667     if (var_match[k] != kUnassigned) {
00668       for (int64 i = 0; i < vars_.size(); i++) {
00669         if (var_match[i] == kUnassigned) {
00670           if (variable_dfs_[i] == 0) {
00671             FindSccVar(i, ft);
00672             if (variable_high_[i] > variable_high_[k]) {
00673               variable_high_[k] = variable_high_[i];
00674             }
00675           } else if ((variable_dfs_[i] > variable_dfs_[k]) &&
00676                      (variable_component_[i] == 0)) {
00677             if (variable_dfs_[i] > variable_high_[k]) {
00678               variable_high_[k] = variable_dfs_[i];
00679             }
00680           }
00681         }
00682       }
00683     }
00684 
00685     if (variable_high_[k] == variable_dfs_[k]) {
00686       component_++;
00687       do {
00688         assert(top_ > 0);
00689         int64 i = stack_[--top_];
00690         int64 t = type_[top_];
00691         if (t == 0) {
00692           variable_component_[i] = component_;
00693         } else if (t == 1) {
00694           value_component_[i - min_value_] = component_;
00695         } else {
00696           sink_component_ = component_;
00697         }
00698         if (t == 0 && i == k) {
00699           break;
00700         }
00701       } while (true);
00702     }
00703   }

00705   void FindSccValue(int64 v, FlowType ft) {
00706     // First variable matched to the value.
00707     int64*  value_match = (ft == UF) ?
00708         under_value_match_.get() :
00709         over_value_match_.get();
00710     // First variable matched to the value.
00711     const int64* const capa = (ft == UF) ? card_mins_.get() : card_max_.get();
00712     // First variable matched to the value.
00713     int64* next = (ft == UF) ? under_next_match_.get() : over_next_match_.get();
00714     // First variable matched to the value.
00715     int64* flow = (ft == UF) ? underflow_.get() : overflow_.get();
00716 
00717     value_dfs_[v - min_value_] = dfs_--;
00718     value_high_[v - min_value_] = value_dfs_[v - min_value_];
00719     stack_[top_] = v;
00720     type_[top_] = 1;
00721     top_++;
00722     assert(top_ <= vars_.size() + num_values_ + 1);
00723 
00724     // First go to the variables assigned to this value.
00725     int64 k = value_match[v - min_value_];
00726     while (k != kUnassigned) {
00727       if (variable_dfs_[k] == 0) {
00728         FindSccVar(k, ft);
00729         if (variable_high_[k] > value_high_[v - min_value_]) {
00730           value_high_[v - min_value_] = variable_high_[k];
00731         }
00732       } else if ((variable_dfs_[k] > value_dfs_[v - min_value_]) &&
00733                  (variable_component_[k] == 0)) {
00734         if (variable_dfs_[k] > value_high_[v - min_value_]) {
00735           value_high_[v - min_value_] = variable_dfs_[k];
00736         }
00737       }
00738       k = next[k];
00739     }
00740 
00741     // Next try to see if you can go to the sink.
00742     if (flow[v - min_value_] < capa[v - min_value_]) {
00743       // go to the sink
00744       if (sink_dfs_ == 0) {
00745         FindSccSink(ft);
00746         if (sink_high_ > value_high_[v - min_value_]) {
00747           value_high_[v - min_value_] = sink_high_;
00748         }
00749       } else if ((sink_dfs_ > value_dfs_[v - min_value_]) &&
00750                 (sink_component_ == 0)) {
00751         if (sink_dfs_ > value_high_[v - min_value_]) {
00752           value_high_[v - min_value_] = sink_dfs_;
00753         }
00754       }
00755     }
00756 
00757     if (value_high_[v - min_value_] == value_dfs_[v - min_value_]) {
00758       component_++;
00759       do {
00760         assert(top_ > 0);
00761         int64 i = stack_[--top_];
00762         int64 t = type_[top_];
00763         if (t == 0) {
00764           variable_component_[i] = component_;
00765         } else if (t == 1) {
00766           value_component_[i - min_value_] = component_;
00767         } else {
00768           sink_component_ = component_;
00769         }
00770         if (t == 1 && i == v) {
00771           break;
00772         }
00773       } while (true);
00774     }
00775   }

00777   void FindSccSink(FlowType ft) {
00778     // First variable matched to the value.
00779     int64* const var_match = (ft == UF) ?
00780         under_variable_match_.get() :
00781         over_variable_match_.get();
00782     // First variable matched to the value.
00783     int64* const flow = (ft == UF) ? underflow_.get() : overflow_.get();
00784 
00785     sink_dfs_ = dfs_--;
00786     sink_high_ = sink_dfs_;
00787     stack_[top_] = kUnassigned;
00788     type_[top_] = 2;
00789     top_++;
00790     assert(top_ <= vars_.size() + num_values_ + 1);
00791 
00792     // From the sink, I can go to the values if the flow in the value
00793     // is larger than the demand in these value.
00794 
00795     for (int64 i = 0; i < vars_.size(); i++) {
00796       int64 w = var_match[i];
00797       if (var_match[i] != kUnassigned) {
00798         if (flow[w - min_value_] > 0) {  // There is no minimum capacity.
00799           if (value_dfs_[w - min_value_] == 0) {
00800             FindSccValue(w, ft);
00801             if (value_high_[w - min_value_] > sink_high_) {
00802               sink_high_ = value_high_[w - min_value_];
00803             }
00804           } else if ((value_dfs_[w - min_value_] > sink_dfs_) &&
00805                     (value_component_[w - min_value_] == 0)) {
00806             if (value_dfs_[w - min_value_] > sink_high_) {
00807               sink_high_ = value_dfs_[w - min_value_];
00808             }
00809           }
00810         }
00811       }
00812     }
00813 
00814     // From the sink we can also go the variables that are not matched.
00815 
00816     for (int64 i = 0; i < vars_.size(); i++) {
00817       if (var_match[i] == kUnassigned) {
00818         if (variable_dfs_[i] == 0) {
00819           FindSccVar(i, ft);
00820           if (variable_high_[i] > sink_high_) {
00821             sink_high_ = variable_high_[i];
00822           }
00823         } else if ((variable_dfs_[i] > sink_dfs_) &&
00824                   (variable_component_[i] == 0)) {
00825           if (variable_dfs_[i] > sink_high_) {
00826             sink_high_ = variable_dfs_[i];
00827           }
00828         }
00829       }
00830     }
00831 
00832     if (sink_high_ == sink_dfs_) {
00833       component_++;
00834       do {
00835         assert(top_ > 0);
00836         int64 i = stack_[--top_];
00837         int64 t = type_[top_];
00838         if (t == 0) {
00839           variable_component_[i] = component_;
00840         } else if (t == 1) {
00841           value_component_[i - min_value_] = component_;
00842         } else {
00843           sink_component_ = component_;
00844         }
00845         if (t == 2) {
00846           break;
00847         }
00848       } while (true);
00849     }

00853   void Prune(int64 min_violation) {
00854     if (min_violation < violation_var_->Max() - 1) {
00855       return;  // The constraint is GAC.
00856     }
00857 
00858     // We compute the SCC in Gu and Go and also if a variable is
00859     // matched in every maximum matching in Gu and Go.
00860 
00861     FindScc(UF);
00862     ComputeIsVarAlwaysMatched(UF);
00863     for (int64 v = 0; v < num_values_; v++) {
00864       under_value_component_[v] = value_component_[v];
00865     }
00866     for (int64 k = 0; k < vars_.size(); k++) {
00867       under_variable_component_[k] = variable_component_[k];
00868     }
00869 
00870     FindScc(OF);
00871     ComputeIsVarAlwaysMatched(OF);
00872 
00873     if (min_violation == violation_var_->Max() - 1) {
00874       for (int64 k = 0; k < vars_.size(); k++) {
00875         if (over_variable_match_[k] == kUnassigned) {
00876           continue;  // All values of this variable are consistent.
00877         }
00878         const int64 var_min = vars_[k]->Min();
00879         const int64 var_max = vars_[k]->Max();
00880         for (int64 w = var_min; w <= var_max; w++) {
00881           if (under_variable_match_[k] != w && over_variable_match_[k] != w) {
00882             if ((under_variable_component_[k] !=
00883                  under_value_component_[w - min_value_] &&
00884                  (card_mins_[w - min_value_] > 0 ||
00885                   is_var_always_matched_in_underflow_[k])) &&
00886                 (variable_component_[k] != value_component_[w - min_value_] &&
00887                  (card_max_[w - min_value_] > 0 ||
00888                   is_var_always_matched_in_overflow_[k])) ) {
00889               vars_[k]->RemoveValue(w);
00890             }
00891           }
00892         }
00893       }
00894     } else if (min_violation == violation_var_->Max()) {
00895       // Under-flow filtering.
00896       for (int64 k = 0; k < vars_.size(); k++) {
00897         if (over_variable_match_[k] == kUnassigned) {
00898           continue;  // All values of this variable are consistent.
00899         }
00900         const int64 var_min = vars_[k]->Min();
00901         const int64 var_max = vars_[k]->Max();
00902         for (int64 w = var_min; w <= var_max; w++) {
00903           if (under_variable_match_[k] != w && over_variable_match_[k] != w) {
00904             if (under_variable_component_[k] !=
00905                 under_value_component_[w - min_value_] &&
00906                 (card_mins_[w - min_value_] > 0 ||
00907                  is_var_always_matched_in_underflow_[k])) {
00908               vars_[k]->RemoveValue(w);
00909             }
00910           }
00911         }
00912       }
00913       // Over-flow filtering.
00914       for (int64 k = 0; k < vars_.size(); k++) {
00915         if (over_variable_match_[k] == kUnassigned) {
00916           continue;  // All values of this variable are consistent.
00917         }
00918         const int64 var_min = vars_[k]->Min();
00919         const int64 var_max = vars_[k]->Max();
00920         for (int64 w = var_min; w <= var_max; w++) {
00921           if (over_variable_match_[k] != w) {
00922             if (variable_component_[k] != value_component_[w - min_value_] &&
00923                 (card_max_[w - min_value_] > 0 ||
00924                  is_var_always_matched_in_overflow_[k])) {
00925               vars_[k]->RemoveValue(w);
00926             }
00927           }
00928         }
00929       }
