00200 void SequenceVar::ComputePossibleFirstsAndLasts(
00201     std::vector<int>* const possible_firsts,
00202     std::vector<int>* const possible_lasts) {
00203   hash_set<int> to_check;
00204   for (int i = 0; i < size_; ++i) {
00205     if (intervals_[i]->MayBePerformed()) {
00206       to_check.insert(i);
00207     }
00208   }
00209   int first = 0;
00210   while (nexts_[first]->Bound()) {
00211     first = nexts_[first]->Min();
00212     to_check.erase(first - 1);
00213     if (first == next_size_) {
00214       return;
00215     }
00216   }
00217 
00218   IntVar* const forward_var = nexts_[first];
00219   std::vector<int> candidates;
00220   int64 smallest_start_max = kint64max;
00221   int ssm_support = -1;
00222   for (int64 i = forward_var->Min(); i <= forward_var->Max(); ++i) {
00223     // TODO(lperron): use domain iterator.
00224     if (i != 0 && i < size_ + 1 && forward_var->Contains(i)) {
00225       const int candidate = i - 1;
00226       candidates.push_back(candidate);
00227       if (intervals_[candidate]->MustBePerformed()) {
00228         if (smallest_start_max > intervals_[candidate]->StartMax()) {
00229           smallest_start_max = intervals_[candidate]->StartMax();
00230           ssm_support = candidate;
00231         }
00232       }
00233     }
00234   }
00235   for (int i = 0; i < candidates.size(); ++i) {
00236     const int candidate = candidates[i];
00237     if (candidate == ssm_support ||
00238         intervals_[candidate]->EndMin() <= smallest_start_max) {
00239       possible_firsts->push_back(candidate);
00240     }
00241   }
00242 
00243   UpdatePrevious();
00244   int last = next_size_;
00245   while (previous_[last] != -1) {
00246     last = previous_[last];
00247     to_check.erase(last - 1);
00248   }
00249 
00250   candidates.clear();
00251   int64 biggest_end_min = kint64min;
00252   int bem_support = -1;
00253   for (ConstIter<hash_set<int> > it(to_check); !it.at_end(); ++it) {
00254     const int candidate = *it;
00255     if (nexts_[candidate + 1]->Contains(last)) {
00256       candidates.push_back(candidate);
00257       if (intervals_[candidate]->MustBePerformed()) {
00258         if (biggest_end_min < intervals_[candidate]->EndMin()) {
00259           biggest_end_min = intervals_[candidate]->EndMin();
00260           bem_support = candidate;
00261         }
00262       }
00263     }
00264   }
00265 
00266   for (int i = 0; i < candidates.size(); ++i) {
00267     const int candidate = candidates[i];
00268     if (candidate == bem_support ||
00269         intervals_[candidate]->StartMax() >= biggest_end_min) {
00270       possible_lasts->push_back(candidate);
00271     }
00272   }

00653         if (candidate_possible_firsts.size() == 0) {
00654           s->Fail();
00655         }
00656         // Only 1 candidate, and non optional: ranking without branching.
00657         if (candidate_possible_firsts.size() == 1 &&
00658             candidate_sequence->Interval(
00659                 candidate_possible_firsts.back())->MustBePerformed()) {
00660           *best_sequence = candidate_sequence;
00661           best_possible_firsts_ = candidate_possible_firsts;
00662           return true;
00663         }
00664 
00665         // Evaluating the sequence.
00666         int64 hmin, hmax, dmin, dmax;
00667         candidate_sequence->HorizonRange(&hmin, &hmax);
00668         candidate_sequence->DurationRange(&dmin, &dmax);
00669         int64 ahmin, ahmax;
00670         candidate_sequence->ActiveHorizonRange(&ahmin, &ahmax);
00671         const int64 current_slack = (hmax - hmin - dmax);
00672         if (current_slack < best_slack ||
00673             (current_slack == best_slack && ahmin < best_ahmin)) {
00674           best_slack = current_slack;
00675           *best_sequence = candidate_sequence;
00676           best_possible_firsts_ = candidate_possible_firsts;
00677           best_ahmin = ahmin;
00678         }
00679       }
00680     }
00681     return *best_sequence != NULL;
00682   }

00684   bool FindSequenceVarRandomly(Solver* const s,
00685                                SequenceVar** const best_sequence) const {
00686     std::vector<int> all_candidates;
00687     std::vector<std::vector<int> > all_possible_firsts;
00688     for (int i = 0; i < size_; ++i) {
00689       SequenceVar* const candidate_sequence = sequences_[i];
00690       int ranked = 0;
00691       int not_ranked = 0;
00692       int unperformed = 0;
00693       candidate_sequence->ComputeStatistics(&ranked, &not_ranked, &unperformed);
00694       if (not_ranked > 0) {
00695         std::vector<int> candidate_possible_firsts;
00696         std::vector<int> candidate_possible_lasts;
00697         candidate_sequence->ComputePossibleFirstsAndLasts(
00698             &candidate_possible_firsts,
00699             &candidate_possible_lasts);
00700         // No possible first, failing.
00701         if (candidate_possible_firsts.size() == 0) {
00702           s->Fail();
00703         }
00704         // Only 1 candidate, and non optional: ranking without branching.
00705         if (candidate_possible_firsts.size() == 1 &&
00706             candidate_sequence->Interval(
00707                 candidate_possible_firsts.back())->MustBePerformed()) {
00708           *best_sequence = candidate_sequence;
00709           best_possible_firsts_ = candidate_possible_firsts;
00710           return true;
00711         }
00712 
00713         all_candidates.push_back(i);
00714         all_possible_firsts.push_back(candidate_possible_firsts);
00715       }
00716     }
00717     if (all_candidates.empty()) {
00718       return false;
00719     }
00720     const int chosen = s->Rand32(all_candidates.size());
00721     *best_sequence = sequences_[all_candidates[chosen]];
00722     best_possible_firsts_ = all_possible_firsts[chosen];
00723     return true;
00724   }

00726   bool FindSequenceVar(Solver* const s,
00727                        SequenceVar** const best_sequence) const {
00728     switch (strategy_) {
00729       case Solver::SEQUENCE_DEFAULT:
00730       case Solver::SEQUENCE_SIMPLE:
