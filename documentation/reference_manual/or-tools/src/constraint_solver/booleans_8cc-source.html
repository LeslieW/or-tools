00087   void VariableBound(int index) {
00088     if (minisat_trail_.Value() < minisat_.decisionLevel()) {
00089       VLOG(2) << "After failure, minisat_trail = " << minisat_trail_.Value()
00090               << ", minisat decision level = " << minisat_.decisionLevel();
00091       minisat_.cancelUntil(minisat_trail_.Value());
00092       CHECK_EQ(minisat_trail_.Value(), minisat_.decisionLevel());
00093       if (backjump_ && backtrack_level_ != -1) {
00094         if (minisat_trail_.Value() > backtrack_level_) {
00095           VLOG(2) << "  - learnt backtrack: " << minisat_trail_.Value()
00096                   << "/" << backtrack_level_;
00097           solver()->Fail();
00098         } else {
00099           backtrack_level_ = -1;
00100         }
00101       }
00102     }
00103     VLOG(2) << "VariableBound: " << vars_[index]->DebugString();
00104     const Minisat::Var var(index);
00105     const int internal_value = toInt(minisat_.value(var));
00106     const int64 var_value = vars_[index]->Value();
00107     if (internal_value != 2 && var_value != internal_value) {
00108       VLOG(2) << "  - internal value = " << internal_value << ", failing";
00109       solver()->Fail();
00110     }
00111     Minisat::Lit lit = Minisat::mkLit(var, var_value);
00112     VLOG(2) << "  - enqueue lit = " << Minisat::toInt(lit)
00113             << " at depth " << minisat_trail_.Value();
00114     backtrack_level_ = minisat_.propagateOneLiteral(lit);
00115     if (backtrack_level_ >= 0) {
00116       VLOG(2) << "  - failure detected, should backtrack to "
00117               << backtrack_level_;
00118       solver()->Fail();
00119     } else {
00120       minisat_trail_.SetValue(solver(), minisat_.decisionLevel());
00121       for (int i = 0; i < minisat_.touched_variables_.size(); ++i) {
00122         const Minisat::Lit lit = minisat_.touched_variables_[i];
00123         const int var = Minisat::var(lit);
00124         const bool assigned_bool = Minisat::sign(lit);
00125         VLOG(2) << "  - var " << var << " was assigned to " << assigned_bool;
00126         demons_[var]->inhibit(solver());
00127         vars_[var]->SetValue(assigned_bool);
00128       }
00129     }
00130   }
