00085   IntExpr* right = NULL;

00392 IntVar* Solver::MakeIsEqualCstVar(IntExpr* const var, int64 value) {
00393   IntExpr* left = NULL;
00394   IntExpr* right = NULL;
00395   if (IsADifference(var, &left, &right)) {
00396     return MakeIsEqualVar(left, MakeSum(right, value));
00397   }
00398   if (var->Max() - var->Min() == 1) {
00399     if (value == var->Min()) {
00400       return MakeDifference(value + 1, var)->Var();
00401     } else if (value == var->Max()) {
00402       return MakeSum(var, -value + 1)->Var();
00403     } else {
00404       return MakeIntConst(0);
00405     }
00406   }
00407   if (var->IsVar()) {
00408     return var->Var()->IsEqual(value);
00409   } else {
00410     IntVar* const boolvar =
00411         MakeBoolVar(StringPrintf("Is(%s == %" GG_LL_FORMAT "d)",
00412                                  var->DebugString().c_str(),
00413                                  value));

00950   virtual void Post() {
00951     demon_ = solver()->MakeConstraintInitialPropagateCallback(this);
00952     if (!var_->Bound()) {
00953       var_->WhenDomain(demon_);
00954     }
00955     if (!boolvar_->Bound()) {
00956       boolvar_->WhenBound(demon_);
00957     }
00958   }

00960   virtual void InitialPropagate() {
00961     if (boolvar_->Min() == 1LL) {
00962       demon_->inhibit(solver());
00963       var_->SetValues(values_.RawData(), values_.size());
00964     } else if (boolvar_->Max() == 1LL) {
00965       int support = support_pos_.Value();
00966       const int64 vmin = var_->Min();
00967       const int64 vmax = var_->Max();
00968       while (support < values_.size() &&
00969              (values_[support] < vmin ||
00970               !var_->Contains(values_[support]))) {
00971         if (values_[support] <= vmax) {
00972           support++;
00973         } else {
00974           support = values_.size();
00975         }
00976       }
00977       support_pos_.SetValue(solver(), support);
00978       if (support >= values_.size()) {
00979         demon_->inhibit(solver());
00980         boolvar_->SetValue(0LL);
00981       } else if (var_->Bound()) {
00982         demon_->inhibit(solver());
00983         boolvar_->SetValue(1LL);
00984       }
00985     } else {  // boolvar_ set to 0.
00986       demon_->inhibit(solver());
00987       var_->RemoveValues(values_.RawData(), values_.size());
00988     }
00989   }

00991   virtual string DebugString() const {
00992     return StringPrintf("IsMemberCt(%s, %s, %s)",
00993                         var_->DebugString().c_str(),
00994                         values_.DebugString().c_str(),
00995                         boolvar_->DebugString().c_str());
00996   }
