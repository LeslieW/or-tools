00049 void MPConstraint::SetCoefficient(const MPVariable* const var, double coeff) {
00050   CHECK_NOTNULL(var);
00051   if (coeff == 0.0) {
00052     hash_map<const MPVariable*, double>::iterator it = coefficients_.find(var);
00053     // If setting a coefficient to 0 when this coefficient did not
00054     // exist or was already 0, do nothing: skip
00055     // interface_->SetCoefficient() and do not store a coefficient in
00056     // the map.  Note that if the coefficient being set to 0 did exist
00057     // and was not 0, we do have to keep a 0 in the coefficients_ map,
00058     // because the extraction of the constraint might rely on it,
00059     // depending on the underlying solver.
00060     if (it != coefficients_.end() && it->second != 0.0) {
00061       const double old_value = it->second;
00062       it->second = 0.0;
00063       interface_->SetCoefficient(this, var, 0.0, old_value);
00064     }
00065     return;
00066   }
00067   std::pair<hash_map<const MPVariable*, double>::iterator, bool>
00068       insertion_result =
00069       coefficients_.insert(std::make_pair(var, coeff));
00070   const double old_value =
00071       insertion_result.second ? 0.0 : insertion_result.first->second;
00072   insertion_result.first->second = coeff;
00073   interface_->SetCoefficient(this, var, coeff, old_value);
00074 }

00361 bool MPSolver::CheckNameValidity(const string& name) {
00362   if (name.empty()) {
00363     LOG(DFATAL)
00364         << "Bug! CheckNameValidity() should never encounter an empty name.";
00365   }
00366   // Allow names that conform to the LP and MPS format.
00367   const int kMaxNameLength = 255;
00368   if (name.size() > kMaxNameLength) {
00369     LOG(WARNING) << "Invalid name " << name
00370                  << ": length > " << kMaxNameLength << "."
00371                  << " Will be unable to write model to file.";
00372     return false;
00373   }
00374   if (name.find_first_of(" +-*<>=:\\") != string::npos) {
00375     LOG(WARNING) << "Invalid name " << name
00376                  << ": contains forbidden character: +-*<>=:\\ space."
00377                  << " Will be unable to write model to file.";
00378     return false;
00379   }
00380   size_t first_occurrence = name.find_first_of(".0123456789");
00381   if (first_occurrence != string::npos && first_occurrence == 0) {
00382     LOG(WARNING) << "Invalid name " << name
00383                  << ": first character should not be . or a number."
00384                  << " Will be unable to write model to file.";
00385     return false;
00386   }
00387   return true;
00388 }

00406 MPSolver::LoadStatus MPSolver::LoadModel(const MPModelProto& input_model) {
00407   hash_map<string, MPVariable*> variables;
00408   for (int i = 0; i < input_model.variables_size(); ++i) {
00409     const MPVariableProto& var_proto = input_model.variables(i);
00410     const string& id = var_proto.id();
00411     if (!ContainsKey(variables, id)) {
00412       MPVariable* variable = MakeNumVar(var_proto.lb(), var_proto.ub(), id);
00413       variable->SetInteger(var_proto.integer());
00414       variables[id] = variable;
00415     } else {
00416       return MPSolver::DUPLICATE_VARIABLE_ID;
00417     }
00418   }
00419   // To detect duplicate variables in each constraint, and in the objective.
00420   hash_set<MPVariable*> tmp_variable_set;
00421   for (int i = 0; i < input_model.constraints_size(); ++i) {
00422     tmp_variable_set.clear();
00423     const MPConstraintProto& ct_proto = input_model.constraints(i);
00424     const string& ct_id = ct_proto.has_id() ? ct_proto.id() : "";
00425     MPConstraint* const ct = MakeRowConstraint(ct_proto.lb(),
00426                                                ct_proto.ub(),
00427                                                ct_id);
00428     for (int j = 0; j < ct_proto.terms_size(); ++j) {
00429       const MPTermProto& term_proto = ct_proto.terms(j);
00430       const string& id = term_proto.variable_id();
00431       MPVariable* variable = FindPtrOrNull(variables, id);
00432       if (variable == NULL) {
00433         return MPSolver::UNKNOWN_VARIABLE_ID;
00434       }
00435       if (!tmp_variable_set.insert(variable).second) {
00436         LOG(WARNING)
00437             << "Multiple terms on the same variable within the same"
00438             << " constraint; keeping only the last term into account.\n"
00439             << "Variable: " << variable->name()
00440             << ", in Constraint: " << ct_id
00441             << ", in Model '" << input_model.name() << "'.";
00442       }
00443       ct->SetCoefficient(variable, term_proto.coefficient());
00444     }
00445   }
00446   tmp_variable_set.clear();
00447   for (int i = 0; i < input_model.objective_terms_size(); ++i) {
00448     const MPTermProto& term_proto = input_model.objective_terms(i);
00449     const string& id = term_proto.variable_id();
00450     MPVariable* variable = FindPtrOrNull(variables, id);
00451     if (variable == NULL) {
00452       return MPSolver::UNKNOWN_VARIABLE_ID;
00453     }
00454     if (!tmp_variable_set.insert(variable).second) {
00455       LOG(WARNING)
00456           << "Multiple terms on the same variable within the"
00457           << " objective; keeping only the last term into account.\n"
00458           << "Variable: " << variable->name()
00459           << ", in Model '" << input_model.name() << "'.";
00460     }
00461     SetObjectiveCoefficient(variable, term_proto.coefficient());
00462   }
00463   SetOptimizationDirection(input_model.maximize());
00464   if (input_model.has_objective_offset()) {
00465     MutableObjective()->SetOffset(input_model.objective_offset());
00466   }
00467   return MPSolver::NO_ERROR;
00468 }

00470 void MPSolver::ExportModel(MPModelProto* output_model) const {
00471   CHECK_NOTNULL(output_model);
00472   if (output_model->variables_size() > 0 ||
00473       output_model->has_maximize() ||
00474       output_model->objective_terms_size() > 0 ||
00475       output_model->constraints_size() > 0 ||

00483     output_model->clear_constraints();

00487   // Variables
00488   for (int j = 0; j < variables_.size(); ++j) {
00489     const MPVariable* const var = variables_[j];

00491     DCHECK(!var->name().empty());
00492     variable_proto->set_id(var->name());
00493     variable_proto->set_lb(var->lb());
00494     variable_proto->set_ub(var->ub());

00500     MPConstraint* const constraint = constraints_[i];
00501     MPConstraintProto* const constraint_proto = output_model->add_constraints();
00502     // Constraint names need to be non-empty.
00503     DCHECK(!constraint->name().empty());
00504     constraint_proto->set_id(constraint->name());
00505     constraint_proto->set_lb(constraint->lb());
00506     constraint_proto->set_ub(constraint->ub());
00507     for (ConstIter<hash_map<const MPVariable*, double> >
00508              it(constraint->coefficients_);
00509          !it.at_end(); ++it) {
00510       const MPVariable* const var = it->first;
00511       const double coef = it->second;
00512       MPTermProto* const term = constraint_proto->add_terms();
00513       term->set_variable_id(var->name());
00514       term->set_coefficient(coef);
00515     }
00516   }
00517 
00518   // Objective
00519   for (hash_map<const MPVariable*, double>::const_iterator it =
00520            objective_->coefficients_.begin();

00522        ++it) {
00523     const MPVariable* const var = it->first;
00524     const double coef = it->second;
00525     MPTermProto* const term = output_model->add_objective_terms();
00526     term->set_variable_id(var->name());

00528   }
00529   output_model->set_maximize(Objective().maximization());
00530   output_model->set_objective_offset(Objective().offset());

00734 int MPSolver::ComputeMaxConstraintSize(int min_constraint_index,
00735                                        int max_constraint_index) const {
00736   int max_constraint_size = 0;
00737   DCHECK_GE(min_constraint_index, 0);
00738   DCHECK_LE(max_constraint_index, constraints_.size());
00739   for (int i = min_constraint_index; i < max_constraint_index; ++i) {
00740     MPConstraint* const ct = constraints_[i];
00741     if (ct->coefficients_.size() > max_constraint_size) {
00742       max_constraint_size = ct->coefficients_.size();
00743     }
00744   }
00745   return max_constraint_size;
00746 }

00748 bool MPSolver::HasInfeasibleConstraints() const {
00749   bool hasInfeasibleConstraints = false;
00750   for (int i = 0; i < constraints_.size(); ++i) {
00751     if (constraints_[i]->lb() > constraints_[i]->ub()) {
00752       LOG(WARNING) << "Constraint " << constraints_[i]->name()
00753                    << " (" << i << ") has contradictory bounds:"
00754                    << " lower bound = " << constraints_[i]->lb()
00755                    << " upper bound = " << constraints_[i]->ub();
00756       hasInfeasibleConstraints = true;
00757     }
00758   }
00759   return hasInfeasibleConstraints;
00760 }

00818 void MPSolverInterface::ExtractModel() {
00819   switch (sync_status_) {
00820     case MUST_RELOAD: {
00821       ExtractNewVariables();
00822       ExtractNewConstraints();
00823       ExtractObjective();
00824 
00825       last_constraint_index_ = solver_->constraints_.size();
00826       last_variable_index_ = solver_->variables_.size();
00827       sync_status_ = MODEL_SYNCHRONIZED;
00828       break;
00829     }
00830     case MODEL_SYNCHRONIZED: {
00831       // Everything has already been extracted.
00832       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00833       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00834       break;
00835     }
00836     case SOLUTION_SYNCHRONIZED: {
00837       // Nothing has changed since last solve.
00838       CHECK_EQ(last_constraint_index_, solver_->constraints_.size());
00839       CHECK_EQ(last_variable_index_, solver_->variables_.size());
00840       break;
00841     }
00842   }
00843 }

00847   last_constraint_index_ = 0;
00848   last_variable_index_ = 0;

00850     MPVariable* const var = solver_->variables_[j];
00851     var->set_index(kNoIndex);

00853   for (int i = 0; i < solver_->constraints_.size(); ++i) {
00854     MPConstraint* const ct = solver_->constraints_[i];
