00023 void StringAppendV(std::string* const dst,
00024                    const char* const format,
00025                    va_list ap) {
00026   // First try with a small fixed size buffer
00027   char space[1024];
00028 
00029   // It's possible for methods that use a va_list to invalidate
00030   // the data in it upon use.  The fix is to make a copy
00031   // of the structure before using it and use that copy instead.
00032   va_list backup_ap;
00033 #if defined(_MSC_VER)
00034   backup_ap = ap;
00035 #else
00036   va_copy(backup_ap, ap);
00037 #endif
00038   int result = vsnprintf(space, sizeof(space), format, backup_ap);
00039   va_end(backup_ap);
00040 
00041   if ((result >= 0) && (result < sizeof(space))) {
00042     // It fit
00043     dst->append(space, result);
00044     return;
00045   }
00046 
00047   // Repeatedly increase buffer size until it fits
00048   int length = sizeof(space);
00049   while (true) {
00050     if (result < 0) {
00051       // Older behavior: just try doubling the buffer size
00052       length *= 2;
00053     } else {
00054       // We need exactly "result+1" characters
00055       length = result+1;
00056     }
00057     char* const buf = new char[length];
00058 
00059     // Restore the va_list before we use it again
00060 #if defined(_MSC_VER)
00061   backup_ap = ap;
00062 #else
00063     va_copy(backup_ap, ap);
00064 #endif
00065     result = vsnprintf(buf, length, format, backup_ap);
00066     va_end(backup_ap);
00067 
00068     if ((result >= 0) && (result < length)) {
00069       // It fit
00070       dst->append(buf, result);
00071       delete[] buf;
00072       return;
00073     }
00074     delete[] buf;
00075   }
00076 }
