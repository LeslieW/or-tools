00022 // Conversions to a 32-bit integer can pass the call to strto[u]l on 32-bit
00023 // platforms, but need a little extra work on 64-bit platforms.
00024 inline int32 strto32(const char *nptr, char **endptr, int base) {
00025   return strtol(nptr, endptr, base);  // NOLINT
00026 }
00027 
00028 inline uint32 strtou32(const char *nptr, char **endptr, int base) {
00029   return strtoul(nptr, endptr, base);  // NOLINT
00030 }
00031 
00032 // For now, long long is 64-bit on all the platforms we care about, so these
00033 // functions can simply pass the call to strto[u]ll.
00034 inline int64 strto64(const char *nptr, char **endptr, int base) {
00035 #if defined(_MSC_VER)
00036   return _strtoi64(nptr, endptr, base);  // NOLINT
00037 #else
00038   return strtoll(nptr, endptr, base);  // NOLINT
00039 #endif  // _MSC_VER
00040 }
00041 
00042 inline uint64 strtou64(const char *nptr, char **endptr, int base) {
00043 #if defined(_MSC_VER)
00044   return _strtoui64(nptr, endptr, base);  // NOLINT
00045 #else
00046   return strtoull(nptr, endptr, base);  // NOLINT
00047 #endif  // _MSC_VER
00048 }

00050 // Although it returns an int, atoi() is implemented in terms of strtol, and
00051 // so has differing overflow and underflow behavior.  atol is the same.
00052 inline int32 atoi32(const char *nptr) {
00053   return strto32(nptr, NULL, 10);
00054 }
00055 
00056 inline int64 atoi64(const char *nptr) {
00057   return strto64(nptr, NULL, 10);
00058 }
00059 
00060 // Convenience versions of the above that take a string argument.
00061 inline int32 atoi32(const std::string &s) {

00064 
00065 inline int64 atoi64(const std::string &s) {
