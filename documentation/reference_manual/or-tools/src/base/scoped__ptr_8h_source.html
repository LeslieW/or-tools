00039 // A scoped_ptr<T> is like a T*, except that the destructor of scoped_ptr<T>
00040 // automatically deletes the pointer it holds (if any).
00041 // That is, scoped_ptr<T> owns the T object that it points to.
00042 // Like a T*, a scoped_ptr<T> may hold either NULL or a pointer to a T object.
00043 //
00044 // The size of a scoped_ptr is small:
00045 // sizeof(scoped_ptr<C>) == sizeof(C*)
00046 template <class C>
00047 class scoped_ptr {
00048  public:
00049   // The element type
00050   typedef C element_type;
00051 
00052   // Constructor.  Defaults to intializing with NULL.
00053   // There is no way to create an uninitialized scoped_ptr.
00054   // The input parameter must be allocated with new.
00055   explicit scoped_ptr(C* p = NULL) : ptr_(p) { }
00056 
00057   // Destructor.  If there is a C object, delete it.
00058   // We don't need to test ptr_ == NULL because C++ does that for us.
00059   ~scoped_ptr() {
00060     enum { type_must_be_complete = sizeof(C) };
00061     delete ptr_;
00062   }
00063 
00064   // Reset.  Deletes the current owned object, if any.
00065   // Then takes ownership of a new object, if given.
00066   // this->reset(this->get()) works.
00067   void reset(C* p = NULL) {
00068     if (p != ptr_) {
00069       enum { type_must_be_complete = sizeof(C) };
00070       delete ptr_;
00071       ptr_ = p;
00072     }
00073   }
00074 
00075   // Accessors to get the owned object.
00076   // operator* and operator-> will assert() if there is no current object.
00077   C& operator*() const {
00078     assert(ptr_ != NULL);
00079     return *ptr_;
00080   }
00081   C* operator->() const  {
00082     assert(ptr_ != NULL);
00083     return ptr_;
00084   }
00085   C* get() const { return ptr_; }
00086 
00087   // Comparison operators.
00088   // These return whether two scoped_ptr refer to the same object, not just to
00089   // two different but equal objects.
00090   bool operator==(C* p) const { return ptr_ == p; }
00091   bool operator!=(C* p) const { return ptr_ != p; }
00092 
00093   // Swap two scoped pointers.
00094   void swap(scoped_ptr& p2) {
00095     C* tmp = ptr_;
00096     ptr_ = p2.ptr_;
00097     p2.ptr_ = tmp;
00098   }
00099 
00100   // Release a pointer.
00101   // The return value is the current pointer held by this object.
00102   // If this object holds a NULL pointer, the return value is NULL.
00103   // After this operation, this object will hold a NULL pointer,
00104   // and will not own the object any more.
00105   C* release() {
00106     C* retVal = ptr_;
00107     ptr_ = NULL;
00108     return retVal;
00109   }
00110 
00111  private:
00112   C* ptr_;
00113 
00114   // friend class that can access copy ctor (although if it actually
00115   // calls a copy ctor, there will be a problem) see below
00116   friend scoped_ptr<C> make_scoped_ptr<C>(C *p);
00117 
00118   // Forbid comparison of scoped_ptr types.  If C2 != C, it totally doesn't
00119   // make sense, and if C2 == C, it still doesn't make sense because you should
00120   // never have the same object owned by two different scoped_ptrs.
00121   template <class C2> bool operator==(scoped_ptr<C2> const& p2) const;
00122   template <class C2> bool operator!=(scoped_ptr<C2> const& p2) const;
00123 
00124   DISALLOW_COPY_AND_ASSIGN(scoped_ptr);
00125 };

00143 template <class C>
00144 scoped_ptr<C> make_scoped_ptr(C *p) {
00145   // This does nothing but to return a scoped_ptr of the type that the passed
00146   // pointer is of.  (This eliminates the need to specify the name of T when
00147   // making a scoped_ptr that is used anonymously/temporarily.)  From an
00148   // access control point of view, we construct an unnamed scoped_ptr here
00149   // which we return and thus copy-construct.  Hence, we need to have access
00150   // to scoped_ptr::scoped_ptr(scoped_ptr const &).  However, it is guaranteed
00151   // that we never actually call the copy constructor, which is a good thing
00152   // as we would call the temporary's object destructor (and thus delete p)
00153   // if we actually did copy some object, here.
00154   return scoped_ptr<C>(p);
00155 }
