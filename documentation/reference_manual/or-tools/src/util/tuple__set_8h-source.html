00014 // Set of integer tuples (fixed-size arrays, all of the same size) with
00015 // a basic API.
00016 // It supports several types of integer arrays transparently, with an
00017 // inherent storage based on int64 arrays.
00018 //
00019 // The key feature is the "lazy" copy:
00020 // - Copying an IntTupleSet won't actually copy the data right away; we
00021 //   will just have several IntTupleSet pointing at the same data.
00022 // - Modifying an IntTupleSet which shares his data with others
00023 //   will create a new, modified instance of the data payload, and make
00024 //   the IntTupleSet point to that new data.
00025 // - Modifying an IntTupleSet that doesn't share its data with any other
00026 //   IntTupleSet will modify the data directly.
00027 // Therefore, you don't need to use const IntTupleSet& in methods. Just do:
00028 // void MyMethod(IntTupleSet tuple_set) { ... }
00029 //
00030 // This class is thread hostile as the copy and reference counter are
00031 // not protected by a mutex.

00173 template <class T> int IntTupleSet::Data::Insert(const std::vector<T>& tuple) {
00174   DCHECK(arity_ == 0 || flat_tuples_.size() % arity_ == 0);
00175   CHECK_EQ(arity_, tuple.size());
00176   DCHECK_EQ(1, num_owners_);
00177   if (!Contains(tuple)) {
00178     const int index = NumTuples();
00179     const int offset = flat_tuples_.size();
00180     flat_tuples_.resize(offset + arity_);
00181     // On mac os X, using this instead of push_back gives a 10x speedup!
00182     for (int i = 0; i < arity_; ++i) {
00183       flat_tuples_[offset + i] = tuple[i];
00184     }
00185     const int64 fingerprint = Fingerprint(tuple);
00186     tuple_fprint_to_index_[fingerprint].push_back(index);
00187     return index;
00188   } else {
00189     return -1;
00190   }
00191 }

00193 template <class T> bool IntTupleSet::Data::Contains(
00194     const std::vector<T>& candidate) const {
00195   if (candidate.size() != arity_) {
00196     return false;
00197   }
00198   const int64 fingerprint = Fingerprint(candidate);
00199   if (ContainsKey(tuple_fprint_to_index_, fingerprint)) {
00200     const std::vector<int>& indices = FindOrDie(tuple_fprint_to_index_, fingerprint);
00201     for (int i = 0; i < indices.size(); ++i) {
00202       const int tuple_index = indices[i];
00203       for (int j = 0; j < arity_; ++j) {
00204         if (candidate[j] != flat_tuples_[tuple_index * arity_ + j]) {
00205           return false;
00206         }
00207       }
00208       return true;
00209     }
00210   }
00211   return false;
00212 }

00215     const std::vector<T>& tuple) const {
00216   switch (arity_) {
00217     case 0:
00218       return 0;
00219     case 1:
00220       return tuple[0];
00221     case 2: {
00222       uint64 x = tuple[0];

00225       mix(x, y, z);

00228     default: {

00232         uint64 z = tuple[i];

00373 inline IntTupleSet IntTupleSet::SortedByColumn(int col) const {
00374   std::vector<IndexValue> keys;

00377     keys.push_back(IndexValue(index, data_->Value(index, col)));
00378   }

00380   const int arity = data_->Arity();

00382   for (int i = 0; i < keys.size(); ++i) {
00383     const int64* tuple_ptr = data_->RawData() + keys[i].index * arity;
00384     sorted.Insert(std::vector<int64>(tuple_ptr, tuple_ptr + arity));
